116  void runAllChildren(OArgs... args) {
117  utils::NullableRef<HierarFlowCallee<O>> succ_ref = std::nullopt;
118  auto &succ_nodes = UnwrapCurrentLinkedNodeRef().succ_nodes;
119  if (!succ_nodes.empty()) {
120  succ_ref = *succ_nodes[0];
121  }
122 
123  while (succ_ref) {
124  auto &succ = succ_ref.value().get();
125  /*
126  * FIXME: Although args should be std::forwarded to keep movable arguments
127  * to be movable, in current implementation, it causes AFL to move an
128  * unexpected values.
129  */
130  auto next_succ_ref = succ(args...);
131  succ_ref.swap(next_succ_ref);
132  }
133  }
