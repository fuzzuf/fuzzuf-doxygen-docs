296  template <typename Node, typename... Args>
297  void operator()(const char *, int, const char *node_name, const Node &,
298  Checkpoint checkpoint, const Args &...) {
299  auto now = std::chrono::high_resolution_clock::now();
300  if (checkpoint == Checkpoint::enter) {
301  if (!call_stack.empty()) {
302  summary[call_stack.back().name].exclusive +=
303  std::chrono::duration_cast<std::chrono::microseconds>(
304  now - call_stack.back().exclusive)
305  .count();
306  }
307  call_stack.emplace_back(node_name, now, now);
308  } else if (checkpoint == Checkpoint::leave) {
309  if (!call_stack.empty()) {
310  if (call_stack.back().name != node_name) {
311  throw exceptions::unexpected_leave_event();
312  }
313  auto existing = summary.find(call_stack.back().name);
314  if (existing != summary.end()) {
315  existing->second.exclusive +=
316  std::chrono::duration_cast<std::chrono::microseconds>(
317  now - call_stack.back().exclusive)
318  .count();
319  existing->second.inclusive +=
320  std::chrono::duration_cast<std::chrono::microseconds>(
321  now - call_stack.back().inclusive)
322  .count();
323  existing->second.count += 1u;
324  } else {
325  summary.insert(std::make_pair(
326  call_stack.back().name,
327  SumT{std::chrono::duration_cast<std::chrono::microseconds>(
328  now - call_stack.back().exclusive)
329  .count(),
330  std::chrono::duration_cast<std::chrono::microseconds>(
331  now - call_stack.back().inclusive)
332  .count(),
333  1u}));
334  }
335  call_stack.pop_back();
336  }
337  if (!call_stack.empty()) {
338  call_stack.back().exclusive = now;
339  }
340  }
341  }
