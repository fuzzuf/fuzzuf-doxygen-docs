33 template <>
34 AFLMidCalleeRef<AFLplusplusState>
35 ApplyDetMutsTemplate<AFLplusplusState>::operator()(
36  std::shared_ptr<AFLplusplusTestcase> testcase) {
37  // We no longer modify this testcase.
38  // So we can reload the file with mmap.
39  testcase->input->LoadByMmap(); // no need to Unload
40 
41  if (!state.orig_perf && state.queued_paths > 10) {
42  this->SetResponseValue(true);
43  return abandon_entry;
44  }
45 
46  /* Skip right away if -d is given, if it has not been chosen sufficiently
47  often to warrant the expensive deterministic stage (fuzz_level), or
48  if it has gone through deterministic testing in earlier, resumed runs
49  (passed_det). */
50 
51  if (state.skip_deterministic ||
52  ((!testcase->passed_det) &&
53  state.orig_perf <
54  (testcase->depth * 30 <= option::GetHavocMaxMult(state) * 100
55  ? testcase->depth * 30
56  : option::GetHavocMaxMult(state) * 100)) ||
57  testcase->passed_det) {
58  state.doing_det = false;
59  return this->GoToDefaultNext();
60  }
61 
62  /* Skip deterministic fuzzing if exec path checksum puts this out of scope
63  for this master instance. */
64 
65  if (state.orig_perf == 0 && state.queued_paths > 10) {
66  this->SetResponseValue(true);
67  return abandon_entry;
68  }
69 
70  state.doing_det = true;
71 
72  auto mutator = AFLMutatorTemplate<AFLplusplusState>(*testcase->input, state);
73 
74  state.stage_val_type = option::STAGE_VAL_NONE;
75 
76  // this will be required in dictionary construction and eff_map construction
77  state.queue_cur_exec_cksum = testcase->exec_cksum;
78 
79  // call deterministic mutations
80  // if they return true, then we should go to abandon_entry
81  auto should_abandon_entry = this->CallSuccessors(mutator);
82 
83  if (should_abandon_entry) {
84  this->SetResponseValue(true);
85  return abandon_entry;
86  }
87 
88  // NOTE: "if (!testcase->passed_det)" seems unnecessary to me
89  // because passed_det == 0 always holds here
90  if (!testcase->passed_det) state.MarkAsDetDone(*testcase);
91 
92  return this->GoToDefaultNext();
93 }

98 AbandonEntryTemplate<AFLplusplusState>::operator()(

115  return this->GoToDefaultNext();

192 void ComputeWeightVector(AFLplusplusState &state, std::vector<double> &vw) {
193  u32 queued_items = state.case_queue.size();
194 
195  double avg_exec_us = 0.0, avg_bitmap_size = 0.0, avg_top_size = 0.0;
196  for (auto &tc : state.case_queue) {
197  avg_exec_us += tc->exec_us;
198  avg_bitmap_size += std::log(tc->bitmap_size);
199  avg_top_size += tc->tc_ref;
200  }
201  avg_exec_us /= queued_items;
202  avg_bitmap_size /= queued_items;
203  avg_top_size /= queued_items;
204 
205  std::transform(state.case_queue.begin(), state.case_queue.end(),
206  std::back_inserter(vw), [&](auto &tc) {
207  return ComputeWeight(state, *tc, avg_exec_us,
208  avg_bitmap_size, avg_top_size);
209  });
210 }

212 double ComputeWeight(const AFLplusplusState &state,
213  const AFLplusplusTestcase &testcase,
214  const double &avg_exec_us, const double &avg_bitmap_size,
215  const double &avg_top_size) {
216  double weight = 1.0;
217 
218  u32 hits = state.n_fuzz[testcase.n_fuzz_entry];
219  if (likely(hits)) {
220  weight *= std::log10(hits) + 1;
221  }
222 
223  weight *= (avg_exec_us / testcase.exec_us);
224  weight *= (std::log(testcase.bitmap_size) / avg_bitmap_size);
225  weight *= (1 + (testcase.tc_ref / avg_top_size));
226  if (unlikely(testcase.favored)) {
227  weight *= 5;
228  }
229  if (unlikely(!(const_cast<AFLplusplusTestcase &>(testcase).WasFuzzed()))) {
230  weight *= 2;
231  }
232 
233  return weight;
234 }
