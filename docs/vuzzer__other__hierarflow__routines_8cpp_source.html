75  ExitStatusFeedback exit_status;
76  std::vector<u64> all_bb;
77  std::vector<boost::dynamic_bitset<>> bb_sets;
78  if (state.loop_cnt > 40 && state.loop_cnt % state.bbslide == 0) {
79  DEBUG("Starting EHB calculation\n");
86  for (const auto& testcase : state.seed_queue) {
87  testcase->input->Load();
88  auto inp_feed = state.RunExecutor(testcase->input->GetBuf(), testcase->input->GetLen(), exit_status);
89  std::map<u64, u32> bb_cov;
90  boost::dynamic_bitset<> bb_set;
91  vuzzer::util::ParseBBCov(inp_feed, bb_cov);
92  vuzzer::util::DictToBitsWithKeys(bb_cov, all_bb, bb_set);
93  bb_sets.emplace_back(bb_set);
94  testcase->input->Unload();
95  }
96 
97  /* Count frequencies of every bbs */
98  const u32 ratio = (state.setting->ehb_bb_ratio/100) * state.setting->pop_size;
99  for (size_t i = 0; i < all_bb.size(); i++) {
100  size_t check_idx = all_bb.size() - 1 - i;
101  u32 bb_appear = 0;
102  for (auto &bb_set : bb_sets) {
103  if (bb_set.size() <= check_idx) continue;
104  bb_appear += bb_set[check_idx];
105  }
106  if (bb_appear > ratio && state.good_bbs.find(all_bb[i]) != state.good_bbs.end()) {
107  state.ehb_inc.insert(all_bb[i]);
108  }
109  /* TODO: Dump all EHB addrs to file */
110  }
111  }
112  CallSuccessors();
113  return GoToDefaultNext();
114 }

121 VUzzerMidCalleeRef ExecutePUT::operator()(void) {
122  DEBUG("ExecutePUT pending(%zu)\n", state.pending_queue.size());
123  ExitStatusFeedback exit_status;
124  /* Execute all inputs from pending_queue */
125  for (const auto& testcase : state.pending_queue) {
126  testcase->input->Load();
127 
128  auto inp_feed = state.RunExecutor(testcase->input->GetBuf(), testcase->input->GetLen(), exit_status);
129 
130  /* Calculate fitness score in child node (i.e. UpdateFitness method) */
131  auto score = CallSuccessors(testcase, inp_feed);
132  testcase->fitness = score;
133  DEBUG("Score %s : %lf", testcase->input->GetPath().c_str(), score);
134 
135  /* If we encount crash, then triage it. */
136  /* TODO: Consider other reasons? */
137  if (exit_status.exit_reason == PUTExitReasonType::FAULT_CRASH) {
138  std::string crash_hash = GetSHA1HashFromFile(testcase->input->GetPath().native(), testcase->input->GetLen());
139  DEBUG("Testcase %s crashed! (%s)\n", testcase->input->GetPath().c_str(), crash_hash.c_str());
140  if (state.crash_hashes.find(crash_hash) == state.crash_hashes.end()) {
141  /* VUzzer has found a new crash input:) */
142  state.crash_hashes.insert(crash_hash);
143  /* TODO: File path format */
144  std::string crash_path = fuzzuf::utils::StrPrintf("%s/crashes/id:%06llu",
145  state.setting->out_dir.c_str(),
146  state.unique_crashes);
147 
148  int fd = fuzzuf::utils::OpenFile(crash_path, O_WRONLY | O_CREAT | O_EXCL, 0600);
149  fuzzuf::utils::WriteFile(fd, testcase->input->GetBuf(), testcase->input->GetLen());
150  fuzzuf::utils::CloseFile(fd);
151  state.unique_crashes++;
152  }
153  /* TODO: Implement STOPONCRASH mode */
154  }
155  testcase->input->Unload();
156  /* Move a seed to seed_queue from pending_queue. */
157  state.seed_queue.emplace_back(testcase);
158  }
159  state.pending_queue.clear();
160  return GoToDefaultNext();
161 }

168 VUzzerMidCalleeRef ExecuteTaintPUT::operator()(void) {
169  DEBUG("ExecuteTaint taint_queue(%zu)\n", state.taint_queue.size());
170  if (state.taint_queue.empty())
171  return GoToDefaultNext();
172 
173  /* Execute PUT with seeds taken from taint_queue */
174  ExitStatusFeedback exit_status;
175  for (const auto& testcase : state.taint_queue) {
176  u64 id = testcase->input->GetID();
177  /* If we have already executed the seed then continue.
178  * FIXME: Occasionally taint executor doesn't record any taint info. Both taint_cmp_offsets and taint_lea_offsets could become empty.
179  */
180  if (state.taint_cmp_offsets.find(id) != state.taint_cmp_offsets.end() ||
181  state.taint_lea_offsets.find(id) != state.taint_lea_offsets.end())
182  continue;
183  testcase->input->Load();
184  auto inp_feed = state.RunTaintExecutor(testcase->input->GetBuf(), testcase->input->GetLen(), exit_status);
185 
186  CallSuccessors(testcase, inp_feed);
187 
188  testcase->input->Unload();
189  }
190  return GoToDefaultNext();
191 }
