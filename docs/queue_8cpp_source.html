33 void Queue::Add(Tree&& tree,
34  const std::vector<uint8_t>&& all_bits,
35  PUTExitReasonType exit_reason,
36  Context& ctx,
37  uint64_t execution_time) {
38  /* Check if all bits are zero */
39  bool all_zero = true;
40  for (size_t i = 0; i < all_bits.size(); i++) {
41  if (all_bits[i] != 0
42  && _bit_to_inputs.find(i) == _bit_to_inputs.end()) {
43  all_zero = false;
44  break;
45  }
46  }
47  if (all_zero) return;
48 
49  std::unordered_set<size_t> fresh_bits;
50  for (size_t i = 0; i < all_bits.size(); i++) {
51  if (all_bits[i]) {
52  if (_bit_to_inputs.find(i) == _bit_to_inputs.end()) {
53  fresh_bits.insert(i);
54  _bit_to_inputs[i] = {};
55  }
56 
57  _bit_to_inputs[i].push_back(_current_id);
58  }
59  }
60 
61  /* Stringify tree */
62  std::string buffer;
63  tree.UnparseTo(ctx, buffer);
64 
65  /* Create file for entry */
66  std::string filepath = fuzzuf::utils::StrPrintf(
67  "%s/queue/id:%09ld,er:%d",
68  _work_dir.c_str(), _current_id, exit_reason
69  );
70  int fd = fuzzuf::utils::OpenFile(filepath,
71  O_WRONLY | O_CREAT | O_TRUNC,
72  S_IWUSR | S_IRUSR); // 0600
73  if (fd == -1) {
74  throw exceptions::unable_to_create_file(
75  fuzzuf::utils::StrPrintf("Cannot save tree: %s", filepath.c_str()),
76  __FILE__, __LINE__
77  );
78  }
79  fuzzuf::utils::WriteFile(fd, buffer.data(), buffer.size());
80  fuzzuf::utils::CloseFile(fd);
81 
82  /* Add entry to queue */
83  auto new_item = std::make_unique<QueueItem>(
84  _current_id,
85  std::move(tree),
86  std::move(fresh_bits),
87  std::move(all_bits),
88  exit_reason,
89  execution_time
90  );
91  _inputs.emplace_back(std::move(new_item));
92 
93  /* Increment current_id */
94  if (_current_id == std::numeric_limits<size_t>::max()) {
95  _current_id = 0;
96  } else {
97  _current_id++;
98  }
99 }

106 std::unique_ptr<QueueItem> Queue::Pop() {
107  DEBUG_ASSERT (!IsEmpty());
108 
109  std::unique_ptr<QueueItem> item(std::move(_inputs.back()));
110  _inputs.pop_back();
111 
112  size_t id = item->id;
113 
114  for (auto it = _bit_to_inputs.begin(); it != _bit_to_inputs.end();) {
115  auto& [k, v] = *it;
116  UNUSED(k);
117 
118  /* Retain elements in v */
119  auto r = std::remove_if(v.begin(), v.end(),
120  [id, &v](size_t vid) {
121  if (vid == id) return true;
122  else return false;
123  });
124  v.erase(r, v.end());
125 
126  if (v.empty()) {
127  it = _bit_to_inputs.erase(it);
128  } else {
129  ++it;
130  }
131  }
132 
133  return item;
134 }

163  _work_dir.c_str(), item->id, item->exit_reason)

168  std::unordered_set<size_t> fresh_bits;

172  fresh_bits.insert(i);
