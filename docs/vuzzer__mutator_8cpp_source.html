83 u32 VUzzerMutator::GetCutPos(u32 limit) {
84  u32 cut_pos = 0;
85  if (state.taint_cmp_offsets.size() && rand() % 10 > 3) {
86  u64 id = input.GetID();
87  std::vector<std::pair<u32, std::set<u32>>> taint_choices;
88 
89  const std::set<u32> *offsets;
90 
91  if (state.taint_cmp_offsets.find(id) != state.taint_cmp_offsets.end()) {
92  offsets = &(state.taint_cmp_offsets.at(id));
93  } else {
94  std::sample(state.taint_cmp_offsets.begin(),
95  state.taint_cmp_offsets.end(),
96  std::back_inserter(taint_choices), 1,
97  std::mt19937{std::random_device{}()});
98  offsets = &(taint_choices[0].second);
99  }
100  std::vector<u32> offsets_within_range;
101  /* TODO: MOSTCOMM, RANDOMCOMM */
102  for (auto offset : *offsets)
103  if (offset < limit) offsets_within_range.emplace_back(offset);
104 
105  if (offsets_within_range.size())
106  cut_pos = offsets_within_range[rand() % offsets_within_range.size()];
107  else
108  cut_pos = limit ? rand() % limit : 0;
109  } else {
110  cut_pos = limit ? rand() % limit : 0;
111  /* TODO: MOSTCOMM */
112  }
113  DEBUG("GetCutPos %u (limit %u, taint_tag %lu)", cut_pos, limit,
114  state.taint_cmp_offsets.size());
115  return cut_pos;
116 }

189 void VUzzerMutator::ChangeBytes() {
190  if (state.taint_cmp_offsets.empty()) return;
191 
192  u64 id = input.GetID();
193  DEBUG("ChangeBytes %llu", id);
194  std::vector<std::pair<u32, std::set<u32>>> taint_choices;
195 
196  const std::set<u32> *offsets;
197  if (state.taint_cmp_offsets.find(id) != state.taint_cmp_offsets.end()) {
198  offsets = &(state.taint_cmp_offsets.at(id));
199  } else {
200  std::sample(state.taint_cmp_offsets.begin(), state.taint_cmp_offsets.end(),
201  std::back_inserter(taint_choices), 1,
202  std::mt19937{std::random_device{}()});
203  offsets = &(taint_choices[0].second);
204  }
205 
206  for (auto o : *offsets) DEBUG("0x%x, ", o);
207  // TODO: Consider MOSTCOMMON
208  if (offsets->size() > 0) {
209  std::vector<u32> off_choices;
210  std::sample(offsets->begin(), offsets->end(),
211  std::back_inserter(off_choices),
212  std::max(1UL, offsets->size() / 4),
213  std::mt19937{std::random_device{}()});
214  for (auto choice : off_choices) {
215  if (choice < len) {
216  DEBUG("Change offset %u", choice);
217  outbuf[choice] =
218  state.all_chars_dict[random() % state.all_chars_dict.size()]
219  .data[0];
220  }
221  }
222  }

232 void VUzzerMutator::ChangeRandomFull() {
233  u32 change_size = std::max(1U, fuzzuf::utils::random::Random<u32>(
234  1U, std::max(1U, (len / denominator))));
235  u32 change_pos = GetCutPos(len - change_size);
236 
237  if (state.full_bytes_dict.size() > 1) {
238  /* Insert random bytes generated from full_bytes_dict at change_pos */
239  std::vector<const dict_t *> all_dicts;
240  all_dicts.emplace_back(&state.full_bytes_dict);
241  auto rand_bytes =
242  vuzzer::util::GenerateRandomBytesFromDict(change_size, all_dicts);
243  DEBUG("Insert at [%u:%u]", change_pos, change_pos + change_size - 1);
244  Insert(change_pos, rand_bytes->data(), change_size);
245  } else if (state.unique_bytes_dict.size() > 2 && len > 3) {
246  /* Insert words chosen from unique_bytes_dict at change_pos1 and change_pos2
247  */
248  std::random_device rd;
249  std::default_random_engine eng(rd());
250  std::uniform_int_distribution<int> distr(1, len - 1);
251  auto word1 =
252  &(state.unique_bytes_dict[random() % state.unique_bytes_dict.size()]
253  .data);
254  auto word2 =
255  &(state.unique_bytes_dict[random() % state.unique_bytes_dict.size()]
256  .data);
257  u32 change_pos1 = distr(eng);
258  u32 change_pos2 = distr(eng);
259  if (change_pos1 > change_pos2) std::swap(change_pos1, change_pos2); // sort
260  DEBUG("Insert at [%u:%lu]", change_pos1, change_pos1 + word1->size());
261  DEBUG("Insert at [%u:%lu]", change_pos2, change_pos2 + word2->size());
262  Insert(change_pos1, word1->data(), word1->size());
263  Insert(change_pos2 + word1->size(), word2->data(),
264  word2->size()); // NOTICE: After first insertion, change_pos2 is
265  // shifted.
266  } else {
267  /* Replace bytes at [change_pos: change_pos+change_size) range of input
268  * buffer with random bytes geranted from all_dicts. */
269  auto rand_bytes =
270  vuzzer::util::GenerateRandomBytesFromDict(change_size, state.all_dicts);
271  DEBUG("Replace at [%u:%u]", change_pos, change_pos + change_size);
272  Replace(change_pos, rand_bytes->data(), change_size);
273  }
274 }

329 void VUzzerMutator::EliminateDoubleNull() {
330  int start_pos = fuzzuf::utils::random::Random<int>(0, len - 1);
331  u8 pattern[] = {'\0', '\0'};
332  u8 replacement[] = "AA"; // TODO: Specify it by argument
333  auto itr = outbuf + start_pos;
334  bool found = false;
335  while (itr + sizeof(pattern) < outbuf + len) {
336  if (std::equal(itr, std::next(itr, sizeof(pattern)), pattern,
337  std::next(pattern, sizeof(pattern)))) {
338  found = true;
339  break;
340  }
341  itr = std::find(itr + 1, outbuf + len, pattern[0]);
342  }
343  if (found) {
344  u32 cut_pos = std::distance(outbuf, itr);
345  Replace(cut_pos, replacement, sizeof(pattern));
346  DEBUG("EliminateNull %d (found \\0\\0 at %d)", start_pos, cut_pos);
347  } else {
348  DEBUG("EliminateNull %d", start_pos);
349  }
350 }

443 void VUzzerMutator::InsertWithOnebyteOverwrite(u32 pos, const u8 *buf,
444  u32 extra_len) {
445  u8 *new_buf = new u8[len + extra_len - 1];
446  /* Head */
447  std::memcpy(new_buf, outbuf, pos);
448 
449  /* Inserted part */
450  std::memcpy(new_buf + pos, buf, extra_len);
451

456  outbuf = new_buf;
457  len += extra_len - 1;

472 void VUzzerMutator::TaintBasedChange() {
473  u64 id = input.GetID();
474  std::vector<std::vector<u8>> values = {{0xff, 0xff, 0xff, 0xff},
475  {0xFE, 0xFF, 0xFF, 0xFF},
476  {0xFE, 0xFF},
477  {0xFF, 0xFE},
478  {0x80, 0x00, 0x00, 0x00},
479  {0x7F, 0xFF}};
480 
481  /* Choose some offsets from taint info about LEA operation. */
482  if (state.taint_lea_offsets.find(id) != state.taint_lea_offsets.end()) {
483  auto &offsets_set = state.taint_lea_offsets.at(id);
484  if (offsets_set.size() > 0) {
485  std::vector<u32> off_choices;
486  std::sample(offsets_set.begin(), offsets_set.end(),
487  std::back_inserter(off_choices),
488  std::max(1UL, offsets_set.size() / 2),
489  std::mt19937{std::random_device{}()});
490  for (auto choice : off_choices) {
491  if (choice < len) {
492  std::vector<u8> value = values[random() % values.size()];
493  InsertWithOnebyteOverwrite(choice, value.data(), value.size());
494  }
495  }
496  }
497  }
498 
499  /* Choose some offsets from taint info about CMP operation. */
500  if (state.taint_cmp_all.find(id) != state.taint_cmp_all.end()) {
501  auto &offsets_map = state.taint_cmp_all.at(id);
502  if (offsets_map.size() > 0) {
503  std::vector<std::pair<u32, std::vector<u32>>>
504  off_choices; // [(off1, [v1_1, v1_2...]), ....]
505  std::sample(offsets_map.rbegin(), offsets_map.rend(),
506  std::back_inserter(off_choices),
507  std::max(1UL, offsets_map.size() / 2),
508  std::mt19937{std::random_device{}()});
509  for (auto choice : off_choices) {
510  u32 offset = choice.first;
511  if (offset < len) {
512  // TODO: MOSTCOMMLAST
513  u32 value = choice.second.at(random() % choice.second.size());
514  // outbuf[offset] = value;//XXX
515  // Convert u32 value to bytes in a little endian
516  // manner(extract_offsetStr, get_hexStr in VUzzer)
517  // TODO: Move to Util
518  u32 size = 0;
519  u8 *bytes;
520  if (value == 0) {
521  size = 1;
522  bytes = new u8[size];
523  bytes[0] = 0;
524  } else {
525  for (u32 v = value; v; v >>= 8) size++;
526  bytes = new u8[size];
527  for (u32 v = value, i = 0; v; v >>= 8, i++) bytes[i] = v & 0xff;
528  }
529  DEBUG("Change %u", offset);
530  InsertWithOnebyteOverwrite(offset, bytes, size);
531  delete[] bytes;
532  }
533  }
534  }
535  }

548 std::pair<std::shared_ptr<exec_input::ExecInput>,

550 VUzzerMutator::SingleCrossOver(const exec_input::ExecInput &target) {
551  std::unique_ptr<exec_input::ExecInputSet> input_set(
552  new exec_input::ExecInputSet());
553  std::random_device rd;
554  std::default_random_engine eng(rd());
555  std::uniform_real_distribution<> distr(0.1, 0.6);
556  u32 len1 = len, len2 = target.GetLen();
557  u8 *buf1 = outbuf, *buf2 = target.GetBuf();
558 
559  double point = distr(eng);
560 
561  u32 cut_pos1 = point * len1;
562  u32 cut_pos2 = point * len2;
563 
564  u32 new_len1 = cut_pos1 + (len2 - cut_pos2);
565  u32 new_len2 = cut_pos2 + (len1 - cut_pos1);
566 
567  DEBUG("SingleCrossOver %u, %u", cut_pos1, cut_pos2);
568  std::unique_ptr<u8[]> new_buf1(new u8[new_len1]);
569  std::unique_ptr<u8[]> new_buf2(new u8[new_len2]);
570 
571  std::memcpy(new_buf1.get(), buf1, cut_pos1);
572  std::memcpy(new_buf1.get() + cut_pos1, buf2 + cut_pos2, len2 - cut_pos2);
573  std::memcpy(new_buf2.get(), buf2, cut_pos2);
574  std::memcpy(new_buf2.get() + cut_pos2, buf1 + cut_pos1, len1 - cut_pos1);
575 
576  auto new_seed1 = input_set->CreateOnMemory(std::move(new_buf1), new_len1);
577  auto new_seed2 = input_set->CreateOnMemory(std::move(new_buf2), new_len2);
578 
579  return std::make_pair(new_seed1, new_seed2);
580 }

586 std::pair<std::shared_ptr<exec_input::ExecInput>,
587  std::shared_ptr<exec_input::ExecInput>>
588 VUzzerMutator::DoubleCrossOver(const exec_input::ExecInput &target) {
589  std::unique_ptr<exec_input::ExecInputSet> input_set(
590  new exec_input::ExecInputSet());
591  std::random_device rd;
592  std::default_random_engine eng(rd());
593  std::uniform_real_distribution<> distr1(0.1, 0.3), distr2(0.6, 0.8);
594 
595  u32 len1 = len, len2 = target.GetLen();
596  u8 *buf1 = outbuf, *buf2 = target.GetBuf();
597 
598  double point1 = distr1(eng), point2 = distr2(eng);
599  u32 cut_pos11 = point1 * len1;
600  u32 cut_pos12 = point2 * len1;
601  u32 cut_pos21 = point1 * len2;
602  u32 cut_pos22 = point2 * len2;
603 
604  u32 new_len1 = cut_pos11 + (cut_pos22 - cut_pos21) + (len1 - cut_pos12);
605  u32 new_len2 = cut_pos21 + (cut_pos12 - cut_pos11) + (len2 - cut_pos22);
606 
607  std::unique_ptr<u8[]> new_buf1(new u8[new_len1]);
608  std::unique_ptr<u8[]> new_buf2(new u8[new_len2]);
609 
610  DEBUG("DoubleCrossOver %u, %u, %u, %u", cut_pos11, cut_pos12, cut_pos21,
611  cut_pos22);
612 
613  std::memcpy(new_buf1.get(), buf1, cut_pos11);
614  std::memcpy(new_buf1.get() + cut_pos11, buf2 + cut_pos21,
615  cut_pos22 - cut_pos21);
616  std::memcpy(new_buf1.get() + cut_pos11 + (cut_pos22 - cut_pos21),
617  buf1 + cut_pos12, len1 - cut_pos12);
618
