37 /*
38  throughout this PUT, we assume there is a mapping between seeds(or input)
39  and integers (bit_seq) the mapping is just the same as little endian. for
40  example, if bit_seq is 4, then the corresponding input would be "00100000" in
41  the follwing comments, sometimes we refer seeds like "the seed corresponding
42  to 4"
43  */
44 std::unordered_map<int, u8> GetCoverage(
45  fuzzuf::bindings::python::PythonFuzzer &fuzzer, int bit_seq) {
46  // create seed corresponding to bit_seq
47  std::vector<u8> buf;
48  for (int i = 0; i < 8; i++) {
49  int d = bit_seq >> i & 1;
50  buf.emplace_back('0' + d);
51  }
52 
53  auto id = fuzzer.AddSeed(8, buf);
54  BOOST_CHECK(id != fuzzuf::exec_input::ExecInput::INVALID_INPUT_ID);
55  auto pyseed_optional = fuzzer.GetPySeed(id);
56  BOOST_CHECK(pyseed_optional.has_value());
57  auto trace_optional = pyseed_optional->GetFeedback().GetBBTrace();
58  BOOST_CHECK(trace_optional.has_value());
59  return trace_optional.value();
60 }

62 // to test both fork server mode and non fork server mode, we specify forksrv
63 // via an argument moreover, "whether BB coverage can be retrieved" should be
64 // independent from need_afl_cov. so we test both cases, need_afl_cov=false,
65 // need_afl_cov=true
66 static void InstrumentCovBB(bool forksrv, bool need_afl_cov) {
67  // cd $(dirname $0)
68  MoveToProgramLocation();
69 
70  // Create root directory
71  std::string root_dir_template("/tmp/fuzzuf_test.XXXXXX");
72  const auto raw_dirname = mkdtemp(root_dir_template.data());
73  if (!raw_dirname) throw -1;
74  BOOST_CHECK(raw_dirname != nullptr);
75 #ifdef HAS_CXX_STD_FILESYSTEM
76  namespace fs = std::filesystem;
77 #else
78  namespace fs = boost::filesystem;
79 #endif
80  auto root_dir = fs::path(raw_dirname);
81  BOOST_SCOPE_EXIT(&root_dir) {
82  // Executed on this test exit
83  fs::remove_all(root_dir);
84  }
85  BOOST_SCOPE_EXIT_END
86 
87  // Create input file
88  auto input_dir = root_dir / "input";
89  auto output_dir = root_dir / "output";
90  BOOST_CHECK_EQUAL(fs::create_directory(input_dir), true);
91  auto input_path = input_dir / "nothing";
92  create_file(input_path.string(),
93  "this seed is just nothing. we add more seeds later");
94 
95  // Create fuzzer instance
96  auto fuzzer = fuzzuf::bindings::python::PythonFuzzer(
97  {"../put_binaries/zeroone"}, input_dir.native(), output_dir.native(),
98  1000, 10000, forksrv, need_afl_cov, true // need_afl_cov, need_bb_cov
99  );
100 
101  // Configurate fuzzer
102  fuzzer.SuppressLog();
103 
104  // create "diffs" between 00000000 and 10000000, between 0000000 and 01000000,
105  // ...
106  auto base_cov = GetCoverage(fuzzer, 0);
107  // coverages cannot be empty
108  BOOST_CHECK(base_cov.size() > 0);
109 
110  DiffInfo diffs[8];
111  for (int i = 0; i < 8; i++) {
112  auto cov = GetCoverage(fuzzer, 1 << i);
113  // coverages cannot be empty
114  BOOST_CHECK(cov.size() > 0);
115 
116  // into DiffInfo::passed, append basic blocks which the seed corresponding
117  // to (1 << i) passes, but which the seed corresponding to 0 doesn't pass
118  for (const auto &itr : cov) {
119  if (!base_cov.count(itr.first)) diffs[i].passed.emplace_back(itr.first);
120  }
121  // into DiffInfo::ignored, append basic blocks which the seed corresponding
122  // to (1 << i) doesn't pass, but which the seed corresponding to 0 passes
123  for (const auto &itr : base_cov) {
124  if (!cov.count(itr.first)) diffs[i].ignored.emplace_back(itr.first);
125  }
126 
127  std::string name(8, '0');
128  name[i] = '1';
129  std::cout << "---" << name << "---\n";
130  std::cout << "IDs of nodes (BBs) which this input should pass:";
131  for (int node_idx : diffs[i].passed) {
132  std::cout << " " << node_idx;
133  }
134  std::cout << "\nIDs of nodes (BBs) which given input should not pass:";
135  for (int node_idx : diffs[i].ignored) {
136  std::cout << " " << node_idx;
137  }
138  std::cout << std::endl;
139 
140  // there must be some differences
141  BOOST_CHECK(diffs[i].passed.size() > 0);
142  BOOST_CHECK(diffs[i].ignored.size() > 0);
143 
144  // moreover, normally the difference should be just one node(where "var = 1"
145  // or "var = 0" takes place)
146  BOOST_CHECK_EQUAL(diffs[i].passed.size(), 1);
147  BOOST_CHECK_EQUAL(diffs[i].ignored.size(), 1);
148  }
149 
150  int lim = 1 << 8;
151  for (int bit_seq = 0; bit_seq < lim; bit_seq++) {
152  auto cov = GetCoverage(fuzzer, bit_seq);
153  for (int i = 0; i < 8; i++) {
154  // check if the i-th bit counting from LSB is set
155  bool is_ith_set = bit_seq >> i & 1;
156  if (is_ith_set) {
157  // the i-th bit of this seed is set
158  // that means seed bit_seq should have passed nodes in diffs[i].passed
159  for (int node_idx : diffs[i].passed) {
160  BOOST_CHECK(cov.count(node_idx) > 0);
161  }
162  // and that seed bit_seq should not have passed nodes in
163  // diffs[i].ignored
164  for (int node_idx : diffs[i].ignored) {
165  BOOST_CHECK(cov.count(node_idx) == 0);
166  }
167  } else {
168  // the i-th bit of this seed is not set
169  // that means seed bit_seq should have passed nodes in diffs[i].ignored
170  for (int node_idx : diffs[i].ignored) {
171  BOOST_CHECK(cov.count(node_idx) > 0);
172  }
173  // and that seed bit_seq should not have passed nodes in diffs[i].passed
174  for (int node_idx : diffs[i].passed) {
175  BOOST_CHECK(cov.count(node_idx) == 0);
176  }
177  }
178  }
179  }
180 }
