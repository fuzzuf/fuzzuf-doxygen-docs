33 class ExecutorKind {
34  public:
35  enum Kind {
36  UNKNOWN = 0, // Unknown Executor Type
37  NATIVE, // Native Executor (e.g. NativeLinuxExecutor)
38  FORKSERVER, // Use fuzzuf's forkserver
39  QEMU, // QEMU Executor
40  CORESIGHT // CoreSight Executor
41  };
42 
43  ExecutorKind() = default;
44 
45  constexpr ExecutorKind(Kind kind) : kind(kind) {}
46  // Allow switch case.
47  constexpr operator Kind() const { return kind; }
48  // Prevent if (executor_kind) usage.
49  explicit operator bool() = delete;
50 
51  const char* c_str() const {
52  switch (kind) {
53  case Kind::NATIVE:
54  return "native";
55  case Kind::FORKSERVER:
56  return "forkserver";
57  case Kind::QEMU:
58  return "qemu";
59  case Kind::CORESIGHT:
60  return "coresight";
61  default:
62  return "unknown";
63  }
64  }
65 
66  private:
67  Kind kind;
68 };
