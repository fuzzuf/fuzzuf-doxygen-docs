26 // Convert from an AFL dictionary format to an array of msgpack
27 
28 int main(int argc, char *argv[]) {
29  namespace po = boost::program_options;
30 
31  po::options_description desc("Options");
32  std::string in_file("input.dict");
33  std::string out_file("output.mp");
34  bool strict = false;
35  desc.add_options()("help,h", "show this message")(
36  "input,i", po::value<std::string>(&in_file), "input filename")(
37  "output,o", po::value<std::string>(&out_file), "output filename")(
38  "strict,s", po::bool_switch(&strict), "strict mode");
39  po::variables_map vm;
40  po::store(po::parse_command_line(argc, argv, desc), vm);
41  po::notify(vm);
42  if (vm.count("help") != 0U) {
43  std::cout << desc << std::endl;
44  exit(0);
45  }
46 
47  fuzzuf::algorithm::libfuzzer::dictionary::StaticDictionary dict;
48  Load(in_file, dict, strict,
49  [](std::string &&m) { std::cerr << m << std::endl; });
50 
51  std::vector<uint8_t> temp;
52 
53  if (dict.size() < (size_t(1U) << 4)) {
54  temp.push_back(0x90U | dict.size());
55  } else if (dict.size() < (size_t(1U) << 16)) {
56  temp.push_back(0xdcU);
57  temp.push_back((dict.size() >> 8) & 0xFFU);
58  temp.push_back(dict.size() & 0xFFU);
59  } else if (dict.size() < (size_t(1U) << 32)) {
60  temp.push_back(0xddU);
61  temp.push_back((dict.size() >> 24) & 0xFFU);
62  temp.push_back((dict.size() >> 16) & 0xFFU);
63  temp.push_back((dict.size() >> 8) & 0xFFU);
64  temp.push_back(dict.size() & 0xFFU);
65  } else {
66  std::cerr << "too many entries" << std::endl;
67  exit(1);
68  }
69 
70  for (const auto &v : dict) {
71  const size_t size = v.get().size();
72  if (size < (size_t(1U) << 8)) {
73  temp.push_back(0xc4);
74  temp.push_back(size);
75  } else if (v.get().size() < (size_t(1U) << 16)) {
76  temp.push_back(0xc5);
77  temp.push_back((size >> 8) & 0xFFU);
78  temp.push_back(size & 0xFFU);
79  } else if (v.get().size() < (size_t(1U) << 32)) {
80  temp.push_back(0xc6);
81  temp.push_back((size >> 24) & 0xFFU);
82  temp.push_back((size >> 16) & 0xFFU);
83  temp.push_back((size >> 8) & 0xFFU);
84  temp.push_back(size & 0xFFU);
85  } else {
86  std::cerr << "too long entry" << std::endl;
87  exit(1);
88  }
89  temp.insert(temp.end(), v.get().begin(), v.get().end());
90  }
91 
92  std::ofstream file(out_file);
93  file.write(reinterpret_cast<const char *>(temp.data()), temp.size());
94 }
