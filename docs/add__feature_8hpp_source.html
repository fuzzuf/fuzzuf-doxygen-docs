51 template <typename State, typename Corpus>
52 auto AddFeature(State &state, Corpus &corpus, size_t index,
53  std::uint32_t new_size, bool shrink)
54  -> std::enable_if_t<is_state_v<State> && is_full_corpus_v<Corpus>, bool> {
55  assert(new_size);
56  index = index % utils::range::rangeSize(state.input_sizes_per_feature);
57  std::uint32_t old_size = state.input_sizes_per_feature[index];
58 
59  if (old_size == 0 || (shrink && old_size > new_size)) {
60  if (old_size > 0) {
61  std::size_t old_index = state.smallest_element_per_feature[index];
62  auto iter = std::next(corpus.corpus.template get<Sequential>().begin(),
63  old_index);
64  bool drop = false;
65  corpus.corpus.template get<Sequential>().modify(
66  iter, [&](auto &testcase) {
67  if (testcase) {
68  assert(testcase.features_count > 0);
69  testcase.features_count--;
70  if (testcase.features_count == 0)
71  drop = true;
72  }
73  });
74  if (drop)
75  ::fuzzuf::algorithm::libfuzzer::corpus::deleteInput(state, corpus,
76  old_index);
77  } else {
78  ++state.added_features_count;
79  if (state.create_info.config.entropic.enabled)
80  AddRareFeature(state, corpus, index);
81  }
82  ++state.updated_features_count;
83  // Inputs.size() is guaranteed to be less than UINT32_MAX by AddToCorpus.
84  state.smallest_element_per_feature[index] =
85  static_cast<uint32_t>(corpus.corpus.size());
86  state.input_sizes_per_feature[index] = new_size;
87  return true;
88  }
89  return false;
90 }
