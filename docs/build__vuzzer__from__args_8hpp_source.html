84  fuzzer_desc.add_options()
85  ("full_dict",
86  po::value<std::string>(&vuzzer_options.full_dict),
87  "Set path to \"full dictionary\". Default is `./full.dict`.")
88  ("unique_dict",
89  po::value<std::string>(&vuzzer_options.unique_dict),
90  "Set path to \"unique dictionary\". Default is `./unique.dict`.")
91  ("weight",
92  po::value<std::string>(&vuzzer_options.weight),
93  "Set path to \"weight file\". Default is `./weight`.")
94  ("inst_bin",
95  po::value<std::string>(&vuzzer_options.inst_bin),
96  "Set path to instrumented binary. Default is `./instrumented.bin`.")
97  ("taint_db",
98  po::value<std::string>(&vuzzer_options.taint_db),
99  "Set path to taint db. Default is `/mnt/polytracker/polytracker.db`.")
100  ("taint_out",
101  po::value<std::string>(&vuzzer_options.taint_out),
102  "Set path to output for taint analysis. Default is `/tmp/taint.out`.")
103  ("pargs",
104  po::value<std::vector<std::string>>(&pargs),
105  "Specify PUT and args for PUT.")
106  ;
107 
108  po::variables_map vm;
109  po::store(
110  po::command_line_parser(fuzzer_args.argc, fuzzer_args.argv)
111  .options(fuzzer_args.global_options_description.add(fuzzer_desc))
112  .positional(pargs_desc)
113  .run(),
114  vm
115  );
116  po::notify(vm);
117 
118  if (global_options.help) {
119  fuzzuf::cli::fuzzer::vuzzer::usage(fuzzer_args.global_options_description);
120  }
121 
122  /* (Pin) Executor of vuzzer requires absolute path of PUT binary */
123  pargs[0] = fs::absolute(pargs[0]).native();
124 
125  PutArgs put(pargs);
126  try {
127  put.Check();
128  } catch (const exceptions::cli_error &e) {
129  std::cerr << "[!] " << e.what() << std::endl;
130  std::cerr << "\tat " << e.file << ":" << e.line << std::endl;
131  fuzzuf::cli::fuzzer::vuzzer::usage(fuzzer_args.global_options_description);
132  }
133 
134  DEBUG("[*] PUT: put = [");
135  for (auto v : put.Args()) {
136  DEBUG("\t\"%s\",", v.c_str());
137  }
138  DEBUG(" ]");

140  using fuzzuf::algorithm::vuzzer::VUzzerSetting;
141  using fuzzuf::algorithm::vuzzer::option::VUzzerTag;
142  using fuzzuf::algorithm::vuzzer::option::GetDefaultOutfile;
143 
144  // Create VUzzerSetting
145  std::shared_ptr<VUzzerSetting> setting(
146  new VUzzerSetting(
147  put.Args(),
148  global_options.in_dir,
149  global_options.out_dir,
150  vuzzer_options.weight,
151  vuzzer_options.full_dict,
152  vuzzer_options.unique_dict,
153  vuzzer_options.inst_bin,
154  vuzzer_options.taint_db,
155  vuzzer_options.taint_out,
156  0, 0
157  )
158  );

160  // PinToolExecutor needs the directory specified by "out_dir" to be already set up

163 
164  // XXX: VUzzer supports PinToolExecutor/PolyTrackerExecutor only. It does not actually support `native` executor.
165  if (global_options.executor != fuzzuf::cli::ExecutorKind::NATIVE) {

171  std::shared_ptr<PinToolExecutor> executor(
172  new PinToolExecutor(
173  FUZZUF_PIN_EXECUTABLE,
174  {TEST_BINARY_DIR "/../tools/bbcounts2/bbcounts2.so", "-o", "bb.out", "-libc", "0"},
175  setting->argv,
176  setting->exec_timelimit_ms,
177  setting->exec_memlimit,
178  setting->out_dir / GetDefaultOutfile()
179  )
180  );
181 
182  // Create PolyTrackerExecutor
183  std::shared_ptr<PolyTrackerExecutor> taint_executor(
184  new PolyTrackerExecutor(
185  TEST_BINARY_DIR "/../tools/polyexecutor/polyexecutor.py",
186  setting->path_to_inst_bin,
187  setting->path_to_taint_db,
188  setting->path_to_taint_file,
189  setting->argv,
190  setting->exec_timelimit_ms,
191  setting->exec_memlimit,
192  setting->out_dir / GetDefaultOutfile()
193  )
194  );
195 
196  // Create VUzzerState
197  using fuzzuf::algorithm::vuzzer::VUzzerState;
198 
199  auto state = std::make_unique<VUzzerState>(setting, executor, taint_executor);
200 
201  return std::unique_ptr<TFuzzer>(
202  dynamic_cast<TFuzzer *>(
203  new TVUzzer(std::move(state))
204  )
205  );
206 }
