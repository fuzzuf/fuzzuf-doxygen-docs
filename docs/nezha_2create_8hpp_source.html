58 template <typename F, typename Ord, typename Sink>
59 auto CreateRunSingleTarget(const fs::path &target_path,
60  const FuzzerCreateInfo &create_info, bool use_output,
61  bool force_add_to_corpus, bool may_delete_file,
62  bool persistent, bool strict_match, size_t i,
63  const Sink &sink) {
64  namespace lf = fuzzuf::algorithm::libfuzzer;
65  namespace hf = fuzzuf::hierarflow;
66  auto create_local_coverage =
67  hf::CreateNode<lf::Clear<F, decltype(Ord::coverage)>>();
68 
69  const auto output_file_path = create_info.output_dir / "result";
70  const auto path_to_write_seed = create_info.output_dir / "cur_input";
71 
72  std::unique_ptr<NativeLinuxExecutor> executor(new NativeLinuxExecutor(
73  {target_path.string(), output_file_path.string()},
74  create_info.exec_timelimit_ms, create_info.exec_memlimit,
75  create_info.forksrv, path_to_write_seed, create_info.afl_shm_size,
76  create_info.bb_shm_size, create_info.cpuid_to_bind, true));
77  auto execute =
78  hf::CreateNode<lf::standard_order::Execute<F, NativeLinuxExecutor, Ord>>(
79  std::move(executor), create_info.use_afl_coverage);
80  auto collect_features =
81  hf::CreateNode<standard_order::CollectFeatures<F, Ord>>(
82  i * (create_info.use_afl_coverage ? create_info.afl_shm_size
83  : create_info.bb_shm_size));
84  auto add_to_corpus = hf::CreateNode<lf::standard_order::AddToCorpus<F, Ord>>(
85  force_add_to_corpus, may_delete_file, persistent, strict_match,
86  create_info.output_dir, Sink(sink));
87  auto new_cov = hf::CreateNode<lf::standard_order::IfNewCoverage<F, Ord>>();
88  auto update_dict =
89  hf::CreateNode<lf::standard_order::UpdateDictionary<F, Ord>>();
90  auto print =
91  create_info.print_pcs
92  ? hf::CreateNode<lf::standard_order::PrintStatusForNewUnit<F, Ord>>(
93  create_info.verbosity, create_info.max_mutations_to_print,
94  create_info.max_unit_size_to_print, sink)
95  : hf::CreateNode<lf::Proxy<F>>();
96 
97  auto nop1 = hf::CreateNode<lf::Proxy<F>>();
98 
99  auto nop2 = hf::CreateNode<lf::Proxy<F>>();
100 
101  auto gather_trace_ = hf::CreateNode<lf::DynamicAppend<F, decltype( Ord::added_to_corpus && Ord::trace )>>();
102 
103  auto gather_output_ =
104  (use_output) ? hf::CreateNode<standard_order::GatherOutput<F, Ord>>()
105  : hf::CreateNode<lf::DynamicAppend<F, decltype( Ord::single_status && Ord::status)>>();
106 
107  auto assign_last_corpus_update_run = hf::CreateNode<lf::DynamicAssign<
108  F, decltype(Ord::count && Ord::last_corpus_update_run)>>();
109 
110  nop1 << (create_local_coverage << execute << collect_features
111  << add_to_corpus ||
112  new_cov << (update_dict || print ||
113  assign_last_corpus_update_run) ||
114  gather_trace_ << gather_output_);
115  return nop1;
116 }

139 template <typename F, typename Ord, typename Sink>
140 auto CreateRunTargets(const std::vector<fs::path> &target_path,
141  const FuzzerCreateInfo &create_info, bool use_output,
142  bool force_add_to_corpus, bool may_delete_file,
143  bool persistent, bool strict_match, const Sink &sink) {
144  namespace lf = fuzzuf::algorithm::libfuzzer;
145  namespace hf = fuzzuf::hierarflow;
146  auto nop1 = hf::CreateNode<lf::Proxy<F>>();
147 
148  auto run = hf::CreateNode<lf::Proxy<F>>();
149 
150  size_t i = 0u;
151  for (auto &p : target_path) {
152  auto single = CreateRunSingleTarget<F, Ord>(
153  p, create_info, use_output, force_add_to_corpus, may_delete_file,
154  persistent, strict_match, i, sink);
155  run << single;
156  ++i;
157  }
158 
159  return run;
160 }

176 template <typename F, typename Ord, typename Sink>
177 auto createRunone(const std::vector<fs::path> &target_path,
178  const FuzzerCreateInfo &create_info, bool use_output,
179  const Sink &sink) {
180  namespace lf = fuzzuf::algorithm::libfuzzer;
181  namespace hf = fuzzuf::hierarflow;
182 
183  auto increment_counter =
184  hf::CreateNode<lf::StaticAppend<F, decltype(Ord::count)>>(1u);
185  auto local_loop =
186  hf::CreateNode<lf::standard_order::RepeatUntilNewCoverage<F, Ord>>(
187  create_info.max_mutation_retry_count);
188 
189  auto mutation_loop =
190  hf::CreateNode<lf::standard_order::RepeatUntilMutated<F, Ord>>(
191  1, create_info.mutation_depth);
192 
193  auto select_input =
194  hf::CreateNode<lf::standard_order::ChooseRandomSeed<F, Ord>>(false);
195  auto select_crossover = hf::CreateNode<
196  lf::ChooseRandomSeed<F, decltype(Ord::state && Ord::corpus && Ord::rng &&
197  Ord::crossover && Ord::exec_result)>>(
198  create_info.crossover_uniform_dist);
199 
200  auto update_dict =
201  hf::CreateNode<lf::standard_order::UpdateDictionary<F, Ord>>();
202 
203  auto increment_mutations_count =
204  hf::CreateNode<lf::standard_order::IncrementMutationsCount<F, Ord>>();
205 
206  auto create_dict_entry =
207  hf::CreateNode<lf::Clear<F, decltype(Ord::dict_history)>>();
208 
209  auto create_history =
210  hf::CreateNode<lf::Clear<F, decltype(Ord::mutation_history)>>();
211 
212  auto nop1 = hf::CreateNode<lf::Proxy<F>>();
213 
214  auto nop2 = hf::CreateNode<lf::Proxy<F>>();
215 
216  auto nop3 = hf::CreateNode<lf::Proxy<F>>();
217 
218  auto nop4 = hf::CreateNode<lf::Proxy<F>>();
219 
220  auto nop5 = hf::CreateNode<lf::Proxy<F>>();
221 
222  auto add_to_solution =
223  use_output
224  ? hf::CreateNode<standard_order::AddToSolution<F, Ord>>(
225  create_info.output_dir)
226  : hf::CreateNode<
227  AddToSolution<F, decltype(Ord::input && Ord::exec_result &&
228  Ord::trace && Ord::known_traces &&
229  Ord::status && Ord::known_status)>>(
230  create_info.output_dir);
231 
232  auto create_trace = hf::CreateNode<lf::Clear<F, decltype(Ord::trace)>>();
233  auto create_outputs = hf::CreateNode<lf::Clear<F, decltype(Ord::outputs)>>();
234 
235  auto update_distribution =
236  hf::CreateNode<lf::standard_order::UpdateDistribution<
237  F, lf::MakeVersion(12u, 0u, 0u), Ord>>(
238  create_info.sparse_energy_updates, create_info.max_mutation_factor,
239  sink);
240 
241  auto update_max_length =
242  create_info.len_control
243  ? hf::CreateNode<lf::standard_order::UpdateMaxLength<F, Ord>>(
244  create_info.max_input_length, create_info.len_control)
245  : hf::CreateNode<lf::Proxy<F>>();
246 
247  /*
248  * Due to current clang-format rule, following graph definition is formatted in unexpected form.
249  * The rule need to be changed to preserve indentation of graph definition.
250  */
251  nop4 << (select_crossover || select_input ||
252  local_loop << (create_history || create_dict_entry ||
253  mutation_loop
254  << lf::createMutator<F, Ord>(create_info) ||
255  increment_mutations_count || create_outputs ||
256  create_trace ||
257  CreateRunTargets<F, Ord>(target_path, create_info,
258  use_output, false, true,
259  false, true, sink) ||
260  add_to_solution) ||
261  increment_counter || update_distribution || update_max_length);
262  return nop4;
263 }

276 template <typename F, typename Ord, typename Sink>
277 auto create(const std::vector<fs::path> &target_path,
278  const libfuzzer::FuzzerCreateInfo &create_info, bool use_output,
279  ExecInputSet &initial_inputs, const Sink &sink) {

281  namespace hf = fuzzuf::hierarflow;
282

284  hf::CreateNode<lf::PartiallyDynamicRepeat<F, decltype(Ord::count)>>(
285  create_info.total_cycles);
286 
287  auto nop1 = hf::CreateNode<lf::Proxy<F>>();
288  auto nop2 = hf::CreateNode<lf::Proxy<F>>();
289 
290  auto dump_state =
291  create_info.print_final_stats
292  ? hf::CreateNode<lf::StaticDump<F, decltype(Ord::state)>>(
293  "state\n", 1, " ", sink)
294  : hf::CreateNode<lf::Proxy<F>>();
295 
296  nop1 << (lf::createInitialize<F, Ord>(target_path[0], create_info,
297  initial_inputs, true, sink) ||
298  global_loop << createRunone<F, Ord>(target_path, create_info,
299  use_output, sink) ||
300  dump_state);
301  return nop1;
302 }
