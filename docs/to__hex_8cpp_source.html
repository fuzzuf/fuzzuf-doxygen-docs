25 void toHex(std::string &message, const std::vector<std::uint8_t> &range) {
26  namespace karma = boost::spirit::karma;
27  karma::generate(std::back_inserter(message),
28  *karma::right_align(2, '0')[karma::hex], range);
29  message += "\n";
30  const auto end = std::find(range.begin(), range.end(), '\0');
31  message += std::string(range.begin(), end);
32  message += "\n";
33 }
