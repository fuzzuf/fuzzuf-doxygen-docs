59 auto AddToCorpus(Corpus &corpus, Range &range, InputInfo &testcase_,
60  bool persistent, const fs::path &path_prefix)
61  -> std::enable_if_t<is_full_corpus_v<Corpus> &&
62  is_input_info_v<InputInfo>> {
63  assert(!utils::range::rangeEmpty(range));
64  // Inputs.size() is cast to uint32_t below.
65  assert(corpus.corpus.size() < std::numeric_limits<uint32_t>::max());
66 
67  std::uint64_t id = 0u;
68  testcase_.sha1 = utils::ToSerializedSha1(range);
69  testcase_.input_size = utils::range::rangeSize(range);
70  if (testcase_.name.empty()) testcase_.name = testcase_.sha1;
71  auto &hashed = corpus.corpus.template get<ByName>();
72  const auto existing = hashed.find(testcase_.name);
73  if (existing != hashed.end()) {
74  const auto old_id = existing->id;
75  corpus.inputs.erase(old_id);
76  }
77  if (!persistent) {
78  auto exec_input = corpus.inputs.CreateOnMemory(
79  range.data(), utils::range::rangeSize(range));
80  assert(exec_input);
81  id = exec_input->GetID();
82  } else {
83  auto exec_input =
84  corpus.inputs.CreateOnDisk(path_prefix / fs::path(testcase_.name));
85  assert(exec_input);
86  exec_input->OverwriteThenUnload(range.data(),
87  utils::range::rangeSize(range));
88  id = exec_input->GetID();
89  }
90  testcase_.id = id;
91  std::sort(testcase_.unique_feature_set.begin(),
92  testcase_.unique_feature_set.end());
93  if (existing != hashed.end()) {
94  testcase_.reduced = true;
95  hashed.modify(existing, [&](auto &input_info) {
96  corpus.inputs.erase(input_info.id);
97  const auto id = input_info.id;
98  input_info = InputInfo();
99  input_info.id = id;

101  });
102  hashed.insert(hashed.end(), testcase_);
103  } else
104  hashed.insert(testcase_);
