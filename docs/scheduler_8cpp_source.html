35 // Tentative efficiency of random fuzzing. TODO: Communicate with AFL for this.
36 constexpr float RAND_FUZZ_EFFICIENCY = 0.0005f;
37 // Decides sleep factor 'f', which will be used to sleep for 'f * elapsed time'.
38 // This means we will utilize 1 / (2 * (f + 1)) of the system resource.
39 float DecideSleepFactor(
40  const std::function<void(std::string &&)> &sink,
41  const options::FuzzOption &opt,
42  int round_execs,
43  int round_tcs
44 ) {
45  const auto grey_conc_efficiency = float( round_tcs ) / float( round_execs );
46  if( opt.verbosity >= 1 ) {
47  std::string message = "[*] Efficiency = ";
48  message += std::to_string( grey_conc_efficiency );
49  sink( std::move( message ) );
50  }
51  // GREY_CONC_EFF : RAND_FUZZ_EFF = 1 : 2 * factor + 1
52  const auto factor = ( grey_conc_efficiency == 0.0f ) ?
53  SLEEP_FACTOR_MAX :
54  ( RAND_FUZZ_EFFICIENCY / grey_conc_efficiency - 1.0f ) / 2.0f;
55  // Bound the factor between minimum and maximum value allowed.
56  return std::max( SLEEP_FACTOR_MIN, std::min( SLEEP_FACTOR_MAX, factor ) );
57 }
58 }

66 // Check the efficiency of the system and sleep for a while to adjust the weight
67 // of resource use with AFL.
68 void CheckAndReserveTime(
69  const std::function<void(std::string &&)> &sink,
70  const options::FuzzOption &opt
71 ) {
72  const auto round_execs = executor::GetRoundExecs();
73  if( round_execs > ROUND_SIZE ) {
74  const auto round_tcs = test_case::GetRoundTestCaseCount();
75  executor::ResetRoundExecs();
76  test_case::ResetRoundTestCaseCount();
77  const auto sleep_factor = DecideSleepFactor( sink, opt, round_execs, round_tcs );
78  const auto round_elapsed = std::chrono::duration_cast< std::chrono::milliseconds >(
79  std::chrono::high_resolution_clock::now() - timer
80  ).count();
81  const auto sleep_time = int( float( round_elapsed ) * sleep_factor );
82  if( opt.verbosity >= 1 ) {
83  std::string message = "[*] Elapsed round time: ";
84  message += std::to_string( int( round_elapsed / 1000 ) );
85  message += " sec.\n[*] Decided sleep time: ";
86  message += std::to_string( int( sleep_time / 1000 ) );
87  message += " sec.\n";
88  sink( std::move( message ) );
89  }
90  std::this_thread::sleep_for( std::chrono::milliseconds( sleep_time ) );
91  timer = std::chrono::high_resolution_clock::now();
92  }
