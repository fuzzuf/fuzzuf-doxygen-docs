120 void helper_eclipser_log_bb(abi_ulong addr) {
121  abi_ulong prev_addr_local;
122  abi_ulong edge, hash;
123  unsigned int byte_idx, byte_mask;
124  unsigned char old_byte, new_byte;
125 
126  // Make sure that 'prev_addr' is always updated even if we just return.
127  prev_addr_local = prev_addr;
128  prev_addr = addr;
129 
130  if (!coverage_fp || !edge_bitmap)
131  return;
132 
133 #ifdef TARGET_X86_64
134  edge = (prev_addr_local << 16) ^ addr;
135 #else
136  edge = (prev_addr_local << 8) ^ addr;
137 #endif
138 
139  // Update bitmap.
140  hash = (edge >> 4) ^ (edge << 8);
141  byte_idx = (hash >> 3) & BITMAP_MASK;
142  byte_mask = 1 << (hash & 0x7); // Use the lowest 3 bits to shift
143  old_byte = edge_bitmap[byte_idx];
144  new_byte = old_byte | byte_mask;
145  if (old_byte != new_byte) {
146  found_new_edge = 1;
147  edge_bitmap[byte_idx] = new_byte;
148  /* Log visited nodes if dbg_fp is not NULL */
149  if (dbg_fp) {
150 #ifdef TARGET_X86_64
151  fprintf(dbg_fp, "(0x%lx)\n", addr);
152 #else
153  fprintf(dbg_fp, "(0x%x)\n", addr);
154 #endif
155  }
156  }
