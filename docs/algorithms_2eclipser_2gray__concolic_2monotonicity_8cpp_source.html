99 bool CheckIntermediate(
100  Tendency tendency,
101  const BigInt &y1,
102  const BigInt &y2,
103  const BigInt &y3
104 ) {
105  if( tendency == Tendency::Incr ) {
106  return y1 < y2 && y2 < y3;
107  }
108  else if( tendency == Tendency::Decr ) {
109  return y1 > y2 && y2 > y3;

111  else {
112  failwith( "Invalid tendency input" );
113  return false; // unreachable;
114  }

130 std::optional< Tendency >
131 CheckMonotonicAux(
132  Signedness sign,
133  const BigInt &prev_x,
134  const BigInt &prev_y,
135  Tendency tendency,
136  const std::vector< std::pair< BigInt, BigInt > >::const_iterator &coordinates_begin,
137  const std::vector< std::pair< BigInt, BigInt > >::const_iterator &coordinates_end
138 ) {
139  if( coordinates_begin == coordinates_end ) {
140  return tendency;
141  }
142  const auto &[x,y] = *coordinates_begin;
143  if( x <= prev_x )
144 #if __GNUC__ >= 9 && __cplusplus > 201703L
145  [[unlikely]]
146 #endif
147  {
148  failwith( "Invalid coordinates" );
149  return Tendency::Undetermined; // unreachable
150  }
151  if( tendency == Tendency::Incr && prev_y <= y ) {
152  return CheckMonotonicAux( sign, x, y, Tendency::Incr, std::next( coordinates_begin ), coordinates_end );
153  }
154  else if( tendency == Tendency::Incr && sign == Signedness::Signed && prev_y > 0 && y < 0 ) {
155  return CheckMonotonicAux( sign, x, y, Tendency::Incr, std::next( coordinates_begin ), coordinates_end );
156  }
157  else if( tendency == Tendency::Decr && prev_y >= y ) {
158  return CheckMonotonicAux( sign, x, y, Tendency::Decr, std::next( coordinates_begin ), coordinates_end );
159  }
160  else if( tendency == Tendency::Decr && sign == Signedness::Signed && prev_y < 0 && y > 0 ) {
161  return CheckMonotonicAux( sign, x, y, Tendency::Decr, std::next( coordinates_begin ), coordinates_end );
162  }
163  else if( tendency == Tendency::Undetermined && prev_y == y ) {
164  return CheckMonotonicAux( sign, x, y, Tendency::Undetermined, std::next( coordinates_begin ), coordinates_end );
165  }
166  else if( tendency == Tendency::Undetermined && prev_y < y ) {
167  return CheckMonotonicAux( sign, x, y, Tendency::Incr, std::next( coordinates_begin ), coordinates_end );
168  }
169  else if( tendency == Tendency::Undetermined && prev_y > y ) {
170  return CheckMonotonicAux( sign, x, y, Tendency::Decr, std::next( coordinates_begin ), coordinates_end );
171  }
172  else {
173  return std::nullopt;
174  }
175 }

198 std::optional< Monotonicity >
199 GenerateAux(
200  Tendency tendency,
201  const BigInt &targ_y,
202  const BigInt &prev_x,
203  const BigInt &prev_y,
204  const std::vector< std::pair< BigInt, BigInt > >::const_iterator &coordinates_begin,
205  const std::vector< std::pair< BigInt, BigInt > >::const_iterator &coordinates_end
206 ) {
207  if( coordinates_begin == coordinates_end ) {
208  return std::nullopt;
209  }
210  else {
211  const auto &[x,y] = *coordinates_begin;
212  if( prev_y == targ_y || y == targ_y ) {
213  return std::nullopt;
214  }
215  else if( CheckIntermediate( tendency, prev_y, targ_y, y ) ) {
216  return Make( tendency, prev_x, prev_y, x, y, targ_y );
217  }
218  else {
219  return GenerateAux( tendency, targ_y, x, y, std::next( coordinates_begin ), coordinates_end );
220  }
221  }

243 std::optional< Monotonicity >
244 Find(
245  const std::vector< BranchInfo >::const_iterator &br_infos_begin,
246  const std::vector< BranchInfo >::const_iterator &br_infos_end
247 ) {
248  if( br_infos_begin == br_infos_end )
249 #if __GNUC__ >= 9 && __cplusplus > 201703L
250  [[unlikely]]
251 #endif
252  {
253  failwith( "Empty branchInfo list provided as input" );
254  return std::nullopt; // unreachable
255  }
256  const auto &head_br_info = *br_infos_begin;
257  const auto sign = ( head_br_info.branch_type == CompareType::UnsignedSize ) ? Signedness::Unsigned : Signedness::Signed;
258  const auto size = head_br_info.operand_size;
259  if( std::find_if(
260  br_infos_begin,
261  br_infos_end,
262  [&head_br_info]( const auto &v ) {
263  return head_br_info.operand1 != v.operand1;
264  }
265  ) == br_infos_end ) {
266  const auto target_y = branch_info::InterpretAs( sign, size, head_br_info.operand1 );
267  std::vector< std::pair< BigInt, BigInt > > coordinates;
268  coordinates.reserve( std::distance( br_infos_begin, br_infos_end ) );
269  std::transform(
270  br_infos_begin,
271  br_infos_end,
272  std::back_inserter( coordinates ),
273  [&]( const auto &br ) {
274  return make_pair(
275  br.try_value,
276  branch_info::InterpretAs( sign, size, br.operand2 )
277  );
278  }
279  );
280  const auto tendency = CheckMonotonic( sign, coordinates );
281  if( !tendency ) {
282  return std::nullopt;
283  }
284  else {
285  return Generate( *tendency, target_y, coordinates );
286  }
287  }
288  else if( std::find_if(
289  br_infos_begin,
290  br_infos_end,
291  [&head_br_info]( const auto &v ) {
292  return head_br_info.operand2 != v.operand2;
293  }
294  ) == br_infos_end ) {
295  const auto target_y = branch_info::InterpretAs( sign, size, head_br_info.operand2 );
296  std::vector< std::pair< BigInt, BigInt > > coordinates;
297  coordinates.reserve( std::distance( br_infos_begin, br_infos_end ) );
298  std::transform(
299  br_infos_begin,
300  br_infos_end,
301  std::back_inserter( coordinates ),
302  [&]( const auto &br ) {
303  return make_pair(
304  br.try_value,
305  branch_info::InterpretAs( sign, size, br.operand1 )
306  );
307  }
308  );
309  const auto tendency = CheckMonotonic( sign, coordinates );
310  if( !tendency ) {
311  return std::nullopt;
312  }
313  else {
314  return Generate( *tendency, target_y, coordinates );
315  }
316  }
317  else {
318  return std::nullopt;
319  }
320 }

322 Monotonicity AdjustByteLen( const Monotonicity &monotonic ) {
323  const auto lower_x = monotonic.lower_x;
324  const auto upper_x = monotonic.upper_x;
325  if( upper_x - lower_x > 1 ) {
326  return monotonic;
327  }
328  else {
329  const auto new_lower_x = lower_x << 8;
330  const auto new_upper_x = ( upper_x << 8 ) + 255;
331  auto new_monotonic = monotonic;
332  new_monotonic.lower_x = new_lower_x;
333  new_monotonic.lower_y = std::nullopt;
334  new_monotonic.upper_x = new_upper_x;
335  new_monotonic.upper_y = std::nullopt;
336  new_monotonic.byte_len = monotonic.byte_len + 1;
337  return new_monotonic;
338  }

341 Monotonicity
342 UpdateInterval( const Monotonicity monotonic, const BigInt &x, const BigInt &y ) {
343  if( monotonic.tendency == Tendency::Incr ) {
344  if( y < monotonic.target_y ) {
345  auto new_monotonic = monotonic;
346  new_monotonic.lower_x = x;
347  new_monotonic.lower_y = y;
348  return new_monotonic;
349  }
350  else {
351  auto new_monotonic = monotonic;
352  new_monotonic.upper_x = x;
353  new_monotonic.upper_y = y;
354  return new_monotonic;
355  }
356  }
357  else if( monotonic.tendency == Tendency::Decr ) {
358  if( y < monotonic.target_y ) {
359  auto new_monotonic = monotonic;
360  new_monotonic.upper_x = x;
361  new_monotonic.upper_y = y;
362  return new_monotonic;
363  }
364  else {
365  auto new_monotonic = monotonic;
366  new_monotonic.lower_x = x;
367  new_monotonic.lower_y = y;
368  return new_monotonic;
369  }
370  }
371  else {
372  failwith( "Unreachable" );
373  return Monotonicity(); // unreachable
374  }
