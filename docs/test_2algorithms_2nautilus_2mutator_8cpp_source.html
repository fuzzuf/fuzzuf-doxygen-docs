41  Context ctx;
42  ctx.AddRule("N1", "r1{N2}{N3}{N4}");
43  ctx.AddRule("N2", "r2");
44  ctx.AddRule("N3", "r3{N1}");
45  ctx.AddRule("N1", "r4");
46  ctx.AddRule("N4", "r5");
47 
48  std::vector<RuleIDOrCustom> rules{
49  RuleIDOrCustom(r1), RuleIDOrCustom(r2), RuleIDOrCustom(r3),
50  RuleIDOrCustom(r4), RuleIDOrCustom(r5),
51  };
52  Tree tree(std::move(rules), ctx);
53 
54  Mutator mutator(ctx);
55  FTesterMut tester =
56  [&r1, &r2, &r3, &r4, &r5](TreeMutation& tree_mut, Context&) {
57  BOOST_CHECK(tree_mut.prefix().at(0) == RuleIDOrCustom(r1));
58  BOOST_CHECK(tree_mut.prefix().at(1) == RuleIDOrCustom(r2));
59  BOOST_CHECK(tree_mut.prefix().at(2) == RuleIDOrCustom(r3));
60 
61  BOOST_CHECK(tree_mut.postfix().at(0) == RuleIDOrCustom(r5));
62 
63  BOOST_CHECK(tree_mut.repl().at(0) == RuleIDOrCustom(r1));
64  BOOST_CHECK(tree_mut.repl().at(1) == RuleIDOrCustom(r2));
65  BOOST_CHECK(tree_mut.repl().at(2) == RuleIDOrCustom(r3));
66  BOOST_CHECK(tree_mut.repl().back() == RuleIDOrCustom(r5));
67  };
68 
69  auto recursions = tree.CalcRecursions(ctx);
70  BOOST_CHECK(recursions);
71 
72  mutator.MutRandomRecursion(tree, recursions.value(), ctx, tester);
73 }
