61 BigIntToBytes( Endian endian, std::size_t size, BigInt value ) {
62  std::vector< std::byte > temp;
63  temp.reserve( size );
64  for( std::size_t i = 0u; i != size; ++i ) {
65  temp.push_back( std::byte( std::uint8_t( value ) ) );
66  value >>= 8;
67  }
68  if( endian == Endian::BE ) {
69  std::reverse( temp.begin(), temp.end() );
70  }
71  return temp;
72 }
