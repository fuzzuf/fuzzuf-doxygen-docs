178 VOID Trace(TRACE trace, VOID *v)
179 {
180  for (BBL bbl = TRACE_BblHead(trace); BBL_Valid(bbl); bbl = BBL_Next(bbl))
181  {
182  /* Things related to stack hash/crash fingerprints */
183  BBL_InsertCall(bbl, IPOINT_ANYWHERE, AFUNPTR(crashBlock), IARG_FAST_ANALYSIS_CALL, IARG_ADDRINT, BBL_Address(bbl), IARG_END);
184  INS tail = BBL_InsTail(bbl);
185  if( INS_IsCall(tail) )
186  {
187  if( INS_IsDirectBranchOrCall(tail))
188  {
189  ADDRINT target = INS_DirectBranchOrCallTargetAddress(tail);
190  INS_InsertPredicatedCall(tail, IPOINT_BEFORE, AFUNPTR(call_direct), IARG_ADDRINT, target, IARG_END);
191  }
192  else
193  {
194  INS_InsertCall(tail, IPOINT_BEFORE, AFUNPTR(call_indirect), IARG_BRANCH_TARGET_ADDR, IARG_BRANCH_TAKEN, IARG_END);
195  }
196  }
197  /* stack hask ends here. */
198 
199  if(isMonitoredAddress(BBL_Address(bbl)))
200  {
201  BBL_InsertCall(bbl, IPOINT_ANYWHERE, AFUNPTR(rememberBlock), IARG_FAST_ANALYSIS_CALL, IARG_ADDRINT, BBL_Address(bbl), IARG_END);
202  }
203  }
204 }
