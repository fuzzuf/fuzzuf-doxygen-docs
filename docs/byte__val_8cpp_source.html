252 void from_json( const nlohmann::json &src, ByteVal &dest ) {
253  if( !src.is_object() ) dest = Fixed();
254  else if( src.find( "type" ) == src.end() ) dest = Fixed();
255  else if( ( src[ "type" ] == "Fixed" ) && ( src.find( "value" ) != src.end() ) ) {
256  dest = Fixed{ std::byte( src[ "value" ]. template get< int >() ) };
257  }
258  else if( ( src[ "type" ] == "Interval" ) && ( src.find( "low" ) != src.end() ) && ( src.find( "high" ) != src.end() ) ) {
259  dest = Interval{ std::byte( src[ "low" ]. template get< int >() ), std::byte( src[ "high" ]. template get< int >() ) };
260  }
261  else if( ( src[ "type" ] == "Undecided" ) && ( src.find( "value" ) != src.end() ) ) {
262  dest = Undecided{ std::byte( src[ "value" ]. template get< int >() ) };
263  }
264  else if( ( src[ "type" ] == "Untouched" ) && ( src.find( "value" ) != src.end() ) ) {
265  dest = Untouched{ std::byte( src[ "value" ]. template get< int >() ) };
266  }
267  else if( ( src[ "type" ] == "Sampled" ) && ( src.find( "value" ) != src.end() ) ) {
268  dest = Sampled{ std::byte( src[ "value" ]. template get< int >() ) };
