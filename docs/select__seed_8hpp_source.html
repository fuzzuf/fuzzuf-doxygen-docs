88 template <typename Corpus>
89 auto GenerateVanillaSchedule(Corpus &corpus, std::vector<double> &weights)
90  -> std::enable_if_t<is_full_corpus_v<Corpus>> {
91  const std::size_t corpus_size = corpus.corpus.size();
92  std::size_t i = 0u;
93  weights.clear();
94  weights.reserve(corpus_size);
95  utils::ForEachMultiIndexValues<false>(
96  corpus.corpus.template get<Sequential>(), [&](auto &input) {
97  weights.push_back(
98  input.features_count
99  ? static_cast<double>((i + 1) *
100  (input.has_focus_function ? 1000 : 1))
101  : 0.);
102  input.weight = weights.back();
103  ++i;
104  });
105 }

118 template <typename State, typename Corpus>
119 auto GenerateEntropicSchedule(State &state, Corpus &corpus,
120  std::vector<double> &weights,
121  std::uint8_t max_mutation_factor)
122  -> std::enable_if_t<is_state_v<State> && is_full_corpus_v<Corpus>, bool> {
123  if (!state.create_info.config.entropic.enabled) return false;
124  const std::size_t corpus_size = corpus.corpus.size();
125  weights.reserve(corpus_size);
126  const auto average_unit_execution_time =
127  std::accumulate(corpus.corpus.begin(), corpus.corpus.end(),
128  std::chrono::microseconds(0),
129  [](auto sum, const auto &input) {
130  return sum + input.time_of_unit;
131  }) /
132  corpus_size;
133  bool vanilla_schedule = true;
134  utils::ForEachMultiIndexValues<true>(
135  corpus.corpus.template get<Sequential>(), [&](auto &input) {
136  if (input.needs_energy_update && input.energy != 0.0) {
137  input.needs_energy_update = false;
138  input.updateEnergy(
139  state.rare_features.size(),
140  state.create_info.config.entropic.scale_per_exec_time,
141  average_unit_execution_time);
142  }
143  });
144  utils::ForEachMultiIndexValues<false>(
145  corpus.corpus.template get<Sequential>(), [&](auto &input) {
146  if (input.features_count == 0) {
147  // If the seed doesn't represent any features, assign zero energy.
148  weights.push_back(0.);
149  } else if (input.executed_mutations_count / max_mutation_factor >
150  state.executed_mutations_count / corpus.corpus.size()) {
151  // If the seed was fuzzed a lot more than average, assign zero energy.
152  weights.push_back(0.);
153  } else {
154  // Otherwise, simply assign the computed energy.
155  weights.push_back(input.energy);
156  }
157  input.weight = weights.back();
158 
159  // If energy for all seeds is zero, fall back to vanilla schedule.
160  if (weights.back() > 0.0) vanilla_schedule = false;
161  });
162  return !vanilla_schedule;
163 }

186 template <Version llvm_version, typename State, typename Corpus, typename RNG>
187 auto UpdateDistribution(State &state, Corpus &corpus, RNG &rng,
188  std::size_t sparse_energy_updates,
189  std::uint8_t max_mutation_factor,
190  const std::function<void(std::string &&)> &sink)
191  -> std::enable_if_t<is_state_v<State> && is_full_corpus_v<Corpus> &&
192  (llvm_version >= MakeVersion(11u, 0u, 0u)),
193  bool> {
194  if (!state.distribution_needs_update &&
195  (!state.create_info.config.entropic.enabled ||
196  random_value(rng, sparse_energy_updates)))
197  return false;
198 
199  state.distribution_needs_update = false;
200  size_t corpus_size = corpus.corpus.size();
201  assert(corpus_size);
202  std::vector<double> intervals;
203  GenerateIntervals(corpus, intervals);
204  std::vector<double> weights;
205 
206  bool vanilla_schedule = true;
207  if (state.create_info.config.entropic.enabled)
208  vanilla_schedule =
209  !GenerateEntropicSchedule(state, corpus, weights, max_mutation_factor);
210 
211  if (vanilla_schedule) GenerateVanillaSchedule(corpus, weights);
212 
213  if (state.create_info.config.debug) DumpDistribution(corpus, weights, sink);
214 
215  if (std::find_if(weights.begin(), weights.end(),
216  [](auto v) { return v != 0; }) == weights.end())
217  std::fill(weights.begin(), weights.end(), 1);
218 
219  state.corpus_distribution = std::piecewise_constant_distribution<double>(
220  intervals.begin(), intervals.end(), weights.begin());
221  return true;
222 }

284 template <typename State, typename RNG, typename Corpus, typename Range,

286 auto SelectSeed(State &state, Corpus &corpus, RNG &rng, Range &range,
287  InputInfo &exec_result, bool uniform_dist)
288  -> std::enable_if_t<
289  is_state_v<State> && is_full_corpus_v<Corpus> &&
290  is_input_info_v<InputInfo> &&
291  utils::range::is_range_of_v<Range, std::uint8_t>,
292  utils::void_t<decltype(utils::range::assign(
293  std::declval<boost::iterator_range<std::uint8_t *>>(),
294  std::declval<Range &>()))>> {
295  std::size_t index = 0u;
296  if (uniform_dist)
297  index = random_value(rng, corpus.corpus.size());
298  else {
299  auto &dist = state.corpus_distribution;
300  index = dist(rng);
301  }
302  assert(index < corpus.corpus.size());
303  auto selected_testcase = std::next(corpus.corpus.begin(), index);
304  if (selected_testcase->enabled) {
305  const exec_input::ExecInput &selected_input =
306  *corpus.inputs.get_ref(selected_testcase->id);
307  const auto *head = selected_input.GetBuf();
308  const auto length = selected_input.GetLen();
309  utils::range::assign(
310  boost::make_iterator_range(head, std::next(head, length)), range);
311  exec_result = InputInfo();
312  exec_result.executed_mutations_count =
313  selected_testcase->executed_mutations_count;
314  } else {
315  range.clear();
316  exec_result = InputInfo();
317  exec_result.executed_mutations_count =
318  selected_testcase->executed_mutations_count;
319  }
