32 size_t CalcSubTreeSizesAndParentsRecTest(Tree& tree, const NodeID& n,
33  Context& ctx) {
34  NodeID cur(n + 1);
35  size_t size = 1;
36  size_t iter_n = tree.GetRule(n, ctx).NumberOfNonterms();
37 
38  for (size_t i = 0; i < iter_n; i++) {
39  tree.paren()[static_cast<size_t>(cur)] = n;
40  size_t sub_size = CalcSubTreeSizesAndParentsRecTest(tree, cur, ctx);
41  cur = cur + sub_size;
42  size += sub_size;
43  }
44 
45  tree.sizes()[static_cast<size_t>(n)] = size;
46  return size;
47 }
