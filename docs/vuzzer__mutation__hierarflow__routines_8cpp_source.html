38 VUzzerMutCalleeRef Mutate::operator()(void) {
39  DEBUG("Mutate seed(%zu)\n", state.seed_queue.size());
40  assert(state.seed_queue.size() >= 2); // Mutator use at least two seeds from seed_queue
41 
42  std::random_device rd;
43  std::default_random_engine eng(rd());
44  std::uniform_real_distribution<> distr_cut(0.4, 0.8), distr_mut(0.1, 1.0);//XXX:
45  u32 best_len = state.seed_queue[0]->input->GetLen(); //XXX
46 
47  /* TODO: filesTrim */
48  /* TODO: best=set(fitnames[:config.BESTP]) */
49  int limit = state.setting->pop_size - state.setting->next_gen_range;
50 
51  for (int i = 0; i < limit; i+=2) {
52  std::vector<std::shared_ptr<VUzzerTestcase>> parents;
53 
54  /* Choose two seeds from seed_queue. */
55  u32 cutp = (int)(distr_cut(eng) * state.seed_queue.size());
56  std::sample(state.seed_queue.begin() + cutp,
57  state.seed_queue.end(),
58  std::back_inserter(parents),
59  2,
60  std::mt19937{std::random_device{}()});
61 
62  /* If we have seeds in taint_queue, choose them randomly. */
63  if (state.taint_queue.size()) {
64  if ((u32)(rand() % 10) > state.setting->next_gen_from_special_prob) {
65  parents[0] = state.taint_queue[rand() % state.taint_queue.size()];
66  }
67  if ((u32)(rand() % 10) > state.setting->next_gen_from_special_prob) {
68  parents[1] = state.taint_queue[rand() % state.taint_queue.size()];
69  }
70  }
71 
72  DEBUG("Chose %s, %s", parents[0]->input->GetPath().c_str(), parents[1]->input->GetPath().c_str());
73 
74  parents[0]->input->LoadByMmap();
75  parents[1]->input->LoadByMmap();
76 
77  if (parents[0]->input->GetLen() > best_len || parents[1]->input->GetLen() > best_len) {
78  DEBUG("No crossover");
79  auto mutator1 = VUzzerMutator( *parents[0]->input, state );
80  auto mutator2 = VUzzerMutator( *parents[1]->input, state );
81  mutator1.MutateRandom();
82  mutator1.TaintBasedChange();
83  mutator2.MutateRandom();
84  mutator2.TaintBasedChange();
85 
86  /* Save new seeds in pending_queue */
87  std::string fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u",
88  state.setting->out_dir.c_str(),
89  state.queued_paths);
90  state.AddToQueue(state.pending_queue, fn, mutator1.GetBuf(), mutator1.GetLen());
91 
92  fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u",
93  state.setting->out_dir.c_str(),
94  state.queued_paths);
95  state.AddToQueue(state.pending_queue, fn, mutator2.GetBuf(), mutator2.GetLen());
96 
97  } else {
98  DEBUG("Crossover");
99  auto crossover = VUzzerMutator( *parents[0]->input, state );
100  /* FIXME: Do not use ExecInputSet(input_set) */
101  auto seeds = crossover.CrossOver(*parents[1]->input);
102 
103  auto mutator1 = VUzzerMutator( *(seeds.first), state );
104  auto mutator2 = VUzzerMutator( *(seeds.second), state );
105 
106  if (distr_mut(eng) > (1.0 - state.setting->mutate_after_crossover_prob)) {
107  mutator1.MutateRandom();
108  mutator1.TaintBasedChange();
109  } else {
110  mutator1.TaintBasedChange();
111  }
112 
113  /* Save a new seed in pending_queue */
114  std::string fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u,crossover",
115  state.setting->out_dir.c_str(),
116  state.queued_paths);
117  state.AddToQueue(state.pending_queue, fn, mutator1.GetBuf(), mutator1.GetLen());
118 
119  if (distr_mut(eng) > (1.0 - state.setting->mutate_after_crossover_prob)) {
120  mutator2.MutateRandom();
121  mutator2.TaintBasedChange();
122  } else {
123  mutator2.TaintBasedChange();
124  }
125 
126  /* Save a new seed in pending_queue */
127  fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u,crossover",
128  state.setting->out_dir.c_str(),
129  state.queued_paths);
130  state.AddToQueue(state.pending_queue, fn, mutator2.GetBuf(), mutator2.GetLen());
131 
132  }
133 
134  parents[0]->input->Unload();
135  parents[1]->input->Unload();
136 
137  }
138  CallSuccessors();
139  return GoToDefaultNext();
140 }
