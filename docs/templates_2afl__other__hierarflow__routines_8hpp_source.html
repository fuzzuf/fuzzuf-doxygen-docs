156  this->SetResponseValue(true);

159  } else {
160  if (UR(100) < option::GetSkipNfavOldProb(state)) {
161  this->SetResponseValue(true);
162  return this->GoToParent();

164  }
165  }
166  }
167 
168  if (state.not_on_tty) {
169  ACTF("Fuzzing test case #%u (%u total, %llu uniq crashes found)...",
170  state.current_entry, state.queued_paths, state.unique_crashes);
171  fflush(stdout);

174  // We don't use LoadByMmap here because we may modify
175  // the underlying file in calibration and trimming.
176  // The original AFL also should not do that though it does.
177  testcase->input->Load();
178

249 /* Trim all new test cases to save cycles when doing deterministic checks. The
250  trimmer uses power-of-two increments somewhere between 1/16 and 1/1024 of
251  file size, to keep the stage short and sweet. */
252 
253 template<class State>
254 static PUTExitReasonType DoTrimCase(
255  State &state, typename State::OwnTestcase &testcase
256 ) {
257  auto &input = *testcase.input;
258 
259  /* Although the trimmer will be less useful when variable behavior is
260  detected, it will still work to some extent, so we don't check for
261  this. */
262 
263  if (input.GetLen() < 5) return PUTExitReasonType::FAULT_NONE;
264 
265  state.bytes_trim_in += input.GetLen();
266 
267  /* Select initial chunk len, starting with large steps. */
268  u32 len_p2 = Util::NextP2(input.GetLen());
269 
270  u32 remove_len = std::max(
271  len_p2 / option::GetTrimStartSteps(state),
272  option::GetTrimMinBytes(state)
273  );
274 
275  /* Continue until the number of steps gets too high or the stepover
276  gets too small. */
277 
278  // end_len is re-calculated everytime len_p2 is changed
279  u32 end_len = std::max(
280  len_p2 / option::GetTrimEndSteps(state),
281  option::GetTrimMinBytes(state)
282  );
283 
284  PersistentMemoryFeedback pers_feed;
285  u32 trim_exec = 0;
286  bool needs_write = false;
287  PUTExitReasonType fault = PUTExitReasonType::FAULT_NONE;
288  while (remove_len >= end_len) {
289  u32 remove_pos = remove_len;
290 
291  state.stage_name = Util::StrPrintf(
292  "trim %s/%s",
293  afl::util::DescribeInteger(remove_len).c_str(),
294  afl::util::DescribeInteger(remove_len).c_str()
295  );
296 
297  state.stage_cur = 0;
298  state.stage_max = input.GetLen() / remove_len;
299 
300  while (remove_pos < input.GetLen()) {
301  u32 trim_avail = std::min(remove_len, input.GetLen() - remove_pos);
302 
303  // FIXME: we can't represent this in fuzzuf without preparing another buf
304  // write_with_gap(in_buf, q->len, remove_pos, trim_avail);
305 
306  u32 test_len = input.GetLen() - trim_avail;
307  std::unique_ptr<u8[]> test_buf(new u8[test_len]);
308  u32 move_tail = input.GetLen() - remove_pos - trim_avail;
309 
310  std::memcpy(test_buf.get(), input.GetBuf(), remove_pos);
311  std::memmove(
312  test_buf.get() + remove_pos,
313  input.GetBuf() + remove_pos + trim_avail,
314  move_tail
315  );
316 
317  ExitStatusFeedback exit_status;
318 
319  InplaceMemoryFeedback inp_feed = state.RunExecutorWithClassifyCounts(
320  test_buf.get(),
321  test_len,
322  exit_status
323  );
324  fault = exit_status.exit_reason;
325  state.trim_execs++;
326 
327  if ( state.stop_soon
328  || fault == PUTExitReasonType::FAULT_ERROR)
329  goto abort_trimming; //FIXME: goto
330 
331  /* Note that we don't keep track of crashes or hangs here; maybe TODO? */
332 
333  u32 cksum = inp_feed.CalcCksum32();
334 
335  /* If the deletion had no impact on the trace, make it permanent. This
336  isn't perfect for variable-path inputs, but we're just making a
337  best-effort pass, so it's not a big deal if we end up with false
338  negatives every now and then. */
339 
340  if (cksum == testcase.exec_cksum) {
341  input.OverwriteKeepingLoaded(std::move(test_buf), test_len);
342 
343  len_p2 = Util::NextP2(input.GetLen());
344  end_len = std::max(
345  len_p2 / option::GetTrimEndSteps(state),
346  option::GetTrimMinBytes(state)
347  );
348 
349  /* Let's save a clean trace, which will be needed by
350  update_bitmap_score once we're done with the trimming stuff. */
351 
352  if (!needs_write) {
353  needs_write = true;
354  pers_feed = inp_feed.ConvertToPersistent();
355  }
356  } else {
357  remove_pos += remove_len;
358  }
359 
360  /* Since this can be slow, update the screen every now and then. */
361 
362  if (trim_exec % state.stats_update_freq == 0) state.ShowStats();
363  trim_exec++;
364  state.stage_cur++;
365  }
366 
367  remove_len >>= 1;
368  }
369 
370  /* If we have made changes to in_buf, we also need to update the on-disk
371  version of the test case. */
372 
373  if (needs_write) {
374  // already saved on disk by "input.OverwriteKeepingLoaded()"
375 
376  state.UpdateBitmapScoreWithRawTrace(
377  testcase,
378  pers_feed.mem.get(),
379  option::GetMapSize<typename State::Tag>()
380  );
381  }
382 
383 abort_trimming:
384  state.bytes_trim_out += input.GetLen();
385  return fault;
386 }

434 template<class State>
435 AFLMidCalleeRef<State> ApplyDetMutsTemplate<State>::operator()(
436  std::shared_ptr<typename State::OwnTestcase> testcase
437 ) {
438 
439  // We no longer modify this testcase.
440  // So we can reload the file with mmap.
441  testcase->input->LoadByMmap(); // no need to Unload
442 
443  /* Skip right away if -d is given, if we have done deterministic fuzzing on
444  this entry ourselves (fuzz_level > 0), or if it has gone through deterministic
445  testing in earlier, resumed runs (passed_det). */
446 
447  if (state.skip_deterministic || testcase->WasFuzzed() || testcase->passed_det) {
448  state.doing_det = false;
449  return this->GoToDefaultNext();
450  }
451 
452  /* Skip deterministic fuzzing if exec path checksum puts this out of scope
453  for this master instance. */
454 
455  if ( state.master_max
456  && (testcase->exec_cksum % state.master_max) != state.master_id - 1) {
457  state.doing_det = false;
458  return this->GoToDefaultNext();
459  }
460 
461  state.doing_det = true;
462 
463  auto mutator = AFLMutatorTemplate<State>( *testcase->input, state );
464 
465  state.stage_val_type = option::STAGE_VAL_NONE;
466 
467  // this will be required in dictionary construction and eff_map construction
468  state.queue_cur_exec_cksum = testcase->exec_cksum;
469 
470  // call deterministic mutations
471  // if they return true, then we should go to abandon_entry
472  auto should_abandon_entry = this->CallSuccessors(mutator);
473 
474  if (should_abandon_entry) {
475  this->SetResponseValue(true);
476  return abandon_entry;
477  }
478 
479  // NOTE: "if (!testcase->passed_det)" seems unnecessary to me
480  // because passed_det == 0 always holds here
481  if (!testcase->passed_det) state.MarkAsDetDone(*testcase);
482 
483  return this->GoToDefaultNext();
484 }
