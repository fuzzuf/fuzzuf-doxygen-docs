76 RProcessInput ProcessInput::operator()(std::unique_ptr<QueueItem>& inp) {
77  auto& node = this->UnwrapCurrentLinkedNodeRef();
78  auto& succ_nodes = node.succ_nodes;
79 
80  /* Call child node according to the state */
81  if (std::holds_alternative<InitState>(inp->state)) {
82  (*succ_nodes[_initialize_state_idx])(*inp); // initialize_state
83 
84  } else if (std::holds_alternative<DetState>(inp->state)) {
85  (*succ_nodes[_apply_det_muts_idx])(*inp); // apply_det_muts
86 
87  } else if (std::holds_alternative<RandomState>(inp->state)) {
88  (*succ_nodes[_apply_rand_muts_idx])(*inp); // apply_rand_muts
89 
90  } else {
91  throw exceptions::wrong_hierarflow_usage(
92  "Unexpected input state encountered in ProcessInput", __FILE__,
93  __LINE__);
94  }
95 
96  /* Mark as finished.
97  We use `std::move` because `inp` will never be used after this. */
98  state.queue.Finished(std::move(inp));
99 
100  return GoToParent(); // back to select_input_and_switch
101 }
