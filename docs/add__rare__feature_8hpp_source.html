51 auto AddRareFeature(State &state, Corpus &corpus, std::uint32_t index)
52  -> std::enable_if_t<is_state_v<State> && is_full_corpus_v<Corpus>> {
53  // Maintain *at least* TopXRarestFeatures many rare features
54  // and all features with a frequency below ConsideredRare.
55  // Remove all other features.
56  while (state.rare_features.size() >
57  state.create_info.config.entropic.number_of_rarest_features &&
58  state.freq_of_most_abundant_rare_feature >
59  state.create_info.config.entropic.feature_frequency_threshold) {
60  // Find most and second most abbundant feature.
61  std::array<std::uint32_t, 2u> most_abundant_rare_feature_indices{
62  state.rare_features[0], state.rare_features[0]};
63 
64  std::size_t Delete = 0;
65 
66  for (std::size_t i = 0; i < state.rare_features.size(); i++) {
67  std::uint32_t index2 = state.rare_features[i];
68  if (state.global_feature_freqs[index2] >=
69  state.global_feature_freqs[most_abundant_rare_feature_indices[0]]) {
70  most_abundant_rare_feature_indices[1] =
71  most_abundant_rare_feature_indices[0];
72  most_abundant_rare_feature_indices[0] = index2;
73  Delete = i;
74  }
75  }
76 
77  // Remove most abundant rare feature.
78  state.rare_features[Delete] = state.rare_features.back();
79  state.rare_features.pop_back();
80 
81  utils::ForEachMultiIndexValues<false>(
82  corpus.corpus.template get<Sequential>(), [&](auto &input) {
83  if (input.delete_feature_freq(most_abundant_rare_feature_indices[0]))
84  input.needs_energy_update = true;
85  });

87  state.freq_of_most_abundant_rare_feature =
88  state.global_feature_freqs[most_abundant_rare_feature_indices[1]];
89  }

95  input.delete_feature_freq(index);
96

99  if (input.energy > 0.0) {
