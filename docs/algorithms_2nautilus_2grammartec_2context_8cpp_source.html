188 size_t Context::CalcNumOptionsForRule(const RuleID& r) const {
189  size_t res = 1;
190 
191  for (const NTermID& nt_id: GetRule(r).Nonterms()) {
192  size_t v = _nts_to_num_options.find(nt_id) == _nts_to_num_options.end()
193  ? 1
194  : _nts_to_num_options.at(nt_id);
195 
196  if (__builtin_mul_overflow(res, v, &res)) {
197  /* Saturate instead of overflow */
198  res = std::numeric_limits<size_t>::max();
199  break;
200  }
201  }
202 
203  return res;
204 }

210 void Context::CalcNumOptions() {
211  for (auto& elem: _nts_to_rules) {
212  if (_nts_to_num_options.find(elem.first) == _nts_to_num_options.end()) {
213  _nts_to_num_options[elem.first] = elem.second.size();
214  }
215  }
216 
217  bool something_changed;
218  do {
219  something_changed = false;
220 
221  for (size_t i = 0; i < _rules.size(); i++) {
222  RuleID rid(i);
223  size_t num = CalcNumOptionsForRule(rid);
224  const NTermID& nt = GetRule(rid).Nonterm();
225 
226  if (_nts_to_num_options.find(nt) == _nts_to_num_options.end())
227  _nts_to_num_options[nt] = num;
228 
229  /* Update maximum number */
230  if (_nts_to_num_options[nt] < num) {
231  _nts_to_num_options[nt] = num;
232  something_changed = true;
233  }
234 
235  _rules_to_num_options[rid] = num;
236  }
237  } while (something_changed);
238 }

281 void Context::CalcMinLen() {
282  bool something_changed;
283 
284  do {
285  std::vector<RuleID> unknown_rules;
286  unknown_rules.reserve(_rules.size());
287  for (size_t i = 0; i < _rules.size(); i++) {
288  unknown_rules.emplace_back(i);
289  }
290  something_changed = false;
291 
292  while (unknown_rules.size()) {
293  size_t last_len = unknown_rules.size();
294 
295  /* Remove every rule with known minimum length */
296  auto r = std::remove_if(
297  unknown_rules.begin(), unknown_rules.end(),
298  [this, &something_changed](const RuleID& rule) {
299  if (std::optional<size_t> min = CalcMinLenForRule(rule)) {
300  const NTermID& nt = GetRule(rule).Nonterm();
301 
302  if (_nts_to_min_size.find(nt) == _nts_to_min_size.end())
303  _nts_to_min_size[nt] = min.value();
304 
305  if (_nts_to_min_size[nt] > min.value()) {
306  /* Update minimum value */
307  _nts_to_min_size[nt] = min.value();
308  something_changed = true;
309  }
310 
311  _rules_to_min_size[rule] = min.value();
312  return true;
313  }
314 
315  return false;
316  }
317  );
318  unknown_rules.erase(r, unknown_rules.end());
319 
320  if (last_len == unknown_rules.size()) {
321  std::cerr << "Found unproductive rules: (missing base/non recursive case?)" << std::endl;
322  for (RuleID& r: unknown_rules) {
323  std::cerr << GetRule(r).DebugShow(*this) << std::endl;
324  }
325  throw exceptions::fuzzuf_runtime_error(
326  "Broken grammar", __FILE__, __LINE__
327  );
328  }
329  }
330  } while (something_changed);
331 
332  CalcRuleOrder();
333 }
