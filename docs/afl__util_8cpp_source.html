38 u32 UR(u32 limit, int rand_fd) {
39  static u32 rand_cnt;
40  if (rand_fd != -1 && unlikely(!rand_cnt--)) {
41  u32 seed[2];
42  Util::ReadFile(rand_fd, &seed, sizeof(seed));
43  srandom(seed[0]);
44 
45  using option::AFLTag;
46  using option::GetReseedRng;
47  rand_cnt = (GetReseedRng<AFLTag>() / 2) + (seed[1] % GetReseedRng<AFLTag>());
48  }
49  return random() % limit;
50 }

173 // In the following section, we define the probability distribution for the havoc mutation.
174 // The definition consists of the following two steps:
175 // 1. initialize the weights that represent
176 // the probabilities of each case being selected in Havoc.
177 // 2. initialize discrete_distribution with the weights.
178 //
179 // The problem here is that the weights should be changed depending on whether AFL has
180 // extras and auto extras(constant strings included in the dictionaries).
181 // Therefore, we need to define 4 sets of weights, each of which represents the probabilities
182 // in the case where AFL has {some, no} extras and {some, no} auto extras.
183 //
184 // Also, right below, we use constexpr and static variables a lot.
185 // AFL doesn't modify the weights and distributions dynamically,
186 // so we don't want to initialize them more than once.
187 // This is why the following functions use constexpr and are a little bit hard to read.
188 
189 // FIXME: is there any better way than this?
190 
191 
192 // Return the weights that represent the probabilities of each case being selected in Havoc.
193 // Ridiculously, we need a constexpr function just in order to
194 // initialize static arrays with enum constants(i.e. to use a kind of designated initialization)
195 
196 static constexpr std::array<double, NUM_CASE> GetCaseWeights(
197  // we provide 4 types of probability weights, depending on
198  // 1. whether AFL has any extras and,
199  // 2. whether AFL has auto extras
200  bool has_extras,
201  bool has_a_extras
202 ) {
203  std::array<double, NUM_CASE> weights {};
204 
205  // We use constexpr + assignment, instead of designated initialization of C lang
206  // This allows us to easily assign and check the weight of each case in havoc.
207 
208  weights[FLIP1] = 2.0;
209  weights[XOR] = 2.0;
210 
211  weights[DELETE_BYTES] = 4.0; // case 11 ... 12
212 
213  weights[CLONE_BYTES] = 1.5; // UR(4) != 0 in case 13
214  weights[INSERT_SAME_BYTE] = 0.5; // UR(4) == 0 in case 13
215 
216  // The following two cases are the same as the above two cases
217  weights[OVERWRITE_WITH_CHUNK] = 1.5;
218  weights[OVERWRITE_WITH_SAME_BYTE] = 0.5;
219 
220  weights[INT8] = 2.0;
221  weights[INT16_LE] = 1.0; // UR(2) == 1 in case 2
222  weights[INT16_BE] = 1.0; // UR(2) == 0 in case 2
223  weights[INT32_LE] = 1.0; // UR(2) == 1 in case 3
224  weights[INT32_BE] = 1.0; // UR(2) == 0 in case 3
225 
226  // SUB and ADD are the same as INT
227  weights[SUB8] = 2.0;
228  weights[SUB16_LE] = 1.0;
229  weights[SUB16_BE] = 1.0;
230  weights[SUB32_LE] = 1.0;
231  weights[SUB32_BE] = 1.0;
232 
233  weights[ADD8] = 2.0;
234  weights[ADD16_LE] = 1.0;
235  weights[ADD16_BE] = 1.0;
236  weights[ADD32_LE] = 1.0;
237  weights[ADD32_BE] = 1.0;
238 
239  if (has_extras && has_a_extras) {
240  weights[INSERT_EXTRA] = 1.0;
241  weights[OVERWRITE_WITH_EXTRA] = 1.0;
242  weights[INSERT_AEXTRA] = 1.0;
243  weights[OVERWRITE_WITH_AEXTRA] = 1.0;
244  } else if (has_extras) {
245  weights[INSERT_EXTRA] = 2.0;
246  weights[OVERWRITE_WITH_EXTRA] = 2.0;
247  } else if (has_a_extras) {
248  weights[INSERT_AEXTRA] = 2.0;
249  weights[OVERWRITE_WITH_AEXTRA] = 2.0;
250  }
251 
252  return weights;
253 }

255 u32 HavocCaseDistrib(
256  const std::vector<dictionary::AFLDictData>& extras,
257  const std::vector<dictionary::AFLDictData>& a_extras
258 ) {
259 
260  // FIXME: replace this engine with our pRNG later to avoid being flooded with pRNGs.
261  static std::random_device seed_gen;
262  static std::mt19937 engine(seed_gen());
263 
264  // Static part: the following part doesn't run after a fuzzing campaign starts.
265 
266  constexpr std::array<double, NUM_CASE> weight_set[2][2] = {
267  { GetCaseWeights(false, false), GetCaseWeights(false, true) },
268  { GetCaseWeights(true, false), GetCaseWeights(true, true) }
269  };
270 
271  // FIXME: actually, libstdc++'s discrete_distribution doesn't use Walker's alias method...
272  // We should implement it by ourselves...
273  static std::discrete_distribution<u32> dists[2][2] = {
274  { std::discrete_distribution<u32>(weight_set[0][0].begin(), weight_set[0][0].end()),
275  std::discrete_distribution<u32>(weight_set[0][1].begin(), weight_set[0][1].end()) },
276  { std::discrete_distribution<u32>(weight_set[1][0].begin(), weight_set[1][0].end()),
277  std::discrete_distribution<u32>(weight_set[1][1].begin(), weight_set[1][1].end()) },
278  };
279 
280  // Dynamic part: the following part runs during a fuzzing campaign
281 
282  bool has_extras = !extras.empty();
283  bool has_aextras = !a_extras.empty();
284  return dists[has_extras][has_aextras](engine);
285 }
