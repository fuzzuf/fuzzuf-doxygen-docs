135 template<class Tag>
136 u32 Mutator<Tag>::ChooseBlockLen(u32 limit) {
137  using namespace fuzzuf::algorithm;
138 
139  u32 min_value, max_value;
140  u32 rlim = 3ULL;
141 
142  // just an alias of afl::util::UR
143  auto UR = [this](u32 limit) {
144  return afl::util::UR(limit, rand_fd);
145  };
146 
147  switch (UR(rlim)) {
148  case 0: min_value = 1;
149  max_value = afl::option::GetHavocBlkSmall<Tag>();
150  break;
151 
152  case 1: min_value = afl::option::GetHavocBlkSmall<Tag>();
153  max_value = afl::option::GetHavocBlkMedium<Tag>();
154  break;
155  default:
156  if (UR(10)) {
157  min_value = afl::option::GetHavocBlkMedium<Tag>();
158  max_value = afl::option::GetHavocBlkLarge<Tag>();
159  } else {
160  min_value = afl::option::GetHavocBlkLarge<Tag>();
161  max_value = afl::option::GetHavocBlkXl<Tag>();
162  }
163  }
164 
165  if (min_value >= limit) min_value = 1;
166 
167  return min_value + UR(std::min(max_value, limit) - min_value + 1);
168 }

200 template<class Tag>
201 u32 Mutator<Tag>::FlipByte(u32 pos, int n) {
202  /* Flip bytes */
203 
204  // why we don't *(T*)(outbuf + pos) ^= T(1 << n) - 1; ?
205  // it can be unaligned memory access
206  if (n == 1) {
207  outbuf[pos] ^= 0xff;
208  } else if (n == 2) {
209  u16 v = ReadMem<u16>(pos);
210  Overwrite<u16>(pos, (v ^ 0xFFFF) );
211  } else if (n == 4) {
212  u32 v = ReadMem<u32>(pos);
213  Overwrite<u32>(pos, (v ^ 0xFFFFFFFF) );
214  }
215 
216  return 1;
217 }

257 // return true if the splice occurred, and false otherwise
258 template<class Tag>
259 bool Mutator<Tag>::Splice(const ExecInput& target) {
260  auto cmp_len = std::min(len, target.GetLen());
261  auto [f_diff, l_diff] = Util::LocateDiffs(outbuf, target.GetBuf(), cmp_len);
262 
263  if (f_diff < 0 || l_diff < 2 || f_diff == l_diff) return false;
264 
265  /* Split somewhere between the first and last differing byte. */
266 
267  using fuzzuf::algorithm::afl::util::UR;
268  u32 split_at = f_diff + UR(l_diff - f_diff, rand_fd);
269 
270  /* Do the thing. */
271 
272  if (!splbuf) {
273  splbuf = new u8[target.GetLen()];
274  } else if (spl_len < target.GetLen()) {
275  // We reallocate a buffer only when its size is not enough
276  delete[] splbuf;
277  splbuf = new u8[target.GetLen()];
278  }
279 
280  spl_len = target.GetLen();
281  std::memcpy(splbuf, outbuf, split_at);
282  std::memcpy(splbuf+split_at, target.GetBuf()+split_at, spl_len - split_at);
283 
284  std::swap(outbuf, splbuf);
285  std::swap(len, spl_len);
286  return true;
287 }
