18 /*
19  PUTのタイムアウトのために設定されるタイマーが、例えば、SIGALRMによって実現されている場合、「1つのプロセスで1つしかタイマーを設定できない」状況になっている。
20  この状況において、仮に2つFuzzerインスタンスを並列に起動したとすると、2つめのFuzzerインスタンスの起動によって、タイマーがリセットされてしまう。
21  そうすると、1つめのFuzzerインスタンスで実行されているPUTが正常にタイムアウトしない可能性がある。これが発生していないことを確認するのがこのテストの目的。
22  テストの方針を図示すると以下のようになる:
23 
24 正常なケース
25  秒数 FuzzerA FuzzerB
26  0 タイムアウト3秒でsleep 4を実行 PUT実行前に2秒待つ
27  タイマー1をセットする。
28  1
29  2 タイムアウト3秒でsleep
30 4を実行 タイマー2をセットする。 3 タイマー1でタイムアウトする。
31  4
32  5 タイマー2でタイム・アウトする。
33 
34 異常なケース
35  秒数 FuzzerA FuzzerB
36  0 タイムアウト3秒でsleep 4を実行 PUT実行前に2秒待つ
37  タイマー(グローバル)をセットする。
38  1
39  2 タイムアウト3秒でsleep
40 4を実行 タイマーがリセットされる。
41  3
42  4 タイマーがリセットされ残り1秒残っている
43  sleep 4が正常終了してしまう
44  5 タイマーでタイム・アウトする。
45  */

63 // Fuzzerのインスタンスを作り、わざとタイムアウトするPUTの実行を行う
64 // fork server mode, non fork server
65 // mode双方テストするため、引数でどちらのモードを使うか指定
66 // resultにはテストが成功した時true、失敗した時falseが入る
67 void LaunchInstance(bool forksrv, bool &result) {
68  // Create root directory
69  std::string root_dir_template("/tmp/fuzzuf_test.XXXXXX");
70  const auto raw_dirname = mkdtemp(root_dir_template.data());
71  if (!raw_dirname) throw -1;
72  BOOST_CHECK(raw_dirname != nullptr);
73 #ifdef HAS_CXX_STD_FILESYSTEM
74  namespace fs = std::filesystem;
75 #else
76  namespace fs = boost::filesystem;
77 #endif
78  auto root_dir = fs::path(raw_dirname);
79  BOOST_SCOPE_EXIT(&root_dir) {
80  // Executed on this test exit
81  fs::remove_all(root_dir);
82  }
83  BOOST_SCOPE_EXIT_END
84 
85  auto input_dir = root_dir / "input";
86  auto output_dir = root_dir / "output";
87  BOOST_CHECK_EQUAL(fs::create_directory(input_dir), true);
88  // AddSeedを使うので初期シードは必要ない
89  // auto input_path = input_dir / "0";
90  // create_file( input_path.string(), "Hello, World!" );
91 
92  // Create fuzzer instance
93  auto fuzzer = fuzzuf::bindings::python::PythonFuzzer(
94  {"../put_binaries/command_wrapper", "/bin/sleep",
95  "4"}, // 4秒sleepする。ただしタイムアウトは3秒
96  input_dir.native(), output_dir.native(), 3000, 10000, forksrv, true,
97  true // need_afl_cov, need_bb_cov
98  );
99 
100  // Configurate fuzzer
101  fuzzer.SuppressLog();
102 
103  auto id = fuzzer.AddSeed(
104  1, {1}); // どんなシードを与えようがタイムアウトして然るべき
105 
106  // これがfalseならテスト失敗。ただしここではBOOST_CHECK_EQUALは呼ばない。
107  // 子スレッド側で呼ばれるとどうなるかわからないので、ちゃんと親で呼ぶ
108  result = (id == fuzzuf::exec_input::ExecInput::INVALID_INPUT_ID);
109 }

111 // 1プロセスで2つFuzzerインスタンスを作って両者でPUTを実行する（threadを利用する）。
112 // ただし、片方を少し遅延させて起動する。
113 // fork server mode, non fork server
114 // mode双方テストするため、引数でどちらのモードを使うか指定
115 void DelayedLaunchInstances(bool forksrv) {
116  // cd $(dirname $0)
117  MoveToProgramLocation();
118 
119  // std::threadで親スレッド・子スレッド側で2つのFuzzerインスタンスを作る。
120 
121  bool child_result = false;
122  bool parent_result = false;
123 
124  std::thread child_thread(LaunchInstance, forksrv, std::ref(child_result));
125  // 2秒待ってから親も起動。
126  sleep(2);
127  LaunchInstance(forksrv, parent_result);
128  child_thread.join();
129 
130  BOOST_CHECK(parent_result);
131  BOOST_CHECK(child_result);
132 }
