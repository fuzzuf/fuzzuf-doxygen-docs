41 void ChunkStore::AddTree(Tree& tree, Context& ctx) {
42  std::string buffer;
43  size_t id = _trees.size();
44  bool contains_new_chunk = false;
45 
46  for (size_t i = 0; i < tree.Size(); i++) {
47  buffer.clear();
48  if (tree.sizes()[i] > 30) {
49  continue;
50  }
51 
52  NodeID n(i);
53  tree.Unparse(n, ctx, buffer);
54 
55  if (_seen_outputs.find(buffer) == _seen_outputs.end()) {
56  /* This tree has never been seen before */
57  _seen_outputs.insert(buffer);
58  const NTermID& nt = tree.GetRule(n, ctx).Nonterm();
59 
60  if (_nts_to_chunks.find(nt) == _nts_to_chunks.end()) {
61  _nts_to_chunks[nt] = {Chunk(id, n)};
62  } else {
63  _nts_to_chunks[nt].emplace_back(id, n);
64  }
65 
66  /* Save tree to file */
67  std::string filepath = fuzzuf::utils::StrPrintf(
68  "%s/chunks/chunk_%09ld",
69  _work_dir.c_str(), _number_of_chunks++
70  );
71  int fd = fuzzuf::utils::OpenFile(filepath,
72  O_WRONLY | O_CREAT | O_TRUNC,
73  S_IWUSR | S_IRUSR); // 0600
74  if (fd == -1) {
75  throw exceptions::unable_to_create_file(
76  fuzzuf::utils::StrPrintf("Cannot save tree: %s", filepath.c_str()),
77  __FILE__, __LINE__
78  );
79  }
80  fuzzuf::utils::WriteFile(fd, buffer.data(), buffer.size());
81  fuzzuf::utils::CloseFile(fd);
82 
83  contains_new_chunk = true;
84  }
85  }
86 
87  if (contains_new_chunk) {
88  _trees.push_back(tree);
89  }
90 }

104 std::optional<AlternativePair> ChunkStore::GetAlternativeTo(
105  const RuleID& r, Context& ctx
106 ) const {
107  NTermID nt = ctx.GetNT(RuleIDOrCustom(r));
108 
109  if (_nts_to_chunks.find(nt) == _nts_to_chunks.end()) {
110  // TODO: is this correct?
111  return std::nullopt;
112  }
113 
114  const std::vector<Chunk>& chunks = _nts_to_chunks.at(nt);
115 
116  std::vector<Chunk> relevant;
117  for (const Chunk& chunk: chunks) {
118  auto& [tid, nid] = chunk;
119  if (_trees.at(tid).GetRuleID(nid) != r) {
120  relevant.emplace_back(tid, nid);
121  }
122  }
123 
124  if (relevant.size() == 0) {
125  return std::nullopt;
126  }
127 
128  Chunk selected = utils::random::Choose(relevant);
129  return std::make_pair(
130  std::move(std::make_unique<Tree>(_trees.at(selected.first))),
131  selected.second
132  );
133 }
