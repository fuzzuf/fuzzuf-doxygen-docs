38 VUzzerState::VUzzerState(std::shared_ptr<const VUzzerSetting> setting, std::shared_ptr<PinToolExecutor> executor, std::shared_ptr<PolyTrackerExecutor> texecutor)

40  executor( executor ),
41  taint_executor( texecutor ),
42  all_chars_dict( 256 ),
43  high_chars_dict( 128 )

122 void VUzzerState::ReadTestcases(void) {
123  struct dirent **nl;
124 
125  auto in_dir = fs::canonical(setting->in_dir); // Get aboslute path
126  MSG("Scanning '%s'...", in_dir.c_str());
127 
128  /* We use scandir() + alphasort() rather than readdir() because otherwise,
129  the ordering of test cases would vary somewhat randomly and would be
130  difficult to control. */
131 
132  int nl_cnt = fuzzuf::utils::ScanDirAlpha(in_dir.string(), &nl);
133  if (nl_cnt < 0) {
134  MSG("\n" cLRD "[-] " cRST
135  "The input directory does not seem to be valid - try again. The fuzzer needs\n"
136  " one or more test case to start with - ideally, a small file under 1 kB\n"
137  " or so. The cases must be stored as regular files directly in the input\n"
138  " directory.\n");
139 
140  ERROR("Unable to open '%s'", in_dir.c_str());
141  }
142 
143  for (int i=0; i < nl_cnt; i++) {
144  struct stat st;
145 
146  std::string fn = fuzzuf::utils::StrPrintf("%s/%s", in_dir.c_str(), nl[i]->d_name);
147 
148  free(nl[i]); /* not tracked */
149 
150  if (lstat(fn.c_str(), &st) != 0 || access(fn.c_str(), R_OK) != 0) {
151  ERROR("Unable to access '%s'", fn.c_str());
152  }
153 
154  /* This also takes care of . and .. */
155 
156  if (!S_ISREG(st.st_mode) || !st.st_size || fn.find("/README.txt") != std::string::npos) {
157  continue;
158  }
159 
160  #if 0
161  if (st.st_size > VUzzerOption::MAX_FILE) {
162  EXIT("Test case '%s' is too big (%s, limit is %s)",
163  fn.c_str(),
164  VUzzer::util::DescribeMemorySize(st.st_size).c_str(),
165  VUzzer::util::DescribeMemorySize(VUzzerOption::MAX_FILE).c_str()
166  );
167  }
168  #endif
169 
170  AddToQueue(pending_queue, fn, nullptr, (u32)st.st_size);
171  }
172 
173  free(nl); /* not tracked */
174 
175  if (pending_queue.empty()) {
176  MSG("\n" cLRD "[-] " cRST
177  "Looks like there are no valid test cases in the input directory! The fuzzer\n"
178  " needs one or more test case to start with - ideally, a small file under\n"
179  " 1 kB or so. The cases must be stored as regular files directly in the\n"
180  " input directory.\n");
181 
182  EXIT("No usable test cases in '%s'", in_dir.c_str());
183  }
184 
185 }
