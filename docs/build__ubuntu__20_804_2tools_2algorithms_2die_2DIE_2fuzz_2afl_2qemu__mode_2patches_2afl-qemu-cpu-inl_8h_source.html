162 static void afl_forkserver(CPUState *cpu) {
163 
164  static unsigned char tmp[4];
165 
166  if (!afl_area_ptr) return;
167 
168  /* Tell the parent that we're alive. If the parent doesn't want
169  to talk, assume that we're not running in forkserver mode. */
170 
171  if (write(FORKSRV_FD + 1, tmp, 4) != 4) return;
172 
173  afl_forksrv_pid = getpid();
174 
175  /* All right, let's await orders... */
176 
177  while (1) {
178 
179  pid_t child_pid;
180  int status, t_fd[2];
181 
182  /* Whoops, parent dead? */
183 
184  if (read(FORKSRV_FD, tmp, 4) != 4) exit(2);
185 
186  /* Establish a channel with child to grab translation commands. We'll
187  read from t_fd[0], child will write to TSL_FD. */
188 
189  if (pipe(t_fd) || dup2(t_fd[1], TSL_FD) < 0) exit(3);
190  close(t_fd[1]);
191 
192  child_pid = fork();
193  if (child_pid < 0) exit(4);
194 
195  if (!child_pid) {
196 
197  /* Child process. Close descriptors and run free. */
198 
199  afl_fork_child = 1;
200  close(FORKSRV_FD);
201  close(FORKSRV_FD + 1);
202  close(t_fd[0]);
203  return;
204 
205  }
206 
207  /* Parent. */
208 
209  close(TSL_FD);
210 
211  if (write(FORKSRV_FD + 1, &child_pid, 4) != 4) exit(5);
212 
213  /* Collect translation requests until child dies and closes the pipe. */
214 
215  afl_wait_tsl(cpu, t_fd[0]);
216 
217  /* Get and relay exit status to parent. */
218 
219  if (waitpid(child_pid, &status, 0) < 0) exit(6);
220  if (write(FORKSRV_FD + 1, &status, 4) != 4) exit(7);
221 
222  }
223 
224 }

229 static inline void afl_maybe_log(abi_ulong cur_loc) {
230 
231  static __thread abi_ulong prev_loc;
232 
233  /* Optimize for cur_loc > afl_end_code, which is the most likely case on
234  Linux systems. */
235 
236  if (cur_loc > afl_end_code || cur_loc < afl_start_code || !afl_area_ptr)
237  return;
238 
239  /* Looks like QEMU always maps to fixed locations, so ASAN is not a
240  concern. Phew. But instruction addresses may be aligned. Let's mangle
241  the value to get something quasi-uniform. */
242 
243  cur_loc = (cur_loc >> 4) ^ (cur_loc << 8);
244  cur_loc &= MAP_SIZE - 1;
245 
246  /* Implement probabilistic instrumentation by looking at scrambled block
247  address. This keeps the instrumented locations stable across runs. */
248 
249  if (cur_loc >= afl_inst_rms) return;
250 
251  afl_area_ptr[cur_loc ^ prev_loc]++;
252  prev_loc = cur_loc >> 1;
253 
254 }

257 /* This code is invoked whenever QEMU decides that it doesn't have a

262 static void afl_request_tsl(target_ulong pc, target_ulong cb, uint64_t flags) {
263 
264  struct afl_tsl t;
265 
266  if (!afl_fork_child) return;
267 
268  t.pc = pc;
269  t.cs_base = cb;
270  t.flags = flags;
271 
272  if (write(TSL_FD, &t, sizeof(struct afl_tsl)) != sizeof(struct afl_tsl))
273  return;
274 
275 }

277 /* This is the other side of the same channel. Since timeouts are handled by
278  afl-fuzz simply killing the child, we can just wait until the pipe breaks. */

280 static void afl_wait_tsl(CPUState *cpu, int fd) {
281 
282  struct afl_tsl t;
283  TranslationBlock *tb;
284 
285  while (1) {
286 
287  /* Broken pipe means it's time to return to the fork server routine. */
288 
289  if (read(fd, &t, sizeof(struct afl_tsl)) != sizeof(struct afl_tsl))
290  break;
291 
292  tb = tb_htable_lookup(cpu, t.pc, t.cs_base, t.flags);
293 
294  if(!tb) {
295  mmap_lock();
296  tb_lock();
297  tb_gen_code(cpu, t.pc, t.cs_base, t.flags, 0);
298  mmap_unlock();
299  tb_unlock();
300  }
301 
302  }
303 
304  close(fd);
