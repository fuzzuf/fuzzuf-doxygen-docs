60 template <class State>
61 void AFLplusplusCustomCases<State>::operator()(
62  u32 case_idx, u8*& outbuf, u32& len,
63  [[maybe_unused]] const std::vector<afl::dictionary::AFLDictData>& extras,
64  [[maybe_unused]] const std::vector<afl::dictionary::AFLDictData>&
65  a_extras) {
66  auto UR = [](u32 limit) { return afl::util::UR(limit, -1); };
67 
68  switch (case_idx) {
69  case AFLPLUSPLUS_ADDBYTE:
70  outbuf[UR(len)]++;
71  break;
72 
73  case AFLPLUSPLUS_SUBBYTE:
74  outbuf[UR(len)]--;
75  break;
76 
77  case AFLPLUSPLUS_SWITCH_BYTES: {
78  if (len < 4) {
79  break;
80  }
81 
82  u32 to_end, switch_to, switch_len, switch_from;
83  switch_from = UR(len);
84  do {
85  switch_to = UR(len);
86  } while (switch_from == switch_to);
87 
88  if (switch_from < switch_to) {
89  switch_len = switch_to - switch_from;
90  to_end = len - switch_to;
91  } else {
92  switch_len = switch_from - switch_to;
93  to_end = len - switch_from;
94  }
95 
96  switch_len = ChooseBlockLen(std::min(switch_len, to_end));
97 
98  std::unique_ptr<u8[]> new_buf(new u8[switch_len]);
99 
100  /* Backup */
101  memcpy(new_buf.get(), outbuf + switch_from, switch_len);
102 
103  /* Switch 1 */
104  memcpy(outbuf + switch_from, outbuf + switch_to, switch_len);
105 
106  /* Switch 2 */
107  memcpy(outbuf + switch_to, new_buf.get(), switch_len);
108 
109  break;
110  }
111 
112  case AFLPLUSPLUS_SPLICE_OVERWRITE:
113  case AFLPLUSPLUS_SPLICE_INSERT: {
114  if (state.case_queue.size() <= 1) break;
115 
116  u32 tid;
117  do {
118  tid = UR(state.queued_paths);
119  } while (tid == state.current_entry);
120 
121  if (tid == state.case_queue.size()) break;
122 
123  auto& target_case = *state.case_queue[tid];
124  target_case.input->Load();
125 
126  u32 target_len = target_case.input->GetLen();
127 
128  using afl::option::GetHavocBlkXl;
129  using afl::option::GetMaxFile;
130  using Tag = typename State::Tag;
131 
132  if (case_idx == AFLPLUSPLUS_SPLICE_OVERWRITE ||
133  len + GetHavocBlkXl<Tag>() >= GetMaxFile<Tag>()) {
134  u32 copy_len = ChooseBlockLen(target_len);
135  if (copy_len > len) copy_len = len;
136 
137  u32 copy_from = UR(target_len - copy_len + 1);
138  u32 copy_to = UR(len - copy_len + 1);
139 
140  memmove(outbuf + copy_to, target_case.input->GetBuf() + copy_from,
141  copy_len);
142  } else {
143  u32 clone_len = ChooseBlockLen(target_len);
144  u32 clone_from = UR(target_len - clone_len + 1);
145  u32 clone_to = UR(len + 1);
146 
147  u8* new_buf = new u8[len + clone_len];
148 
149  /* Head */
150  std::memcpy(new_buf, outbuf, clone_to);
151 
152  /* Inserted part */
153  std::memcpy(new_buf + clone_to,
154  target_case.input->GetBuf() + clone_from, clone_len);
155 
156  /* Tail */
157  std::memcpy(new_buf + clone_to + clone_len, outbuf + clone_to,
158  len - clone_to);
159 
160  delete[] outbuf;
161  outbuf = new_buf;
162  len += clone_len;
163  }
164 
165  target_case.input->Unload();
166 
167  break;
168  }
169 
170  default:
171  break;
172  }
173 }
