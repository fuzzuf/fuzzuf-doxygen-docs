62 // Used only for CLI
63 template <class TFuzzer, class TAFLFuzzer, class TExecutor>
64 std::unique_ptr<TFuzzer> BuildAFLFastFuzzerFromArgs(
65  FuzzerArgs &fuzzer_args, GlobalFuzzerOptions &global_options) {
66  po::positional_options_description pargs_desc;
67  pargs_desc.add("fuzzer", 1);
68  pargs_desc.add("pargs", -1);
69 
70  AFLFastFuzzerOptions aflfast_options;
71 
72  po::options_description fuzzer_desc("AFLFast options");
73  std::vector<std::string> pargs;
74  fuzzer_desc.add_options()(
75  "forksrv",
76  po::value<bool>(&aflfast_options.forksrv)
77  ->default_value(aflfast_options.forksrv),
78  "Enable/disable fork server mode. default is true.")(
79  "dict_file,x",
80  po::value<std::vector<std::string>>(&aflfast_options.dict_file)
81  ->composing(),
82  "Load additional dictionary file.")
83  // If you want to add fuzzer specific options, add options here
84  ("pargs", po::value<std::vector<std::string>>(&pargs),
85  "Specify PUT and args for PUT.")(
86  "frida",
87  po::value<bool>(&aflfast_options.frida_mode)
88  ->default_value(aflfast_options.frida_mode),
89  "Enable/disable frida mode. Default to false.");
90 
91  po::variables_map vm;
92  po::store(
93  po::command_line_parser(fuzzer_args.argc, fuzzer_args.argv)
94  .options(fuzzer_args.global_options_description.add(fuzzer_desc))
95  .positional(pargs_desc)
96  .run(),
97  vm);
98  po::notify(vm);
99 
100  if (global_options.help) {
101  fuzzuf::cli::fuzzer::aflfast::usage(fuzzer_args.global_options_description);
102  }
103 
104  using fuzzuf::algorithm::afl::option::GetMemLimit;
105  using fuzzuf::algorithm::aflfast::option::AFLFastTag;
106 
107  u32 mem_limit =
108  global_options.exec_memlimit.value_or(GetMemLimit<AFLFastTag>());
109  if (aflfast_options.frida_mode) {
110  setenv("__AFL_DEFER_FORKSRV", "1", 1);
111  fs::path frida_bin =
112  fs::path(fuzzer_args.argv[0]).parent_path() / "afl-frida-trace.so";
113  setenv("LD_PRELOAD", frida_bin.c_str(), 1);
114 
115  if (mem_limit > 0) {
116  struct stat statbuf;
117  if ((stat(frida_bin.c_str(), &statbuf)) == -1) {
118  std::cerr
119  << cLRD << "[-] File afl-frida-trace.so not found\n"
120  << " Build one first with cmake where -DENABLE_FRIDA_TRACE=1"
121  << cRST << std::endl;
122  }
123  // Need to add the size of the library
124  mem_limit += statbuf.st_size;
125  }
126  }
127 
128  PutArgs put(pargs);
129  try {
130  put.Check();
131  } catch (const exceptions::cli_error &e) {
132  std::cerr << "[!] " << e.what() << std::endl;
133  std::cerr << "\tat " << e.file << ":" << e.line << std::endl;
134  fuzzuf::cli::fuzzer::aflfast::usage(fuzzer_args.global_options_description);
135  }
136 
137  // Trace level log
138  DEBUG("[*] PUT: put = [");
139  for (auto v : put.Args()) {
140  DEBUG("\t\"%s\",", v.c_str());
141  }
142  DEBUG(" ]");
143 
144  using fuzzuf::algorithm::afl::option::GetExecTimeout;
145  using fuzzuf::algorithm::aflfast::AFLFastSetting;
146  using fuzzuf::algorithm::aflfast::option::FAST;
147 
148  // Create AFLFastSetting
149 
150  auto setting = std::make_shared<const AFLFastSetting>(
151  put.Args(), global_options.in_dir, global_options.out_dir,
152  global_options.exec_timelimit_ms.value_or(GetExecTimeout<AFLFastTag>()),
153  mem_limit, aflfast_options.forksrv,
154  /* dumb_mode */ false, // FIXME: add dumb_mode
155  fuzzuf::utils::CPUID_BIND_WHICHEVER, FAST);
156 
157  // NativeLinuxExecutor needs the directory specified by "out_dir" to be
158  // already set up so we need to create the directory first, and then
159  // initialize Executor
160  fuzzuf::utils::SetupDirs(setting->out_dir.string());
161 
162  using fuzzuf::algorithm::afl::option::GetDefaultOutfile;
163  using fuzzuf::algorithm::afl::option::GetMapSize;
164  using fuzzuf::cli::ExecutorKind;
165 
166  std::shared_ptr<TExecutor> executor;
167  switch (global_options.executor) {
168  case ExecutorKind::NATIVE: {
169  auto nle = std::make_shared<fuzzuf::executor::NativeLinuxExecutor>(
170  setting->argv, setting->exec_timelimit_ms, setting->exec_memlimit,
171  setting->forksrv, setting->out_dir / GetDefaultOutfile<AFLFastTag>(),
172  GetMapSize<AFLFastTag>(), // afl_shm_size
173  0 // bb_shm_size
174  );
175  executor = std::make_shared<TExecutor>(std::move(nle));
176  break;
177  }
178 
179  case ExecutorKind::QEMU: {
180  // NOTE: Assuming GetMapSize<AFLFastTag>() == QEMUExecutor::QEMU_SHM_SIZE
181  auto qe = std::make_shared<fuzzuf::executor::QEMUExecutor>(
182  global_options.proxy_path.value(), setting->argv,
183  setting->exec_timelimit_ms, setting->exec_memlimit, setting->forksrv,
184  setting->out_dir / GetDefaultOutfile<AFLFastTag>());
185  executor = std::make_shared<TExecutor>(std::move(qe));
186  break;
187  }
188 
189 #ifdef __aarch64__
190  case ExecutorKind::CORESIGHT: {
191  auto cse = std::make_shared<fuzzuf::executor::CoreSightExecutor>(
192  global_options.proxy_path.value(), setting->argv,
193  setting->exec_timelimit_ms, setting->exec_memlimit, setting->forksrv,
194  setting->out_dir / GetDefaultOutfile<AFLFastTag>(),
195  GetMapSize<AFLFastTag>() // afl_shm_size
196  );
197  executor = std::make_shared<TExecutor>(std::move(cse));
198  break;
199  }
200 #endif
201 
202  default:
203  EXIT("Unsupported executor: '%s'", global_options.executor.c_str());
204  }
205 
206  auto mutop_optimizer = std::unique_ptr<optimizer::Optimizer<u32>>(
207  new algorithm::afl::AFLHavocCaseDistrib());
208 
209  // Create AFLFastState
210  using fuzzuf::algorithm::aflfast::AFLFastState;
211  auto state = std::make_unique<AFLFastState>(setting, executor,
212  std::move(mutop_optimizer));
213 
214  // Load dictionary
215  for (const auto &d : aflfast_options.dict_file) {
216  using fuzzuf::algorithm::afl::dictionary::AFLDictData;
217 
218  const std::function<void(std::string &&)> f = [](std::string s) {
219  ERROR("Dictionary error: %s", s.c_str());
220  };
221 
222  fuzzuf::algorithm::afl::dictionary::load(d, state->extras, false, f);
223  }
224  fuzzuf::algorithm::afl::dictionary::SortDictByLength(state->extras);
225 
226  return std::unique_ptr<TFuzzer>(
227  dynamic_cast<TFuzzer *>(new TAFLFuzzer(std::move(state))));
228 }
