78 template <typename Dict>
79 void UpdateDictionary(
80  Dict &dict, std::vector<utils::range::RangeValueT<Dict> *> &dict_entries) {
81  std::vector<utils::range::RangeValueT<Dict>> copied_dict_entries;
82  copied_dict_entries.reserve(dict_entries.size());
83  std::transform(dict_entries.begin(), dict_entries.end(),
84  std::back_inserter(copied_dict_entries), [](const auto &p) {
85  // PersistentAutoDictionary.AddWithSuccessCountOne(DE);
86  p->increment_success_count();
87  assert(p->get().size());
88  return *p;
89  });
90  for (auto &dict_entry : copied_dict_entries) {
91  const auto &new_word = dict_entry.get();
92  if (std::find_if(dict.begin(), dict.end(), [&](auto &v) {
93  if (dict_entry.get_hash() != v.get_hash())
94  return false;
95  const auto &existing_word = v.get();
96  // Linear search is fine here as this happens seldom.
97  return std::equal(existing_word.begin(), existing_word.end(),
98  new_word.begin(), new_word.end());
99  }) == dict.end()) {
100  dict_entry.reset_use_count();
101  utils::range::append(dict_entry, dict);
102  }
103  }
