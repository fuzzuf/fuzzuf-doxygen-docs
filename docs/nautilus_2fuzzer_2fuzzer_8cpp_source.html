67 void NautilusFuzzer::BuildFuzzFlow() {
68  using fuzzuf::hierarflow::CreateNode;
69  using fuzzuf::hierarflow::CreateIrregularNode;
70  using namespace fuzzuf::algorithm::nautilus::fuzzer::routine::other;
71  using namespace fuzzuf::algorithm::nautilus::fuzzer::routine::mutation;
72  using namespace fuzzuf::algorithm::nautilus::fuzzer::routine::update;
73 
74  fuzz_loop = CreateNode<FuzzLoop>(*state);
75 
76  /* Mutation flow */
77  auto mut_rules = CreateNode<MutRules>(*state);
78  auto splice = CreateNode<MutSplice>(*state);
79  auto havoc = CreateNode<MutHavoc>(*state);
80  auto havoc_rec = CreateNode<MutHavocRec>(*state);
81 
82  /* Processing flow */
83  auto initialize_state = CreateNode<InitializeState>(*state);
84  auto apply_det_muts = CreateNode<ApplyDetMuts>(*state);
85  auto apply_rand_muts = CreateNode<ApplyRandMuts>(*state);
86 
87  /* Main flow */
88  auto process_next_input = CreateIrregularNode<ProcessInput>(
89  *state,
90  initialize_state.GetCalleeIndexRef(),
91  apply_det_muts.GetCalleeIndexRef(),
92  apply_rand_muts.GetCalleeIndexRef()
93  );
94  auto generate_input = CreateNode<GenerateInput>(*state);
95  auto select_input_and_switch = CreateIrregularNode<SelectInput>(
96  *state,
97  process_next_input.GetCalleeIndexRef(),
98  generate_input.GetCalleeIndexRef()
99  );
100  auto update_state = CreateNode<UpdateState>(*state);
101 
102  fuzz_loop << (
103  select_input_and_switch <= (
104  process_next_input
105  || generate_input
106  )
107  || update_state
108  );
109 
110  process_next_input <= (
111  initialize_state
112  || apply_det_muts << (
113  mut_rules
114  || splice.HardLink()
115  || havoc.HardLink()
116  || havoc_rec.HardLink()
117  )
118  || apply_rand_muts << (
119  splice.HardLink()
120  || havoc.HardLink()
121  || havoc_rec.HardLink()
122  )
123  );
124 }

151  }
152

154  if (!fs::exists(setting->path_to_grammar)) {

184 bool NautilusFuzzer::ParseAndAddRule(Context& ctx,
185  std::string nt,
186  json rule,
187  bool recursive /*=false*/) {
188  if (rule.is_string()) {
189  /* Simple string rule: just add it */
190  ctx.AddRule(nt, rule.get<std::string>());
191  return true;
192 
193  } else if (rule.is_array()) {
194  /* Array rule: Union or binary */
195 
196  /* Check if every rule is integer or string */
197  if (std::all_of(rule.begin(), rule.end(),
198  [](const json& e) {
199  return e.is_number_integer() || e.is_string();
200  })) {
201  /* This is a binary rule */
202  std::string r;
203  for (const json& e: rule) {
204  if (e.is_string()) {
205  /* Simply concat string */
206  r += e.get<std::string>();
207 
208  } else {
209  /* Read as a character code */
210  size_t c = e.get<size_t>();
211  if (c >= 0x100) {
212  std::cerr << "[-] Invalid character code" << std::endl
213  << " NT : " << nt << std::endl
214  << " RULE: " << rule << std::endl
215  << " The value " << c << " is out-of-range." << std::endl;
216  std::exit(1);
217  }
218 
219  r.push_back((char)c);
220  }
221  }
222 
223  ctx.AddRule(nt, r);
224  return true;
225 
226  } else if (!recursive) {
227  /* This is a union rule */
228  for (const json& e: rule) {
229  if (!NautilusFuzzer::ParseAndAddRule(ctx, nt, e, true)) {
230  std::cerr << "[-] Invalid rule" << std::endl
231  << " NT : " << nt << std::endl
232  << " RULES: " << rule << std::endl
233  << " RULE : " << e << std::endl;
234  std::exit(1);
235  }
236  }
237  return true;
238 
239  } else {
240  /* Invalid recursive array */
241  return false;
242  }
243 
244  }
245 
246  /* Invalid type */
247  std::cerr << "[-] Invalid rule" << std::endl
248  << " NT : " << nt << std::endl
249  << " RULE: " << rule << std::endl;
250  std::exit(1);
251 }

259 void NautilusFuzzer::LoadGrammar(Context& ctx, fs::path grammar_path) {
260  /* Create new context and save it */
261  if (grammar_path.extension() == ".json") {
262  /* Check file */
263  if (!fs::exists(grammar_path) || fs::is_directory(grammar_path)) {
264  std::cerr << "[-] Grammar file does not exist or not a file." << std::endl
265  << " Path: " << grammar_path << std::endl;
266  std::exit(1);
267  }
268 
269  /* Load JSON grammar */
270  std::ifstream ifs(grammar_path.string());
271  try {
272  json rules = json::parse(ifs);
273 
274  if (!rules.is_array()) {
275  std::cerr << "[-] Invalid rules (Rules must be array)" << std::endl;
276  std::exit(1);
277  } else if (rules.size() == 0) {
278  std::cerr << "[-] Rule file doesn't include any rules" << std::endl;
279  std::exit(1);
280  } else if (!rules[0].is_array()
281  || rules[0].size() != 2
282  || !rules[0].get<json>()[0].is_string()) {
283  std::cerr << "[-] First rule is invalid" << std::endl
284  << " It must be an array with 2 elements. The first" << std::endl
285  << " one must be a string representing the name" << std::endl
286  << " of the nonterminal." << std::endl;
287  std::exit(1);
288  }
289 
290  /* Add rules */
291  std::string root = "{" + rules[0].get<json>()[0].get<std::string>() + "}";
292  NautilusFuzzer::ParseAndAddRule(ctx, "START", root);
293 
294  for (auto& rule: rules) {
295  if (!rule.is_array()
296  || rule.size() != 2
297  || !rule[0].is_string()) {
298  std::cerr << "[-] Invalid rule" << std::endl
299  << " It must be an array with 2 elements. The first" << std::endl
300  << " one must be a string representing the name" << std::endl
301  << " of the nonterminal." << std::endl;
302  std::cerr << " RULE: " << rule << std::endl;
303  std::exit(1);
304  }
305 
306  NautilusFuzzer::ParseAndAddRule(ctx,
307  rule[0].get<std::string>(),
308  rule[1].get<json>());
309  }
310 
311  } catch (std::exception& e) {
312  /* JSON parse error */
313  std::cerr << "[-] Cannot parse grammar file" << std::endl
314  << e.what() << std::endl;
315  std::exit(1);
316  }
317 
318  } else if (grammar_path.extension() == ".py") {
319  /* TODO: Support Python-written grammar */
320  throw exceptions::not_implemented(
321  "Grammar defined in Python is not supported yet", __FILE__, __LINE__
322  );
323 
324  } else {
325  throw exceptions::fuzzuf_runtime_error(
326  "Unknown grammar type ('.json' expected)", __FILE__, __LINE__
327  );
328  }
329 }
