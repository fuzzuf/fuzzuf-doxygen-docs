102 std::optional< std::pair< BigInt, BigInt > > SolveAux(
103  const Fraction &slope,
104  BigInt x0,
105  BigInt y0,
106  Signedness /*sign*/,
107  BigInt target_y
108 ) {
109  const auto candidate = x0 + ( target_y - y0 ) * slope.denominator() / slope.numerator();
110  const auto check_y = y0 + ( candidate - x0 ) * slope.numerator() / slope.denominator();
111  if ( target_y == check_y ) {
112  return std::make_pair( candidate - 1, candidate + 1 );
113  }
114  else if ( check_y > target_y && slope.numerator() > 0 ) {
115  return std::make_pair( candidate - 1, candidate );
116  }
117  else if ( check_y > target_y && slope.numerator() < 0 ) {
118  return std::make_pair( candidate, candidate + 1 );
119  }
120  else if ( check_y < target_y && slope.numerator() > 0 ) {
121  return std::make_pair( candidate, candidate + 1 );
122  }
123  else if ( check_y < target_y && slope.numerator() < 0 ) {
124  return std::make_pair( candidate - 1, candidate );
125  }
126  else return std::nullopt;
127 }

129 boost::container::static_vector< std::pair< BigInt, BigInt >, 3u > Solve(
130  const Fraction &slope,
131  BigInt x0,
132  BigInt y0,
133  BigInt target_y,
134  std::size_t chunk_size,
135  std::size_t cmp_size,
136  Signedness sign
137 ) {
138  std::array< BigInt, 3u > target_ys;
139  if( sign == Signedness::Signed ) {
140  const BigInt signed_wrap = GetSignedMax( cmp_size ) + BigInt( 1 );
141  target_ys[ 0 ] = -signed_wrap;
142  target_ys[ 1 ] = target_y;
143  target_ys[ 2 ] = signed_wrap;
144  }
145  else if( sign == Signedness::Unsigned ) {
146  const BigInt unsigned_wrap = GetUnsignedMax( cmp_size ) + BigInt( 1 );
147  target_ys[ 0 ] = 0;
148  target_ys[ 1 ] = target_y;
149  target_ys[ 2 ] = unsigned_wrap;
150  }
151  else {
152  throw exceptions::invalid_argument( "Unknown signedness", __FILE__, __LINE__ );
153  }
154  boost::container::static_vector< std::pair< BigInt, BigInt >, 3u > solved;
155  for( const auto &y: target_ys ) {
156  const auto solved_maybe = SolveAux( slope, x0, y0, sign, y );
157  if( solved_maybe ) {
158  solved.push_back( *solved_maybe );
159  }
160  }
161  std::sort( solved.begin(), solved.end() );
162  auto unique_end = std::unique( solved.begin(), solved.end() );
163  solved.erase( std::remove_if(
164  solved.begin(),
165  unique_end,
166  [chunk_size]( const auto &v ) {
167  const auto &[low,high] = v;
168  return !(
169  0 <= low &&
170  high <= GetUnsignedMax( chunk_size )
171  );
172  }
173  ), solved.end() );
174  return solved;
175 }

201 Result FindAsNByteChunk(
202  const Context &ctx,
203  Endian endian,
204  std::size_t chunk_size,
205  const BranchInfo &br_info1,
206  const BranchInfo &br_info2,
207  const BranchInfo &br_info3
208 ) {
209  const std::size_t cmp_size = br_info1.operand_size;
210  const auto sign = ( br_info1.branch_type == CompareType::SignedSize ) ? Signedness::Signed : Signedness::Unsigned;
211  if( ctx.bytes.size() < chunk_size - 1 ) {
212  failwith( "Invalid size" );
213  return Result{}; // unreachable
214  }
215  const auto x1 = BytesToBigInt( endian, ConcatBytes( chunk_size, br_info1, ctx ) );
216  const auto x2 = BytesToBigInt( endian, ConcatBytes( chunk_size, br_info2, ctx ) );
217  const auto x3 = BytesToBigInt( endian, ConcatBytes( chunk_size, br_info3, ctx ) );
218  if( br_info1.operand1 == br_info2.operand1 && br_info2.operand1 == br_info3.operand1 ) {
219  const auto y1 = branch_info::InterpretAs( sign, cmp_size, br_info1.operand2 );
220  const auto y2 = branch_info::InterpretAs( sign, cmp_size, br_info2.operand2 );
221  const auto y3 = branch_info::InterpretAs( sign, cmp_size, br_info3.operand2 );
222  const auto slope = FindCommonSlope( cmp_size, x1, x2, x3, y1, y2, y3 );
223  if( slope.numerator() == 0 ) {
224  return NonLinear();
225  }
226  else {
227  const auto target_y = BigInt( br_info1.operand1 );
228  return Generate( endian, chunk_size, cmp_size, slope, target_y, x1, y1, sign );
229  }
230  }
231  else if( br_info1.operand2 == br_info2.operand2 && br_info2.operand2 == br_info3.operand2 ) {
232  const auto y1 = branch_info::InterpretAs( sign, cmp_size, br_info1.operand1 );
233  const auto y2 = branch_info::InterpretAs( sign, cmp_size, br_info2.operand1 );
234  const auto y3 = branch_info::InterpretAs( sign, cmp_size, br_info3.operand1 );
235  const auto slope = FindCommonSlope( cmp_size, x1, x2, x3, y1, y2, y3 );
236  if( slope.numerator() == 0 ) {
237  return NonLinear();
238  }
239  else {
240  const auto target_y = BigInt( br_info1.operand2 );
241  return Generate( endian, chunk_size, cmp_size, slope, target_y, x1, y1, sign );
242  }
243  }
244  else {
245  return NonLinear();
246  }
247 }

249 std::optional< Solvable > FindAux(
250  const Context &ctx,
251  const std::array< std::pair< Endian, std::size_t >, 7u >::const_iterator &types_begin,
252  const std::array< std::pair< Endian, std::size_t >, 7u >::const_iterator &types_end,
253  const std::vector< BranchInfo > &br_info_triple
254 ) {
255  if( types_begin == types_end ) {
256  return std::nullopt;
257  }
258  const auto &[endian,chunk_size] = *types_begin;
259  return std::visit(
260  [&]( const auto &v ) -> std::optional< Solvable > {
261  if constexpr ( std::is_same_v< utils::type_traits::RemoveCvrT< decltype( v ) >, NonLinear > ) {
262  return std::nullopt;
263  }
264  else if constexpr ( std::is_same_v< utils::type_traits::RemoveCvrT< decltype( v ) >, Unsolvable > ) {
265  return FindAux( ctx, std::next( types_begin ), types_end, br_info_triple );
266  }
267  else if constexpr ( std::is_same_v< utils::type_traits::RemoveCvrT< decltype( v ) >, Solvable > ) {
268  return v;
269  }
270  else {
271  return std::nullopt;
272  }
273  },
274  FindAsNByteChunk(
275  ctx,
276  endian,
277  chunk_size,
278  br_info_triple[ 0 ],
279  br_info_triple[ 1 ],
280  br_info_triple[ 2 ]
281  )
282  );
283 }

286  const Context &ctx,
287  const std::vector< BranchInfo > &br_info_triple
288 ) {
289  constexpr std::array< std::pair< Endian, std::size_t >, 7u > types{{
290  { Endian::BE, 1u },
291  { Endian::BE, 2u },
292  { Endian::LE, 2u },
293  { Endian::BE, 4u },
294  { Endian::LE, 4u },

296  { Endian::LE, 8u }
297  }};
298  const auto max_len = ctx.bytes.size() + 1u;
299  const auto types_end = std::find_if(
300  types.begin(),
301  types.end(),
302  [max_len]( const auto &v ) {
303  return v.second > max_len;
304  }
305  );
306  return FindAux( ctx, types.begin(), types_end, br_info_triple );
