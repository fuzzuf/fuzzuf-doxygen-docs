69  LinuxForkServerExecutorParameters &&args)
70  : Executor(args.argv, args.exec_timelimit_ms, args.exec_memlimit,
71  args.path_to_write_input.string()),

77  record_stdout_and_err(args.record_stdout_and_err),
78  filesystem(std::move(args.allowed_path)),
79  put_channel() // TODO: Make channel configurable outside of executor
80 {
81  fuzzuf::utils::CheckCrashHandling();
82 
83  SetCArgvAndDecideInputMode();
84 
85  // NOTE: The following code should be implemented in
86  // Executor::OpenExecutorDependantFiles()
87  // But, currently LinuxForkServerExecutor is not major executor. So we
88  // cannot do that.
89  if (stdin_mode) {
90  auto path = fuzzuf::utils::StrPrintf(
91  "/dev/shm/fuzzuf-cc.forkserver.executor_id-%d.stdin", getpid());
92  input_fd =
93  fuzzuf::utils::OpenFile(path, O_RDWR | O_CREAT | O_CLOEXEC, 0600);
94  } else {
95  input_fd = fuzzuf::utils::OpenFile(args.path_to_write_input.string(),
96  O_RDWR | O_CREAT | O_CLOEXEC, 0600);
97  }
98 
99  // Allocate shared memory on initialization of Executor
100  // It is sufficient if each LinuxForkServerExecutor::Run() can refer the
101  // memory
102  SetupSharedMemories();
103  SetupEnvironmentVariablesForTarget();
104  args.AflCoverageSizeToEnvironmentVariables();
105  CreateJoinedEnvironmentVariables(std::move(args.environment_variables));
106 
107  // Configure PUT runtime settings
108  put_channel.SetupForkServer((char *const *)cargv.data(),
109  raw_environment_variables);
110  put_channel.SetPUTExecutionTimeout(args.exec_timelimit_ms * 1000);
111  if (stdin_mode) {
112  put_channel.ReadStdin();
113  }
114  if (args.record_stdout_and_err) {
115  put_channel.SaveStdoutStderr();
116  }

189 void LinuxForkServerExecutor::Run(const u8 *buf, u32 len, u32 timeout_ms) {
190  // locked until std::shared_ptr<u8> lock is used in other places
191  while (IsFeedbackLocked()) {
192  usleep(100);
193  }
194 
195  if (timeout_ms == 0) {
196  // if timeout_ms is 0, then we use exec_timelimit_ms;
197  timeout_ms = exec_timelimit_ms;
198  }
199  if (timeout_ms != last_timeout_ms) {
200  // Update timeout
201  put_channel.SetPUTExecutionTimeout(timeout_ms * 1000);
202  last_timeout_ms = timeout_ms;
203  }
204 
205  // Aliases
206  ResetSharedMemories();
207 
208  WriteTestInputToFile(buf, len);
209 
210  //#if 0
211  // TODO: Since the information priority is Trace level that is less important
212  // than Debug, it should be hidden when runlevel is Debug.
213  DEBUG("Run: ");
214  DEBUG("%s ", cargv[0]);
215  std::for_each(cargv.begin(), cargv.end(),
216  [](const char *v) { DEBUG("%s ", v); });
217  DEBUG("\n")
218  //#endif
219 
220  last_exit_reason = PUTExitReasonType::FAULT_NONE;
221  last_signal = 0;
222 
223  // FIXME: When persistent mode is implemented, this tmp must be set to the
224  // value that represent if the last execution failed for timeout.
225 
226  // Request creating PUT process to fork server
227  ExecutePUTAPIResponse response = this->put_channel.ExecutePUT();
228 
229  // NOTE: Avoids reading shared memory before PUT exit.
230  /* Any subsequent operations on trace_bits must not be moved by the
231  compiler below this point. Past this location, trace_bits[] behave
232  very normally and do not have to be treated as volatile. */
233  MEM_BARRIER();
234 
235  DEBUG("Response { error=%d, exit_status=%d, signal_number=%d }\n",
236  response.error, response.exit_code, response.signal_number);
237 
238  if (response.error) {
239  last_exit_reason = PUTExitReasonType::FAULT_ERROR;
240  return;
241  }
242 
243  /* Report outcome to caller. */
244  // FIXME: PUT実行がシグナルで止まったときにバグる。
245  // APIのレスポンスが返ったということはPUTが終了したという前提で書いた。
246  // TODO: 余裕があったら PUTExitReasonType
247  // に終了コードとシグナル番号を持たせたい
248  bool child_timed_out = response.signal_number == SIGKILL;
249  if (response.signal_number > 0) {
250  last_signal = response.signal_number;
251 
252  if (child_timed_out && last_signal == SIGKILL) {
253  DEBUG("Reached PUT execution timeout");
254  last_exit_reason = PUTExitReasonType::FAULT_TMOUT;
255  } else {
256  last_exit_reason = PUTExitReasonType::FAULT_CRASH;
257  }
258 
259  return;
260  }
261 
262  if (uses_asan && response.exit_code == MSAN_ERROR) {
263  last_exit_reason = PUTExitReasonType::FAULT_CRASH;
264  return;
265  }
266 
267  return;
268 }

355 // Since PUT that is instrumented using afl-clang-fast or fuzzuf-cc

357 // Although, this is actually what the child process running PUT should do,
358 // since there are no environment variables need update for each PUT execution
359 // for now, by using feature that environment variables are inherited to child
360 // process, it is enough to do just once ( let's move it if not ). As the
361 // additional advantage, it can avoid to waste Copy on Write of heap region due
362 // to StrPrintf.

364  // Pass the id of shared memory to PUT.

366  fuzzuf_bb_coverage.SetupEnvironmentVariable();
367  // NOTE: Call extra_feedback.SetupEnvironmentVariable(...) from
368  // XXXExecutorInterface
369 
370  /* This should improve performance a bit, since it stops the linker from
371  doing extra work post-fork(). */
372  if (!getenv("LD_BIND_LAZY")) setenv("LD_BIND_NOW", "1", 1);
373 
374  // Since MSAN, ASAN and UBSAN related configurations below are inherited from
375  // AFL and not used in fuzzuf, it is not required. But since those
376  // functionality is considered implementable without conflicts in fuzzuf in

422 void LinuxForkServerExecutor::CreateJoinedEnvironmentVariables(
423  std::vector<std::string> &&extra) {
424  environment_variables.clear();
425  raw_environment_variables.clear();
426  for (auto e = environ; *e; ++e) environment_variables.push_back(*e);
427  std::move(extra.begin(), extra.end(),
428  std::back_inserter(environment_variables));
429  environment_variables.shrink_to_fit();
430  raw_environment_variables.reserve(environment_variables.size());
431  std::transform(environment_variables.begin(), environment_variables.end(),
432  std::back_inserter(raw_environment_variables),
433  [](const auto &e) { return e.c_str(); });
434  raw_environment_variables.push_back(nullptr);
435  raw_environment_variables.shrink_to_fit();
436 }
