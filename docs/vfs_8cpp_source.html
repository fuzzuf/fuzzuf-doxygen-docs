7 VFS::VFS(std::vector<fs::path> &&allowed_path_) {
8  if (allowed_path_.empty()) {
9  allowed_path.reset(new std::vector<fs::path>{});
10  current_workdir = "/nowhere";
11  return;
12  }
13  std::vector<fs::path> duplicated_allowed_path;
14  duplicated_allowed_path.reserve(allowed_path_.size());
15  if (!allowed_path_[0].has_root_directory())
16  throw exceptions::invalid_argument(
17  "First allowed path must contain root directory", __FILE__, __LINE__);
18  {
19  auto can = allowed_path_[0].lexically_normal();
20  current_workdir = can;
21  }
22  std::transform(allowed_path_.begin(), allowed_path_.end(),
23  std::back_inserter(duplicated_allowed_path),
24  [this](const auto &v) {
25  auto abs = Absolute(v);
26  auto can = abs.lexically_normal();
27  return can;
28  });
29  std::sort(duplicated_allowed_path.begin(), duplicated_allowed_path.end());
30  allowed_path.reset(new std::vector<fs::path>{});
31  allowed_path->push_back(std::move(duplicated_allowed_path[0]));
32  current_workdir = (*allowed_path)[0];
33  fs::path prev = duplicated_allowed_path[0];
34  auto prev_len = std::distance(prev.begin(), prev.end());
35  /*
36  * Remove duplicated or included paths.
37  */
38  for (auto cur = std::next(duplicated_allowed_path.begin());
39  cur != duplicated_allowed_path.end(); ++cur) {
40  auto cur_len = std::distance(cur->begin(), cur->end());
41  bool should_retain;
42  /*
43  * Since the paths are sorted, the path shorter than previous one definitely
44  * means no inclusion nor duplication. So the path should be left.
45  */
46  if (prev_len > cur_len) {
47  should_retain = true;
48  }
49  /*
50  * If the path doesn't contain previous path as prefix, the path is not
51  * included nor duplicated. So the path should be left.
52  */
53  else if (!std::equal(prev.begin(), prev.end(), cur->begin(),
54  std::next(cur->begin(), prev_len))) {
55  should_retain = true;
56  }
57  /*
58  * Otherwise, the path is same as or included by previous one.
59  * So the path should be discarded.
60  */
61  else {
62  should_retain = false;
63  }
64 
65  if (should_retain) {
66  prev = *cur;
67  allowed_path->push_back(std::move(*cur));
68  prev_len = std::distance(prev.begin(), prev.end());
69  }
70  }
71  /*
72  * Although Windows has current directories for each drives, VFS provides *NIX
73  * style single current directory. As the result, accepting multiple drives
74  * causes confusing results on windows environment. Therefore, we forbid
75  * multiple drives from being included in the current implementation.
76  */
77  if (std::find_if(allowed_path->begin(), allowed_path->end(),
78  [rn = (*allowed_path)[0].root_name()](const auto &p) {
79  return p.root_name() != rn;
80  }) != allowed_path->end())
81  throw exceptions::invalid_argument("Inconsistent root name", __FILE__,
82  __LINE__);
83 }

85 std::optional<fs::path> VFS::IsAllowedPath(const fs::path &p) const {
86  const auto abs = Absolute(p);
87  auto requested = abs.lexically_normal();
88  const auto ge =
89  std::upper_bound(allowed_path->begin(), allowed_path->end(), requested);
90  if (ge == allowed_path->begin()) {
91  return std::nullopt;
92  }
93  const auto closest = std::prev(ge);
94  const auto closest_len = std::distance(closest->begin(), closest->end());
95  const auto requested_len = std::distance(requested.begin(), requested.end());
96  if (closest_len > requested_len) {
97  return std::nullopt;
98  }
99  if (!std::equal(closest->begin(), closest->end(), requested.begin(),
100  std::next(requested.begin(), closest_len))) {
101  return std::nullopt;
102  }
103  return requested;
104 }
