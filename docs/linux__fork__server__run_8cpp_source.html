33 // Check if NativeLinuxExecutor correctly allocates the shared memory of
34 // variable size
35 BOOST_AUTO_TEST_CASE(LinuxForkServerExecutorVariableShm) {
36  // We set each of afl_shm_size and bb_shm_size to 5 different values:
37  // 0, 1, PAGE_SIZE-1, PAGE_SIZE, and PAGE_SIZE+1.
38  // If afl_shm_size == 0, executor.afl_shmid should be
39  // NativeLinuxExecutor::INVALID_SHMID. Otherwise, afl_shmid should have a
40  // valid value. We check the size of this shared memory. Although shmget
41  // internally rounds up the specified size to a multiple of PAGE_SIZE when
42  // allocating shared memories, shmctl returns the original value of size.
43  // The same is true for bb_shm_size.
44 
45  std::string root_dir_template("/tmp/fuzzuf_test.XXXXXX");
46  auto *const raw_dirname = mkdtemp(root_dir_template.data());
47  BOOST_CHECK(raw_dirname != nullptr);
48 
49  auto root_dir = fs::path(raw_dirname);
50  BOOST_SCOPE_EXIT(&root_dir) { fs::remove_all(root_dir); }
51  BOOST_SCOPE_EXIT_END
52 
53  // Create input/output dirctory
54  auto input_dir = root_dir / "input";
55  auto output_dir = root_dir / "output";
56  BOOST_CHECK_EQUAL(fs::create_directory(input_dir), true);
57  BOOST_CHECK_EQUAL(fs::create_directory(output_dir), true);
58 
59  auto path_to_write_seed = output_dir / "cur_input";
60 
61  long val = sysconf(_SC_PAGESIZE);
62  BOOST_CHECK(val != -1); // Make sure sysconf succeeds
63  u32 PAGE_SIZE = (u32)val;
64 
65  // std::vector<u32> checked_sizes = {0, 1, PAGE_SIZE - 1, PAGE_SIZE,
66  // PAGE_SIZE + 1};
67  std::vector<u32> checked_sizes = {PAGE_SIZE};
68 
69  for (u32 afl : checked_sizes) {
70  for (u32 bb : checked_sizes) {
71  LinuxForkServerExecutor executor(
72  LinuxForkServerExecutorParameters()
73  .set_argv(
74  std::vector<std::string>{TEST_SOURCE_DIR "/put_binaries/cat"})
75  .set_exec_timelimit_ms(1000)
76  .set_exec_memlimit(10000)
77  .set_path_to_write_input(path_to_write_seed)
78  .set_afl_shm_size(afl)
79  .set_bb_shm_size(bb)
80  .set_record_stdout_and_err(true)
81  .move());
82 
83  std::string input("Hello, World!");
84  executor.Run(reinterpret_cast<const u8 *>(input.c_str()), input.size());
85 
86  // (1) Check if content of stdin is passed to stdout and stderr
87  {
88  std::string s = "stdout: " + input;
89  fuzzuf::executor::output_t answer(s.begin(), s.end());
90  auto stdout = executor.MoveStdOut();
91  std::cout << std::string(stdout.begin(), stdout.end()) << std::endl;
92  BOOST_CHECK_EQUAL_COLLECTIONS(stdout.begin(), stdout.end(),
93  answer.begin(), answer.end());
94  }
95  {
96  std::string s = "stderr: " + input;
97  fuzzuf::executor::output_t answer(s.begin(), s.end());
98  auto stderr = executor.MoveStdErr();
99  std::cout << std::string(stderr.begin(), stderr.end()) << std::endl;
100  BOOST_CHECK_EQUAL_COLLECTIONS(stderr.begin(), stderr.end(),
101  answer.begin(), answer.end());
102  }
103 
104  // (2) Check if coverage store is allocated as intended
105  if (afl == 0) {
106  BOOST_CHECK_EQUAL(executor.GetAFLShmID(),
107  ShmCovAttacher::INVALID_SHMID);
108  } else {
109  struct shmid_ds info;
110  int res = shmctl(executor.GetAFLShmID(), IPC_STAT, &info);
111  BOOST_CHECK_EQUAL(res, 0);
112  BOOST_CHECK_EQUAL(info.shm_segsz, afl);
113  }
114 
115  if (bb == 0) {
116  BOOST_CHECK_EQUAL(executor.GetBBShmID(), ShmCovAttacher::INVALID_SHMID);
117  } else {
118  struct shmid_ds info;
119  int res = shmctl(executor.GetBBShmID(), IPC_STAT, &info);
120  BOOST_CHECK_EQUAL(res, 0);
121  BOOST_CHECK_EQUAL(info.shm_segsz, bb);
122  }
123 
124  // (3) Check if executor has collected coverage info
125  // ビットマップが0でないことを確認する（ビットマップは未実装）
126  }
127  }
128 }
