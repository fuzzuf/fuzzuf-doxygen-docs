39  IJON_NUM_CASE // this represents the number of cases in IJON's havoc
40 };

42 static constexpr std::array<double, IJON_NUM_CASE> IJONGetCaseWeights(bool has_extras, bool has_aextras) {
43  using afl::util::AFLGetCaseWeights;
44 
45  // IJON's probability distribution of mutation operators is based on that of AFL.
46  std::array<double, NUM_CASE> base_weights = AFLGetCaseWeights(has_extras, has_aextras);
47 
48  // The number of cases in havoc in IJON is different than in AFL.
49  // So we need to copy it first.
50  std::array<double, IJON_NUM_CASE> ret{};
51  for (u32 i=0; i<NUM_CASE; i++) {
52  ret[i] = base_weights[i];
53  }
54 
55  // IJON replaces some of the cases in havoc with its custom cases.
56  // So we need to change the weights of these cases in the probability distributions accordingly.
57  // See the followings:
58  // https://github.com/RUB-SysSec/ijon/blob/4cb8ae04d/afl-fuzz.c#L6332
59  // https://github.com/RUB-SysSec/ijon/blob/4cb8ae04d/afl-fuzz.c#L6471
60 
61  ret[IJON_DELETE_BYTES] = ret[DELETE_BYTES];
62  ret[DELETE_BYTES] = 0.0;
63 
64  ret[IJON_INSERT_EXTRA] = ret[INSERT_EXTRA];
65  ret[INSERT_EXTRA] = 0.0;
66 
67  ret[IJON_INSERT_AEXTRA] = ret[INSERT_AEXTRA];
68  ret[INSERT_AEXTRA] = 0.0;
69 
70  return ret;
71 }

78 u32 IJONHavocCaseDistrib(
79  const std::vector<afl::dictionary::AFLDictData>& extras,
80  const std::vector<afl::dictionary::AFLDictData>& a_extras
81 ) {
82 
83  // FIXME: replace this engine with our pRNG later to avoid being flooded with pRNGs.
84  static std::random_device seed_gen;
85  static std::mt19937 engine(seed_gen());
86 
87  // Static part: the following part doesn't run after a fuzzing campaign starts.
88 
89  constexpr std::array<double, IJON_NUM_CASE> weight_set[2][2] = {
90  { IJONGetCaseWeights(false, false), IJONGetCaseWeights(false, true) },
91  { IJONGetCaseWeights(true, false), IJONGetCaseWeights(true, true) }
92  };
93 
94  // FIXME: actually, libstdc++'s discrete_distribution doesn't use Walker's alias method...
95  // We should implement it by ourselves...
96  static std::discrete_distribution<u32> dists[2][2] = {
97  { std::discrete_distribution<u32>(weight_set[0][0].begin(), weight_set[0][0].end()),
98  std::discrete_distribution<u32>(weight_set[0][1].begin(), weight_set[0][1].end()) },
99  { std::discrete_distribution<u32>(weight_set[1][0].begin(), weight_set[1][0].end()),
100  std::discrete_distribution<u32>(weight_set[1][1].begin(), weight_set[1][1].end()) },
101  };
102 
103  // Dynamic part: the following part runs during a fuzzing campaign
104 
105  bool has_extras = !extras.empty();
106  bool has_aextras = !a_extras.empty();
107  return dists[has_extras][has_aextras](engine);
108 }

110 void IJONCustomCases(
111  u32 case_idx,
112  u8*& outbuf,
113  u32& len,
114  const std::vector<afl::dictionary::AFLDictData>& extras,
115  const std::vector<afl::dictionary::AFLDictData>& a_extras
116 ) {
117  switch (case_idx) {
118  case IJON_DELETE_BYTES:
119  // IJON intentionally ignores DELETE_BYTES and does nothing here.
120  // Nevertheless, we don't remove this case because removing this case
121  // slightly increases the expected number of mutation operators applied at once in havoc.
122  break;
123 
124  case IJON_INSERT_EXTRA : [[fallthrough]];
125  case IJON_INSERT_AEXTRA: {
126  // IJON always inserts keywords at the end of inputs.
127  u32 insert_at = len;
128 
129  /* Insert an extra. Do the same dice-rolling stuff as for the
130  previous case. */
131 
132  bool use_auto = case_idx == IJON_INSERT_AEXTRA;
133 
134  // CaseDistrib must not select these cases when there is no dictionary.
135  // But this is difficult to be guaranteed, so we put asserts here.
136  if (use_auto) DEBUG_ASSERT(!a_extras.empty());
137  else DEBUG_ASSERT(!extras.empty());
138 
139  // FIXME: replace AFL's pRNGs with ours later.
140  using afl::util::UR;
141  using afl::dictionary::AFLDictData;
142  u32 idx = use_auto ? UR(a_extras.size(), -1) : UR(extras.size(), -1);
143  const AFLDictData &extra = use_auto ? a_extras[idx] : extras[idx];
144 
145  u32 extra_len = extra.data.size();
146  if (len + extra_len >= afl::option::GetMaxFile<option::IJONTag>()) break;
147 
148  u8* new_buf = new u8[len + extra_len];
149 
150  /* Head */
151  std::memcpy(new_buf, outbuf, insert_at);
152 
153  /* Inserted part */
154  std::memcpy(new_buf + insert_at, &extra.data[0], extra_len);
155 
156  /* Tail */
157  std::memcpy(new_buf + insert_at + extra_len, outbuf + insert_at,
158  len - insert_at);
159 
160  delete[] outbuf;
161  outbuf = new_buf;
162  len += extra_len;
163 
164  break;
165  }
166 
167  default:
168  break;
169  }
170 }
