52 bool Mutator::MinimizeTree(Tree& tree, std::unordered_set<size_t>& bits,
53  Context& ctx, size_t start_index, size_t end_index,
54  const FTester& tester) {
55  size_t i = start_index;
56 
57  /* For each nonterminal, we generate the smallest possible subtree */
58  while (i < tree.Size()) {
59  NodeID n(i);
60  NTermID nt = tree.GetRule(n, ctx).Nonterm();
61 
62  /* Make sure it's the smallest */
63  if (tree.SubTreeSize(n) > ctx.GetMinLenForNT(nt)) {
64  /* We sequentially replace each node's subtree with the smallest one */
65  _scratchpad.GenerateFromNT(nt, ctx.GetMinLenForNT(nt), ctx);
66  if (std::optional<Tree> t = Mutator::TestAndConvert(
67  tree, n, _scratchpad, NodeID(0), ctx, bits, tester)) {
68  /* If the new transitions are still triggered, replace the tree */
69  tree = std::move(t.value());
70  }
71  }
72 
73  if (++i == end_index) {
74  return false;
75  }
76  }
77 
78  return true;
79 };

96 bool Mutator::MinimizeRec(Tree& tree, std::unordered_set<size_t>& bits,
97  Context& ctx, size_t start_index, size_t end_index,
98  const FTester& tester) const {

103

114  return false;
115  }
116  }

135 bool Mutator::MutRules(Tree& tree, Context& ctx, size_t start_index,
136  size_t end_index, const FTesterMut& tester) {
137  for (size_t i = start_index; i < end_index; i++) {
138  if (i == tree.Size()) {
139  return true;
140  }
141 
142  /* Get all possible rules of each node */
143  NodeID n(i);
144  const RuleID& old_rule_id = tree.GetRuleID(n);
145  const std::vector<RuleID>& rule_ids =
146  ctx.GetRulesForNT(ctx.GetNT(RuleIDOrCustom(old_rule_id)));
147 
148  for (const RuleID& new_rule_id : rule_ids) {
149  /* Mutate by a new rule */
150  if (old_rule_id != new_rule_id) {
151  size_t random_size = ctx.GetRandomLenForRuleID(new_rule_id);
152  _scratchpad.GenerateFromRule(new_rule_id, random_size, ctx);
153 
154  TreeMutation repl =
155  tree.MutateReplaceFromTree(n, _scratchpad, NodeID(0));
156  tester(repl, ctx);
157  }
158  }
159  }
160 
161  return false;
162 }

177 void Mutator::MutSplice(Tree& tree, Context& ctx, const ChunkStore& cks,
178  const FTesterMut& tester) const {

230 void Mutator::MutRandomRecursion(Tree& tree,
231  const std::vector<RecursionInfo>& recursions,
232  Context& ctx, const FTesterMut& tester) const {
233  if (recursions.size() == 0) return;
234 
235  /* The degree of nesting */
236  size_t max_len_of_recursions = 2 << utils::random::Random<size_t>(1, 10);
237 
238  /* Get a random recursion from the tree */
239  const RecursionInfo& recursion_info = utils::random::Choose(recursions);
240  auto [rec0, rec1] = recursion_info.GetRandomRecursionPair();
241 
242  size_t recursion_len_pre =
243  static_cast<size_t>(rec1) - static_cast<size_t>(rec0);
244  size_t recursion_len_total = tree.SubTreeSize(rec0) - tree.SubTreeSize(rec1);
245  size_t recursion_len_post = recursion_len_total - recursion_len_pre;
246  size_t num_of_recursions = max_len_of_recursions / recursion_len_total;
247 
248  DEBUG_ASSERT((ssize_t)recursion_len_pre >= 0);
249  DEBUG_ASSERT((ssize_t)recursion_len_total >= 0);
250  DEBUG_ASSERT((ssize_t)recursion_len_post >= 0);
251 
252  /* Insert pre recursion */
253  size_t postfix = tree.SubTreeSize(rec1);
254 
255  /* Allocate enough buffer to avoid realloc */
256  std::vector<RuleIDOrCustom> rules_new;
257  rules_new.reserve(recursion_len_pre * num_of_recursions + postfix +
258  recursion_len_post * num_of_recursions);
259 
260  std::vector<size_t> sizes_new;
261  sizes_new.reserve(recursion_len_pre * num_of_recursions + postfix +
262  recursion_len_post * num_of_recursions);
263 
264  /* Repeat recursion */
265  for (size_t i = 0; i < num_of_recursions * recursion_len_pre; i++) {
266  rules_new.emplace_back(
267  tree.GetRuleOrCustom(rec0 + (i % recursion_len_pre)));
268  sizes_new.emplace_back(
269  tree.sizes().at(static_cast<size_t>(rec0) + (i % recursion_len_pre)));
270  }
271 
272  /* Append the ending of the original tree */
273  for (size_t i = 0; i < postfix; i++) {
274  rules_new.emplace_back(tree.GetRuleOrCustom(rec1 + i));
275  sizes_new.emplace_back(tree.sizes().at(static_cast<size_t>(rec1) + i));
276  }
277 
278  /* Adjust the sizes */
279  for (size_t i = 0; i < num_of_recursions * recursion_len_pre; i++) {
280  if (sizes_new[i] >= recursion_len_pre) {
281  sizes_new[i] +=
282  (num_of_recursions - i / recursion_len_pre - 1) * recursion_len_total;
283  }
284  }
285 
286  /* Append post recursion */
287  for (size_t i = 0; i < num_of_recursions * recursion_len_post; i++) {
288  rules_new.emplace_back(
289  tree.GetRuleOrCustom(rec1 + postfix + (i % recursion_len_post)));
290  sizes_new.emplace_back(tree.sizes().at(static_cast<size_t>(rec1) + postfix +
291  (i % recursion_len_post)));
292  }
293 
294  Tree recursion_tree(rules_new, sizes_new, {});
295  TreeMutation repl =
296  tree.MutateReplaceFromTree(rec1, recursion_tree, NodeID(0));
297 
298  tester(repl, ctx);
299 }

308 std::optional<NodeID> Mutator::FindParentWithNT(const Tree& tree,
309  const NodeID& node,
310  Context& ctx) {
311  const NTermID& nt = tree.GetRule(node, ctx).Nonterm();
312 
313  NodeID cur = node;
314  while (std::optional<NodeID> parent = tree.GetParent(cur)) {
315  if (tree.GetRule(parent.value(), ctx).Nonterm() == nt) {
316  return parent.value();
317  }
318 
319  cur = parent.value();
320  }
321 
322  return std::nullopt;
323 }
