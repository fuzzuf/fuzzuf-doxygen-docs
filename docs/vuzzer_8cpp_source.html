54 void VUzzer::PerformDryRun(VUzzerState& state) {
55  feedback::ExitStatusFeedback exit_status;
56  DEBUG("Starting dry run now...");
57 
58  if (state.pending_queue.size() < 3) {
59  ERROR("Not sufficient initial files");
60  }
61 
62  /* Execute PUT with initial seeds (valid inputs) */
63  for (const auto& testcase : state.pending_queue) {
64  testcase->input->Load();
65  auto inp_feed = state.RunExecutor(testcase->input->GetBuf(),
66  testcase->input->GetLen(), exit_status);
67  std::map<u64, u32> bb_cov;
68  boost::dynamic_bitset<> bb_set;
69  vuzzer::util::ParseBBCov(inp_feed, bb_cov);
70  for (const auto& bbc : bb_cov) state.good_bbs.insert(bbc.first);
71  testcase->input->Unload();
72  }
73 
74  DEBUG("Good BBs");
75  for (const auto& bb : state.good_bbs) DEBUG("0x%llx,", bb);
76 
77  /* Execute PUT with invalid inputs. */
78  /* Generate invalid random inputs by mutating initial seeds. */
79  DEBUG("Starting bad inputs...");
80  /* Original VUzzer execute create_files_dry(30) twice. */
81  for (int i = 0; i < 60; i++) {
82  const auto& testcase =
83  state.pending_queue[rand() % state.pending_queue.size()];
84  testcase->input->Load();
85  auto mutator = VUzzerMutator(*testcase->input, state);
86  mutator.TotallyRandom();
87 
88  auto inp_feed =
89  state.RunExecutor(mutator.GetBuf(), mutator.GetLen(), exit_status);
90  std::map<u64, u32> bb_cov;
91  vuzzer::util::ParseBBCov(inp_feed, bb_cov);
92  for (const auto& bbc : bb_cov) {
93  u64 addr = bbc.first;
94  if (state.good_bbs.find(addr) == state.good_bbs.end())
95  state.ehb.insert(addr);
96  }
97  testcase->input->Unload();
98  }
99 
100  DEBUG("EHBs");
101  for (const auto& bb : state.ehb) DEBUG("0x%llx,", bb);
102 
103  /* Get taint info from execution with initial seeds
104  ** It'll be used in FillSeeds
105  */
106  DEBUG("Get taint info from initial seeds");
107  for (const auto& testcase : state.pending_queue) {
108  testcase->input->Load();
109  auto inp_feed = state.RunTaintExecutor(
110  testcase->input->GetBuf(), testcase->input->GetLen(), exit_status);
111  vuzzer::util::ParseTaintInfo(state, testcase, inp_feed);
112  testcase->input->Unload();
113  }
114 }

121 void VUzzer::FillSeeds(VUzzerState& state, u32 size) {
122  std::random_device rd;
123  std::default_random_engine eng(rd());
124  std::uniform_real_distribution<> distr_cut(0.1, 1.0);
125 
126  u32 _sz = size;
127  DEBUG("FillSeeds with size (%u)", _sz);
128  /* Generate a certain number of seeds by mutating initial seeds */
129  std::vector<std::shared_ptr<VUzzerTestcase>> initial_queue =
130  state.pending_queue;
131  for (u32 sz = 0; sz < size;) {
132  std::vector<std::shared_ptr<VUzzerTestcase>> parents;
133  if (distr_cut(eng) >
134  (1.0 - state.setting->fill_seeds_with_crossover_prob) &&
135  size - sz > 1) {
136  DEBUG("Crossover");
137  std::sample(initial_queue.begin(), initial_queue.end(),
138  std::back_inserter(parents), 2,
139  std::mt19937{std::random_device{}()});
140 
141  DEBUG("Chose %s, %s", parents[0]->input->GetPath().c_str(),
142  parents[1]->input->GetPath().c_str());
143 
144  parents[0]->input->LoadByMmap();
145  parents[1]->input->LoadByMmap();
146 
147  auto crossover = VUzzerMutator(*parents[0]->input, state);
148  /* FIXME: Do not use ExecInputSet(input_set) */
149  auto seeds = crossover.CrossOver(*parents[1]->input);
150 
151  auto mutator1 = VUzzerMutator(*(seeds.first), state);
152  auto mutator2 = VUzzerMutator(*(seeds.second), state);
153 
154  mutator1.TaintBasedChange();
155  mutator2.TaintBasedChange();
156 
157  parents[0]->input->Unload();
158  parents[1]->input->Unload();
159 
160  std::string fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u",
161  state.setting->out_dir.c_str(),
162  state.queued_paths);
163  state.AddToQueue(state.pending_queue, fn, mutator1.GetBuf(),
164  mutator1.GetLen());
165 
166  fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u",
167  state.setting->out_dir.c_str(),
168  state.queued_paths);
169  state.AddToQueue(state.pending_queue, fn, mutator2.GetBuf(),
170  mutator2.GetLen());
171 
172  sz += 2;
173  } else {
174  std::sample(state.pending_queue.begin(), state.pending_queue.end(),
175  std::back_inserter(parents), 1,
176  std::mt19937{std::random_device{}()});
177 
178  parents[0]->input->LoadByMmap();
179 
180  auto mutator = VUzzerMutator(*parents[0]->input, state);
181 
182  mutator.MutateRandom();
183  mutator.TaintBasedChange();
184 
185  std::string fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u",
186  state.setting->out_dir.c_str(),
187  state.queued_paths);
188  state.AddToQueue(state.pending_queue, fn, mutator.GetBuf(),
189  mutator.GetLen());
190 
191  parents[0]->input->Unload();
192  sz++;
193  }
194  }
195 }

197 VUzzer::VUzzer(std::unique_ptr<VUzzerState>&& state_ref)
198  : state(std::move(state_ref)) {
199  /* Parse BB weight file */
200  vuzzer::util::ParseBBWeights(*state, state->setting->path_to_weight_file);
201 
202  /* Load dictionaries generated by static analysis tool (BB-weight.py) */
203  fuzzuf::algorithm::afl::dictionary::load(
204  state->setting->path_to_full_dict.native(), state->full_bytes_dict, false,
205  [](std::string&& m) { ERROR("%s", m.c_str()); });
206 
207  fuzzuf::algorithm::afl::dictionary::load(
208  state->setting->path_to_unique_dict.native(), state->unique_bytes_dict,
209  false, [](std::string&& m) { ERROR("%s", m.c_str()); });
210 
211  if (state->full_bytes_dict.size()) {
212  state->all_dicts.emplace_back(&(state->full_bytes_dict));
213  state->all_dicts.emplace_back(&(state->full_bytes_dict));
214  state->all_dicts.emplace_back(&(state->high_chars_dict));
215  state->all_dicts.emplace_back(&(state->unique_bytes_dict));
216  } else if (state->unique_bytes_dict.size()) {
217  state->all_dicts.emplace_back(&(state->unique_bytes_dict));
218  state->all_dicts.emplace_back(&(state->unique_bytes_dict));
219  state->all_dicts.emplace_back(&(state->high_chars_dict));
220  } else {
221  state->all_dicts.emplace_back(&(state->all_chars_dict));
222  }
223 
224  BuildFuzzFlow();
225  state->ReadTestcases();
226 
227  PerformDryRun(*state);
228 
229  u32 seed_size = static_cast<u32>(state->pending_queue.size());
230 
231  if (seed_size < state->setting->pop_size)
232  FillSeeds(*state, state->setting->pop_size - seed_size);
233 }

237 void VUzzer::BuildFuzzFlow() {
238  using namespace fuzzuf::algorithm::vuzzer::routine::other;
239  using namespace fuzzuf::algorithm::vuzzer::routine::mutation;
240  using namespace fuzzuf::algorithm::vuzzer::routine::update;
241 
242  using fuzzuf::hierarflow::CreateDummyParent;
243  using fuzzuf::hierarflow::CreateNode;
244 
245  // the head node
246  fuzz_loop = CreateNode<FuzzLoop>(*state);
247 
248  // middle nodes
249  auto decide_keep = CreateNode<DecideKeep>(*state);
250  auto run_ehb = CreateNode<RunEHB>(*state);
251  auto execute = CreateNode<ExecutePUT>(*state);
252  auto update_fitness = CreateNode<UpdateFitness>(*state);
253  auto trim_queue = CreateNode<TrimQueue>(*state);
254  auto execute_taint = CreateNode<ExecuteTaintPUT>(*state);
255  auto update_taint = CreateNode<UpdateTaint>(*state);
256  auto mutate = CreateNode<Mutate>(*state);
257  auto update_queue = CreateNode<UpdateQueue>(*state);
258 
259  fuzz_loop << decide_keep << run_ehb
260  << (execute << update_fitness << trim_queue ||
261  execute_taint << update_taint || mutate || update_queue);
262 }
