40 DIEMutCalleeRef DIEMutate::operator()(std::shared_ptr<DIETestcase> testcase) {
41  int fd;
42  u8 *buf_js, *buf_ty;
43  u32 len_js, len_ty;
44  std::vector<std::string> cmd;
45 
46  fs::path path_die = fs::absolute(state.setting->die_dir);
47  fs::path path_esfuzz = path_die / "fuzz/TS/esfuzz.js";
48  fs::path path_mutate = state.setting->out_dir / "mutated";
49 
50  /* Update status */
51  state.stage_short = "DIE";
52  state.stage_name = "DIE";
53 
54  /* Generate a seed used in esfuzz */
55  u32 seed = afl::util::UR(UINT32_MAX, state.rand_fd);
56 
57  /* Number of scripts to generate in this mutation */
58  int mut_cnt = state.setting->mut_cnt;
59 
60  state.stage_cur = 0;
61  state.stage_max = mut_cnt;
62 
63  /* Call esfuzz to mutate testcase */
64  cmd = {
65  "timeout",
66  "30",
67  state.setting->cmd_node,
68  path_esfuzz.string(), // Path to esfuzz.js
69  testcase->input->GetPath().string(), // Input JS
70  path_mutate.string(), // Output directory
71  afl::util::DescribeInteger(mut_cnt), // Number of mutation
72  afl::util::DescribeInteger(seed) // Seed
73  };
74  fuzzuf::utils::ExecuteCommand(cmd);
75 
76  /* Add generated files to queue */
77  for (int n = 0; n < mut_cnt; n++) {
78  state.stage_cur++;
79 
80  /* Create path string of output js and type files */
81  fs::path path_js =
82  fuzzuf::utils::StrPrintf("%s/%d.js", path_mutate.c_str(), n);
83  fs::path path_type = path_js.string() + ".t";
84 
85  if (!fs::exists(path_js) || !fs::exists(path_type)) {
86  /* esfuzz died for some reason */
87  continue;
88  }
89 
90  len_js = fs::file_size(path_js);
91  len_ty = fs::file_size(path_type);
92  if (len_js == 0 || len_ty == 0) {
93  /* esfuzz couldn't write file for some reason */
94  continue;
95  }
96 
97  /* Load JS file */
98  fd = fuzzuf::utils::OpenFile(path_js.string(), O_RDONLY);
99  buf_js = static_cast<u8*>(
100  mmap(nullptr, len_js, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0));
101  fuzzuf::utils::CloseFile(fd);
102 
103  if (buf_js == MAP_FAILED) {
104  ERROR("Unable to mmap '%s' : %s", path_js.c_str(), strerror(errno));
105  }
106 
107  /* Load type file */
108  fd = fuzzuf::utils::OpenFile(path_type.string(), O_RDONLY);
109  buf_ty = static_cast<u8*>(
110  mmap(nullptr, len_ty, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd, 0));
111  fuzzuf::utils::CloseFile(fd);
112 
113  if (buf_ty == MAP_FAILED) {
114  ERROR("Unable to mmap '%s' : %s", path_type.c_str(), strerror(errno));
115  }
116 
117  /* Execute PUT */
118  if (this->CallSuccessors(buf_js, len_js, buf_ty, len_ty)) {
119  munmap(buf_js, len_js);
120  munmap(buf_ty, len_ty);
121  return this->GoToParent();
122  }
123 
124  munmap(buf_js, len_js);
125  munmap(buf_ty, len_ty);
126  }
127 
128  return GoToDefaultNext();
129 }

143 DIEExecCalleeRef DIEExecute::operator()(const u8* buf_js,
144  u32 len_js, // js file
145  const u8* buf_ty,
146  u32 len_ty // type file
147 ) {
148  feedback::ExitStatusFeedback exit_status;
149 
150  /* Execution is the same as that of AFL */
151  auto inp_feed =
152  state.RunExecutorWithClassifyCounts(buf_js, len_js, exit_status);
153 
154  /* But we pass type file as well as JavaScript */
155  CallSuccessors(buf_js, len_js, buf_ty, len_ty, inp_feed, exit_status);
156 
157  return GoToDefaultNext();
158 }

174 DIEUpdateCalleeRef DIEUpdate::operator()(
175  const u8* buf_js, u32 len_js, // js file
176  const u8* buf_ty, u32 len_ty, // type file
177  feedback::InplaceMemoryFeedback& inp_feed,
178  feedback::ExitStatusFeedback& exit_status) {
179  if (state.stop_soon) {
180  return GoToParent();
181  }
182 
183  if (exit_status.exit_reason == feedback::PUTExitReasonType::FAULT_TMOUT) {
184  if (state.subseq_tmouts++ > afl::option::GetTmoutLimit(state)) {
185  state.cur_skipped_paths++;
186  return GoToParent();
187  }
188  } else
189  state.subseq_tmouts = 0;
190 
191  /* Don't forget to save type file as well as js file */
192  if (state.SaveIfInteresting(buf_js, len_js, buf_ty, len_ty, inp_feed,
193  exit_status)) {
194  state.queued_discovered++;
195  }
196 
197  if (state.stage_cur % state.stats_update_freq == 0 ||
198  state.stage_cur + 1 == state.stage_max) {
199  state.ShowStats();
200  }
201 
202  return GoToDefaultNext();
203 }
