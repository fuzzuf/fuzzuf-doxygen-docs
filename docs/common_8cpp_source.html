69 int ExecuteCommand(std::vector<std::string>& args) {
70  if (args.size() == 0)
71  return -1; // Too short arguments
72 
73  auto pid = fork();
74  if (pid < 0)
75  ERROR("fork() failed");
76 
77  if (pid == 0) {
78  /* Child process: Execute command */
79  // Prepare arguments
80  char **argv = new char *[args.size() + 1];
81 
82  for (size_t i = 0; i < args.size(); i++)
83  argv[i] = strdup(args[i].c_str());
84 
85  argv[args.size()] = NULL;
86 
87  // Execute command
88  if (execvp(args[0].c_str(), argv) == -1) {
89  std::string cmd = "";
90  for (auto arg: args)
91  cmd += arg + " ";
92  ERROR("execvp() failed: %s", cmd.c_str());
93  }
94 
95  _exit(-1); // Unreachable
96 
97  } else {
98  /* Parent process: Wait for child*/
99  int wstatus;
100 
101  if (waitpid(pid, &wstatus, 0) < 0)
102  ERROR("waitpid() failed");
103 
104  return WEXITSTATUS(wstatus);
105  }
106 }

142 /*
143  since read/write sometimes reads/writes less bytes than specified by the 3rd
144  argument 'n', it's safe to wrap read/write so that they should read/write
145  just 'n' bytes
146  */
147 ssize_t read_n(int fd, void *buf, size_t n, bool original_behaviour) {
148  size_t nread = 0;
149  while (nread < n) {
150  ssize_t res = read(fd, (char *)buf + nread, n - nread);
151  if (res == -1) {
152  // moreover, some of errors indicate we should retry
153  // see man page
154  if (errno == EAGAIN || errno == EINTR || errno == EWOULDBLOCK) {
155  continue;
156  }
157  if (original_behaviour)
158  return -1;
159  else
160  throw fuzzuf::utils::errno_to_system_error(errno);
161  }
162  if (res == 0) {
163  break;
164  }
165  nread += res;
166  }
167  return nread;
168 }

170 ssize_t write_n(int fd, const void *buf, size_t n) {
171  size_t nwritten = 0;
172  while (nwritten < n) {
173  ssize_t res = write(fd, (char *)buf + nwritten, n - nwritten);
174  if (res == -1) {
175  if (errno == EAGAIN || errno == EINTR || errno == EWOULDBLOCK) {
176  continue;
177  }
178  return -1;
179  }
180  nwritten += res;
181  }
182  return nwritten;
183 }

214 // 時間制限付きReadFile
215 // https://github.com/AFLplusplus/AFLplusplus/blob/stable/src/afl-forkserver.c#L135-L210
216 // timeout_ms ミリ秒後にタイムアウトする
217 // 戻り値は、エラー発生時に0、タイムアウト時にtimeout_ms+1
218 // それ以外の時はかかった時間(ミリ秒)
219 // fdがinvalid等のエラーに加えて、fdからlenバイト読めない場合もエラーになるので注意
220 // タイマーはselectシステムコールによって実現している
221 // select,
222 // readそれぞれがsignalのinterruptなどにより、失敗することを考慮に入れないといけない
223 // あるいは、readは指定したバイト数lenだけ一度に読み込むとは限らない
224 // したがって、ループによって複数回処理する必要がある
225 // 複数回実行される時の注意点として、残り時間の更新がある。
226 // Linuxにおいてはselectが引数timeoutの値を書き換え、「select中に経過した時間をtimeoutの値から差し引く処理」が実行される。すなわち自動で残り時間が更新されていく。
227 // それ以外の環境においては書き換えられないことが知られているため、時間を計測し、自前で残りの時間を計算する必要がある
228 // ただしこれらはドキュメントで保証されている挙動ではないように見えるため、どう対応するかは今後検討しなければならない

230 u32 ReadFileTimed(int fd, void *buf, u32 len, u32 timeout_ms) {
231  ssize_t nread = 0;
232  struct timeval timeout;
233 
234  timeout.tv_sec = (timeout_ms / 1000);
235  timeout.tv_usec = (timeout_ms % 1000) * 1000;
236 
237 #if !defined(__linux__)
238  u64 read_start = GetCurTimeMs();
239 #endif
240 
241  while (true) {
242  // loop内でselectを使うときは毎回初期化が必要(select man参照)
243  fd_set readfds;
244  FD_ZERO(&readfds);
245  FD_SET(fd, &readfds);
246 
247  // selectでtimeout時間だけ待つ
248  int sret = select(fd + 1, &readfds, NULL, NULL, &timeout);
249 
250  // 戻り値が正、0、負で場合分け
251  if (likely(sret >
252  0)) { // 戻り値が正の時、fdから読み出し準備ができているはず
253  // 読み出し処理。ただしreadがシグナル割り込みなどで失敗する可能性があるため、do
254  // whileで囲む
255  do {
256  ssize_t rret = read(fd, (char *)buf + nread, len - nread);
257 
258  // selectが消費した時間の計測と残り時間の更新
259 #if defined(__linux__)
260  u32 exec_ms =
261  timeout_ms - (timeout.tv_sec * 1000 + timeout.tv_usec / 1000);
262 #else
263  u32 exec_ms =
264  (u32)std::min((u64)timeout_ms, GetCurTimeMs() - read_start);
265  u32 remain_ms = timeout_ms - exec_ms;
266  timeout.tv_sec = (remain_ms / 1000);
267  timeout.tv_usec = (remain_ms % 1000) * 1000;
268 #endif
269 
270  if (likely(rret) > 0) { // fdからバイトが読めている、ただし
271  // lenバイト読めているかは分からない
272  nread += rret; // 今まで読んだバイト数を合算
273  if (likely(nread == len)) { // 一致していたらreturnしてよい
274  return std::max<u32>(
275  exec_ms, 1); // 0を返すとエラーになってしまうので1で切り上げる
276  }
277  break; // 一致していない場合はまだreadがいる。ただしselectに戻るのでbreak
278  } else if (
279  unlikely(
280  rret ==
281  0)) { // 0が返ってきたらファイルの末端に来ているということ。lenバイト読めないので失敗
282  return 0;
283  } else if (
284  errno != EINTR && errno != EAGAIN &&
285  errno !=
286  EWOULDBLOCK) { // この時点でrretは-1確定。先に完全なエラーのみ処理
287  return 0;
288  }
289 
290  // 割り込み等でreadをやり直したいが、すでに時間超過していたらタイムアウト
291  if (exec_ms >= timeout_ms) {
292  return timeout_ms + 1;
293  }
294 
295  // 何事もなければやり直し
296  // ここに来るのはつまりEINTR, EAGAIN, EWOULDBLOCKなどが発生し
297  // かつ時間超過していない場合
298  } while (1);
299  } else if (unlikely(
300  sret ==
301  0)) { // タイムアウトしてselectがreturnしたことを意味する
302  return timeout_ms + 1;
303  }
304 
305  // sret < 0を意味する。EINTRだけはシグナル割り込みによる失敗なので再度トライ
306  // FIXME: 非Linux環境でtimeoutを修正しないことによりbusy
307  // loopになるレアケースがあるか（ないと思うし時間計測頻繁にやると重いのであまり入れたくない）
308  // 現状timeoutの修正をreadのloop内でやっているため、外側だけをloopされるとbusy
309  // loopになる恐れはなくはない
310  if (likely(errno == EINTR))
311  continue;
312 
313  // その他何かしらの失敗なのでエラー
314  return 0;
315  }
316 }

366 int GetCpuCore() {
367  int cpu_core_count = 0;
368 #if defined(__APPLE__) || defined(__FreeBSD__) || defined(__OpenBSD__)
369  size_t s = sizeof(cpu_core_count);
370  /* On *BSD systems, we can just use a sysctl to get the number of CPUs. */
371 #ifdef __APPLE__
372  if (sysctlbyname("hw.logicalcpu", &cpu_core_count, &s, NULL, 0) < 0)
373  return;
374 #else
375  int s_name[2] = {CTL_HW, HW_NCPU};
376  if (sysctl(s_name, 2, &cpu_core_count, &s, NULL, 0) < 0)
377  return;
378 #endif /* ^__APPLE__ */
379 
380 #else
381 #ifdef HAVE_AFFINITY
382  cpu_core_count = sysconf(_SC_NPROCESSORS_ONLN);
383 #else
384  FILE *f = fopen("/proc/stat", "r");
385  u8 tmp[1024];
386 
387  if (!f)
388  return;
389 
390  while (fgets(tmp, sizeof(tmp), f))
391  if (!strncmp(tmp, "cpu", 3) && isdigit(tmp[3]))
392  cpu_core_count++;
393 
394  fclose(f);
395 
396 #endif /* ^HAVE_AFFINITY */
397 
398 #endif /* ^(__APPLE__ || __FreeBSD__ || __OpenBSD__) */
399  return cpu_core_count;
400 }

402 std::set<int> GetFreeCpu(int allcpu) {
403  std::set<int> used, frees;
404  struct dirent *de;
405  DIR *d = opendir("/proc");
406  if (!d) {
407  ERROR("Unable to access /proc - can't scan for free CPU cores.");
408  }
409 
410  // FIXME: maybe this should be a random value because this sleep is inserted
411  // in order to avoid lots of processes doing this procedure at the same time
412  usleep(500 * 250);
413 
414  /* Scan all /proc/<pid>/status entries, checking for Cpus_allowed_list.
415  Flag all processes bound to a specific CPU using cpu_used[]. This will
416  fail for some exotic binding setups, but is likely good enough in almost
417  all real-world use cases. */
418  while ((de = readdir(d))) {
419  std::string fn;
420  FILE *f;
421  char tmp[MAX_LINE];
422  bool has_vmsize = false;
423 
424  if (!isdigit(de->d_name[0]))
425  continue;
426 
427  fn = StrPrintf("/proc/%s/status", de->d_name);
428 
429  if (!(f = fopen(fn.c_str(), "r"))) {
430  continue;
431  }
432 
433  while (fgets(tmp, MAX_LINE, f)) {
434  int hval;
435  /* Processes without VmSize are probably kernel tasks. */
436  if (!strncmp(tmp, "VmSize:\t", 8))
437  has_vmsize = true;
438 
439  if (!strncmp(tmp, "Cpus_allowed_list:\t", 19) && !strchr(tmp, '-') &&
440  !strchr(tmp, ',') && sscanf(tmp + 19, "%d", &hval) == 1 &&
441  has_vmsize) {
442  used.insert(hval);
443  break;
444  }
445  }
446 
447  fclose(f);
448  }
449  closedir(d);
450 
451  for (int i = 0; i < allcpu; i++) {
452  if (used.find(i) == used.end())
453  frees.insert(i);
454  }
455  return frees;
456 }

474 int BindCpu(int cpu_core_count, int cpuid_to_bind) {
475 #if defined(__linux__)
476  if (cpuid_to_bind == CPUID_DO_NOT_BIND) {
477  return CPUID_DO_NOT_BIND;
478  }
479 
480  int binded_cpuid = CPUID_BIND_WHICHEVER;
481 
482  // If cpuid_to_bind is not CPUID_DO_NOT_BIND,
483  // then this function tries to bind this process to a cpu core somehow
484  std::set<int> vacant_cpus = fuzzuf::utils::GetFreeCpu(cpu_core_count);
485 
486  if (cpuid_to_bind == CPUID_BIND_WHICHEVER) {
487  // this means "don't care which cpu core, but should bind"
488 
489  if (vacant_cpus.empty()) {
490  ERROR("No more free CPU cores");
491  }
492 
493  binded_cpuid = *vacant_cpus.begin(); // just return a random number
494  } else {
495  if (cpuid_to_bind < 0 || cpu_core_count <= cpuid_to_bind) {
496  ERROR("The CPU core id to bind should be between 0 and %d", cpu_core_count - 1);
497  }
498 
499  if (vacant_cpus.count(cpuid_to_bind) == 0) {
500  ERROR("The CPU core #%d to bind is not free!", cpuid_to_bind);
501  }
502 
503  binded_cpuid = cpuid_to_bind;
504  }
505 
506  cpu_set_t c;
507  CPU_ZERO(&c);
508  CPU_SET(binded_cpuid, &c);
509 
510  if (sched_setaffinity(0, sizeof(c), &c)) {
511  ERROR("sched_setaffinity failed");
512  }
513 
514  return binded_cpuid;
515 #else /* defined(__linux__) */
516  if (cpuid_to_bind != CPUID_DO_NOT_BIND) {
517  DEBUG("In this environment, processes cannot be binded to a cpu core.");
518  }
519  return CPUID_DO_NOT_BIND;
520 #endif /* ^defined(__linux__) */
521 }

635 u32 CountBits(const u8 *mem, u32 len) {
636  assert(len % sizeof(u32) == 0);
637  const u32 *ptr = reinterpret_cast<const u32 *>(mem);
