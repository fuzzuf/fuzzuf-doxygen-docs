51 auto ForEachNonZeroByte(const Range &data, Integer first_feature, Callback cb)
52  -> std::enable_if_t<

59  constexpr std::size_t step = sizeof(std::uintptr_t) / sizeof(std::uint8_t);
60  constexpr std::size_t step_mask = step - 1u;
61  auto iter = data.begin();

63  // Iterate by 1 byte until either the alignment boundary or the end.
64  for (; reinterpret_cast<std::uintptr_t>(iter) & step_mask && iter < end;
65  ++iter) {
66  if (auto v = *iter)
67  cb(first_feature, Integer(std::distance(data.begin(), iter)), v);
68  }
69  // Iterate by Step bytes at a time.

71  if (std::uintptr_t bundle = *reinterpret_cast<const uintptr_t *>(iter)) {
72  bundle = utils::htole<std::uintptr_t>()(bundle);
73  for (std::size_t i = 0u; i < step; ++i, bundle >>= 8) {
74  if (uint8_t v = bundle & 0xFFu)
75  cb(first_feature, Integer(std::distance(data.begin(), iter) + i), v);
76  }
77  }

80  for (; iter < end; ++iter) {
