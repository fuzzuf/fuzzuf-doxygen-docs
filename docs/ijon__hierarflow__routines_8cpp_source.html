87 static void StoreMaxInput(IJONState &state, u32 idx, const u8 *data, u32 len) {
88  // NOTE: is it no problem to overwrite/unload `all_inputs[idx]`,
89  // even though this input can be still loaded in IJON's flow?
90  // The answer is yes because Mutator copies inputs to its own buffer,
91  // which means modifying these inputs doesn't affect Mutator.
92  // If Mutator's implementation is changed, then we have to
93  // add to ExecInput a member function that checks loaded or unloaded,
94  // and use OverwriteThenUnload and OverwriteKeepingLoaded accordingly.
95  state.all_inputs[idx]->OverwriteThenUnload(data, len);
96 
97  fs::path copy_fn =
98  state.max_dir / Util::StrPrintf("finding_%lu_%lu",
99  state.num_updates,
100  time(NULL));
101 
102  state.all_inputs[idx]->Copy(copy_fn);
103  state.num_updates++;
104 }
