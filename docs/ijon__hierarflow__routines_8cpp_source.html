112 static void StoreMaxInput(IJONState &state, u32 idx, const u8 *data, u32 len) {
113  // NOTE: is it no problem to overwrite/unload `all_inputs[idx]`,
114  // even though this input can be still loaded in IJON's flow?
115  // The answer is yes because Mutator copies inputs to its own buffer,
116  // which means modifying these inputs doesn't affect Mutator.
117  // If Mutator's implementation is changed, then we have to
118  // add to ExecInput a member function that checks loaded or unloaded,
119  // and use OverwriteThenUnload and OverwriteKeepingLoaded accordingly.
120  state.all_inputs[idx]->OverwriteThenUnload(data, len);
121 
122  fs::path copy_fn =
123  state.max_dir / fuzzuf::utils::StrPrintf("finding_%lu_%lu",
124  state.num_updates, time(NULL));
125 
126  state.all_inputs[idx]->Copy(copy_fn);
127  state.num_updates++;
128 }

134 IJONUpdCalleeRef IJONUpdate::operator()(
135  const u8 *buf, u32 len, feedback::InplaceMemoryFeedback &inp_feed,
136  feedback::ExitStatusFeedback &exit_status) {
137  // Originally, this procedure is done in save_if_interesting.
138  // And the update occurs only if the following condition is not met.
139  if (exit_status.exit_reason != state.crash_mode) return GoToDefaultNext();
140 
141  bool should_minify = len > 512;
142 
143  auto lambda = [this, &should_minify, buf, len](const u8 *trace_bits,
144  u32 /*unused*/) {
145  const std::uint64_t *afl_max =
146  reinterpret_cast<const std::uint64_t *>(std::next(trace_bits, offset));
147  for (u32 i = 0; i < option::GetMaxMapSize<option::IJONTag>(); i++) {
148  bool need_update = false;
149 
150  if (afl_max[i] > state.max_map[i]) {
151  need_update = true;
152 
153  if (state.max_map[i] == 0) {
154  // This i-th input will be nonempty shortly,
155  // so let's push it to nonempty_inputs.
156  state.nonempty_inputs.emplace_back(state.all_inputs[i]);
157  }
158 
159  state.max_map[i] = afl_max[i];
160  utils::StdoutLogger::Println(fuzzuf::utils::StrPrintf(
161  "updated maxmap %d: %lx (len: %ld)", i, state.max_map[i], len));
162  } else if (should_minify && afl_max[i] == state.max_map[i] &&
163  len < state.all_inputs[i]->GetLen()) {
164  need_update = true;
165  utils::StdoutLogger::Println(fuzzuf::utils::StrPrintf(
166  "minimized maxmap %d: %lx (len: %ld)", i, state.max_map[i], len));
167  }
168 
169  if (need_update) StoreMaxInput(state, i, buf, len);
170  }
171  };
172 
173  inp_feed.ShowMemoryToFunc(lambda);
174  return GoToDefaultNext();
175 }
