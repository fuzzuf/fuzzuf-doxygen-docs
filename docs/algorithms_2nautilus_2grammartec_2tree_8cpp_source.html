70 void Tree::CalcParents(Context& ctx) {
71  if (Size() == 0)
72  return;
73 
74  std::vector<std::pair<NTermID, NodeID>> stack;
75  stack.emplace_back(GetRule(NodeID(0), ctx).Nonterm(), NodeID(0));
76 
77  for (size_t i = 0; i < Size(); i++) {
78  NodeID node_id(i);
79  const NTermID& nonterm = GetRule(node_id, ctx).Nonterm();
80 
81  if (stack.size() == 0)
82  throw exceptions::fuzzuf_runtime_error(
83  "Not a valid tree for unparsing!", __FILE__, __LINE__
84  );
85 
86  auto [nterm_id, node] = stack.back();
87  stack.pop_back();
88  if (nterm_id != nonterm)
89  throw exceptions::fuzzuf_runtime_error(
90  "Not a valid tree for unparsing!", __FILE__, __LINE__
91  );
92 
93  _paren[i] = node;
94 
95  const Rule& rule = GetRule(node_id, ctx);
96  std::vector<NTermID> nonterms = rule.Nonterms();
97  for (auto it = nonterms.crbegin(); it != nonterms.crend(); ++it) {
98  stack.emplace_back(*it, node_id);
99  }
100  }
101 }

313 std::optional<std::vector<RecursionInfo>> Tree::CalcRecursions(Context& ctx) {
314  std::vector<RecursionInfo> ret;
315  std::unordered_set<NTermID> done_nterms;
316 
317  for (const RuleIDOrCustom& rule: _rules) {
318  const NTermID& nterm = ctx.GetNT(rule);

320  std::optional<RecursionInfo> ri = RecursionInfo::New(*this, nterm, ctx);

329  if (ret.empty())
330  return std::nullopt;
331 
332  return ret;

345 const RuleIDOrCustom& TreeMutation::GetAt(const NodeID& n) const {
346  size_t i = static_cast<size_t>(n);
347  size_t end0 = _prefix.size();
348  size_t end1 = end0 + _repl.size();
349  size_t end2 = end1 + _postfix.size();

363  "Index out of bound for rule access", __FILE__, __LINE__

508 bool Unparser::UnparseOneStep() {
509  if (_stack.size() == 0)
510  return false;
511 
512  auto& data = _stack.back().value();
513  _stack.pop_back();
514 
515  if (std::holds_alternative<Term>(data)) {
516  // Terminal symbol
517  Write(std::get<Term>(data));
518 
519  } else if (std::holds_alternative<NTerm>(data)) {
520  // Nonterminal symbol
521  Nonterm(std::get<NTerm>(data));
522 
523  } else {
524  throw exceptions::unreachable(
525  "Unexpected stack top", __FILE__, __LINE__
526  );
527  }
528 
529  return true;
530 }

586 void Unparser::NextPlain(const PlainRule& r) {
587  for (auto it = r.children.crbegin(); it != r.children.crend(); it++) {
588  const RuleChild& rule_child = *it;
589 
590  if (std::holds_alternative<Term>(rule_child.value())) {

595  // Push as tonterminal

598  } else {

601  );
602  }
