45 // Check if Mutator::Havoc crashes or causes any access violations.
46 // At the same time, this test makes sure that all the switch cases
47 // are implemented without omissions in Havoc.
48 BOOST_AUTO_TEST_CASE(MutatorHavoc) {
49  // We prepare HavocCase::NUM_CASE number of distributions.
50  // The i-th distribution always returns i.
51  // That is, chances of the integer i being returned are 100%.
52  // Using these distributions, we can check if each switch case
53  // is really implemented, and doesn't crash.
54 
55  // The seed can be anything, but it should be long to some extent
56  // and all its bytes should be different.
57  std::vector<u8> seed(100);
58  std::iota(seed.begin(), seed.end(), 1);
59 
60  // Also, extras and a_extras can be anything, but they should have at least
61  // one element.
62  using fuzzuf::algorithm::afl::dictionary::AFLDictData;
63  std::vector<AFLDictData> extras(1, AFLDictData({100, 101, 102, 103}));
64  std::vector<AFLDictData> a_extras(1, AFLDictData({'H', 'e', 'l', 'l', 'o'}));
65 
66  fuzzuf::exec_input::ExecInputSet
67  input_set; // to create OnMemoryInputSet, we need the set(factory)
68 
69  for (u32 i = 0; i < fuzzuf::mutator::NUM_CASE; i++) {
70  std::cout << "check the " << i << "-th mutation." << std::endl;
71 
72  // Create an instance of Mutator.
73  auto input = input_set.CreateOnMemory(&seed[0], seed.size());
74  auto mutator = fuzzuf::mutator::Mutator<TestTag>(*input);
75 
76  // Create the i-th distribution.
77  auto case_dist = ConstantMutopSelector(i);
78 
79  auto custom_cases = [](u32, u8*, u32, const std::vector<AFLDictData>&,
80  const std::vector<AFLDictData>&) {
81  BOOST_CHECK(false); // this should be never called
82  };
83 
84  mutator.Havoc(1024, extras, a_extras, case_dist, custom_cases);
85 
86  // Make sure that Havoc actually modified the input.
87  std::vector<u8> modified_seed(mutator.GetBuf(),
88  mutator.GetBuf() + mutator.GetLen());
89  BOOST_CHECK(seed != modified_seed);
90  }
91 
92  // Make sure that Havoc passes through the default case
93  // when a distribution return an integer other than 0, 1, ..., NUM_CASE-1
94  auto input = input_set.CreateOnMemory(&seed[0], seed.size());
95  auto mutator = fuzzuf::mutator::Mutator<TestTag>(*input);
96 
97  // Create a distribution that always returns NUM_CASE.
98  auto case_dist = ConstantMutopSelector(fuzzuf::mutator::NUM_CASE);
99 
100  bool passed_custom_cases = false;
101  auto custom_cases = [&passed_custom_cases](u32, u8*, u32,
102  const std::vector<AFLDictData>&,
103  const std::vector<AFLDictData>&) {
104  passed_custom_cases = true;
105  };
106  mutator.Havoc(1, extras, a_extras, case_dist, custom_cases);
107 
108  BOOST_CHECK(passed_custom_cases);
109 
110  // Because custom_cases does nothing this time, the input cannot be modified.
111  std::vector<u8> modified_seed(mutator.GetBuf(),
112  mutator.GetBuf() + mutator.GetLen());
113  BOOST_CHECK(seed == modified_seed);
114 }
