40  if (!raw_dirname)
41  throw -1;

56  // Create fuzzer instance
57  // FIXME: we can use BuildAFLFastFuzzerFromArgs after it supports forksrv as
58  // an option
59  using namespace fuzzuf::algorithm::afl;

67  std::shared_ptr<AFLFastSetting> setting(new AFLFastSetting(

69  input_dir.native(), output_dir.native(),
70  option::GetExecTimeout<AFLFastTag>(), option::GetMemLimit<AFLFastTag>(),

72  Util::CPUID_BIND_WHICHEVER, aflfastoption::FAST));
73 
74  // NativeLinuxExecutor needs the directory specified by "out_dir" to be

78 
79  // Create NativeLinuxExecutor

81  setting->argv, setting->exec_timelimit_ms, setting->exec_memlimit,

99 }
