122  template <class InputIterator>
123  WalkerDiscreteDistribution(const InputIterator s, const InputIterator e) {
124  size_t size = std::distance(s, e);
125  if (size == 0)
126  throw std::out_of_range("Array must not be empty");
127 
128  /* Check and cast weight */
129  _threshold.reserve(size);
130  for (auto it = s; it != e; ++it) {
131  double p = static_cast<double>(*it);
132  if (p < 0.0 || std::isnan(p))
133  throw std::range_error("Weight must not be negative or NaN");
134 
135  _threshold.push_back(p);
136  }
137 
138  /* Calculate sum of weights */
139  const double n = static_cast<double>(size);
140  double sum = std::accumulate(_threshold.begin(), _threshold.end(), 0.0);
141 
142  if (sum == std::numeric_limits<double>::infinity())
143  throw std::range_error("Sum of weights must not be inifinity");
144  else if (sum <= 0.0)
145  throw std::range_error("Sum of weights must be positive");
146 
147  /* Normalize weights so that average becomes 1 */
148  for (double& p: _threshold) {
149  p = (p / sum) * n;
150  }
151 
152  /* Prepare index */
153  _index.resize(size);
154  std::iota(_index.begin(), _index.end(), 0);
155 
156  /* Split weights into two groups */
157  std::vector<size_t> small, large;
158  size_t i = 0;
159  for (double p: _threshold) {
160  if (p < 1.0) {
161  small.push_back(i++);
162  } else {
163  large.push_back(i++);
164  }
165  }
166 
167  while (!small.empty() && !large.empty()) {
168  size_t j = small.back();
169  small.pop_back();
170  size_t k = large.back();
171 
172  /* j-th entry has j-th and k-th weights */
173  _index[j] = k;
174 
175  /* Fill remaining region by k-th weights */
176  _threshold[k] -= (1.0 - _threshold[j]);
177 
178  if (_threshold[k] < 1.0) {
179  /* If k-th weight gets smaller than average, put it into small group */
180  small.push_back(k);
181  large.pop_back();
182  }
183  }
184  }
