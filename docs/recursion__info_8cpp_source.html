81 std::optional<Parent> RecursionInfo::FindParents(Tree& t, const NTermID& nt,
82  Context& ctx) {
83  using Stack = std::pair<std::optional<NodeID>, size_t>;
84  std::vector<Stack> stack;
85  stack.emplace_back(std::nullopt, 0);
86 
87  std::optional<Parent> res = std::nullopt;
88 
89  for (size_t i = 0; i < t.rules().size(); i++) {
90  NodeID node(i);
91  auto [maybe_parent, depth] = stack.back();
92  stack.pop_back();
93 
94  if (ctx.GetNT(t.rules()[i]) == nt) {
95  if (maybe_parent) {
96  std::unordered_map<NodeID, NodeID> parents;
97  std::vector<NodeID> ids;
98  std::vector<size_t> weights;
99  if (res) std::tie(parents, ids, weights) = res.value();
100 
101  parents[node] = maybe_parent.value();
102  ids.emplace_back(node);
103  weights.emplace_back(depth);
104 
105  res = Parent(parents, ids, weights);
106  }
107 
108  maybe_parent = node;
109  }
110 
111  size_t n_children = ctx.GetNumChildren(t.rules()[i]);
112  for (size_t j = 0; j < n_children; j++) {
113  stack.emplace_back(maybe_parent, depth + 1);
114  }
115  }
116 
117  return res;
118 }
