111 auto PostProcess(
112  const boost::program_options::options_description &desc,
113  int argc,
114  const char *argv[],
115  const cli::GlobalFuzzerOptions &global,
116  std::function<void(std::string &&)> &&sink,
117  options::FuzzOption &dest
118 ) -> bool {
119  namespace po = boost::program_options;
120  po::variables_map vm;
121  po::store(
122  po::command_line_parser( argc, argv )
123  .options( desc )
124  .style(
125  po::command_line_style::default_style |
126  po::command_line_style::allow_long_disguise
127  )
128  .run(),
129  vm
130  );
131  po::notify(vm);
132 
133  if (global.logger != utils::Logger::Stdout && global.log_file) {
134  std::shared_ptr<std::fstream> fd(new std::fstream(
135  global.log_file->string(), std::ios::out | std::ios::binary));
136  dest.sink = [fd = std::move(fd)](std::string &&m) {
137  *fd << m << std::flush;
138  };
139  } else {
140  dest.sink = std::move( sink );
141  }
142 
143  if (global.help == 1) {
144  std::ostringstream out;
145  out << "Usage : " << std::endl;
146  out << " $ fuzzuf [eclipser] -- [options]\n";
147  out << desc << std::endl;
148  dest.sink(std::string(out.str()));
149  return false;
150  }
151 
152  if( !fs::exists( dest.target_prog ) ) {
153  std::ostringstream out;
154  out << "Target program " << dest.target_prog << " does not exist." << std::endl;
155  dest.sink(std::string(out.str()));
156  return false;
157  }
158 
159  fs::create_directories( dest.out_dir );
160  fs::create_directories( dest.sync_dir );
161 
162  if( vm.count( "architecture" ) ) {
163  if( vm[ "architecture" ].as< std::string >() == "x86" ) {
164  dest.architecture = Arch::X86;
165  }
166  else if( vm[ "architecture" ].as< std::string >() == "x64" ) {
167  dest.architecture = Arch::X64;
168  }
169  else {
170  std::ostringstream out;
171  out << "The architecture " << vm[ "architecture" ].as< std::string >() << " is not supported." << std::endl;
172  dest.sink(std::string(out.str()));
173  return false;
174  }
175  }
176  dest.fork_server = !vm[ "noforkserver" ].as< bool >();
177  if( vm.count( "filepath" ) ) {
178  dest.fuzz_source = FileInput{ vm[ "filepath" ].as< std::string >() };
179  }
180  else {
181  dest.fuzz_source = StdInput{};
182  }
183  dest.rng.seed(std::random_device()());
184 
185  options::SplitArgs( dest );
186 
187  return true;
188 }
