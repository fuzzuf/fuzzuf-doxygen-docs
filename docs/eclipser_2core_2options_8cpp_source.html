37  dest[ "architecture" ] = src.architecture;
38  dest[ "fork_server" ] = src.fork_server;
39  dest[ "input_dir" ] = src.input_dir;
40  dest[ "arg" ] = src.arg;
41  dest[ "fuzz_source" ] = src.fuzz_source;
42  dest[ "n_solve" ] = src.n_solve;

57  }
58  }
59  }

62  for( const auto &s: src.native_raw_args ) {

65  }

68 void SplitArgs( FuzzOption &opts ) {
69  const auto arg = executor::SplitCmdLineArg( opts.arg );
70  constexpr static std::array< Tracer, 3u > tracers{
71  Tracer::Coverage,
72  Tracer::Branch,
73  Tracer::BBCount
74  };
75  {
76  std::vector< std::string > cmd{ opts.target_prog };
77  cmd.reserve( arg.size() + 1u );
78  cmd.insert(
79  cmd.end(),
80  arg.begin(),
81  arg.end()
82  );
83  opts.native_splited_args = std::move( cmd );
84  std::vector< char* > raw;
85  raw.reserve( arg.size() + 1u );
86  std::transform(
87  opts.native_splited_args.begin(),
88  opts.native_splited_args.end(),
89  std::back_inserter( raw ),
90  []( const auto &v ) {
91  return const_cast< char* >( v.c_str() );
92  }
93  );
94  opts.native_raw_args = std::move( raw );
95  }
96  for( Tracer t: tracers ) {
97  opts.splited_args.insert(
98  std::make_pair(
99  t,
100  std::vector< std::string >{
101  executor::SelectTracer( t, opts.architecture ).string()
102  }
103  )
104  );
105  std::vector< char* > raw;
106  raw.reserve( arg.size() + 2u );
107  auto iter = opts.splited_args.find( t );
108  std::transform(
109  iter->second.begin(),
110  iter->second.end(),
111  std::back_inserter( raw ),
112  []( const auto &v ) {
113  return const_cast< char* >( v.c_str() );
114  }
115  );
116  std::transform(
117  opts.native_splited_args.begin(),
118  opts.native_splited_args.end(),
119  std::back_inserter( raw ),
120  []( const auto &v ) {
121  return const_cast< char* >( v.c_str() );
122  }
123  );
124  opts.raw_args.insert(
125  std::make_pair(
126  t,
127  std::move( raw )
128  )
129  );
130  }
131 }
