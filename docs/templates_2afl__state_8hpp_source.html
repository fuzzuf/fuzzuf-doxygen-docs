121 template<class Testcase>
122 PUTExitReasonType AFLStateTemplate<Testcase>::CalibrateCaseWithFeedDestroyed(
123  Testcase &testcase,
124  const u8 *buf,
125  u32 len,
126  InplaceMemoryFeedback &inp_feed,
127  ExitStatusFeedback &exit_status,
128  u32 handicap,
129  bool from_queue
130 ) {
131  std::array<u8, option::GetMapSize<Tag>()> first_trace;
132 
133  bool first_run = testcase.exec_cksum == 0;
134 
135  s32 old_sc = stage_cur;
136  s32 old_sm = stage_max;
137  std::string old_sn = std::move(stage_name);
138 
139  u32 use_tmout;
140  if (!from_queue || resuming_fuzz) {
141  use_tmout = std::max(setting->exec_timelimit_ms + option::GetCalTmoutAdd(*this),
142  setting->exec_timelimit_ms * option::GetCalTmoutPerc(*this) / 100);
143  } else {
144  use_tmout = setting->exec_timelimit_ms;
145  }
146 
147  testcase.cal_failed++;
148 
149  stage_name = "calibration";
150  stage_max = fast_cal ? 3 : option::GetCalCycles(*this);
151 
152  u8 hnb = 0;
153  u8 new_bits = 0;
154  if (testcase.exec_cksum) {
155  inp_feed.ShowMemoryToFunc(
156  [this, &first_trace, &hnb](const u8* trace_bits, u32 /* map_size */) {
157  std::memcpy(first_trace.data(), trace_bits, option::GetMapSize<Tag>());
158  hnb = HasNewBits(trace_bits, &virgin_bits[0], option::GetMapSize<Tag>());
159  }
160  );
161 
162  if (hnb > new_bits) new_bits = hnb;
163  }
164 
165  bool var_detected = false;
166  u64 start_us = fuzzuf::utils::GetCurTimeUs();
167  u64 stop_us;
168  for (stage_cur=0; stage_cur < stage_max; stage_cur++) {
169  std::cout << __FILE__ << " " << __LINE__ << " " << stage_cur << std::endl;
170 
171  if (!first_run && stage_cur % stats_update_freq == 0) {
172  ShowStats();
173  }
174 
175  InplaceMemoryFeedback::DiscardActive(std::move(inp_feed));
176  inp_feed =
177  RunExecutorWithClassifyCounts(buf, len, exit_status, use_tmout);
178 
179  /* stop_soon is set by the handler for Ctrl+C. When it's pressed,
180  we want to bail out quickly. */
181 
182  if (stop_soon || exit_status.exit_reason != crash_mode)
183  goto abort_calibration; // FIXME: goto
184 
185  std::cout << __FILE__ << " " << __LINE__ << " " << setting->dumb_mode << " " << stage_cur << " " << inp_feed.CountNonZeroBytes() << std::endl;
186  if (!setting->dumb_mode && !stage_cur && !inp_feed.CountNonZeroBytes()) {
187  std::cout << __FILE__ << " " << __LINE__ << std::endl;
188  exit_status.exit_reason = PUTExitReasonType::FAULT_NOINST;
189  goto abort_calibration; // FIXME: goto
190  }
191 
192  u32 cksum = inp_feed.CalcCksum32();
193 
194  if (testcase.exec_cksum != cksum) {
195  inp_feed.ShowMemoryToFunc(
196  [this, &hnb](const u8* trace_bits, u32 /* map_size */) {
197  hnb = HasNewBits(trace_bits, &virgin_bits[0], option::GetMapSize<Tag>());
198  }
199  );
200 
201  if (hnb > new_bits) new_bits = hnb;
202 
203  if (testcase.exec_cksum) {
204  inp_feed.ShowMemoryToFunc(
205  [this, &first_trace](const u8* trace_bits, u32 /* map_size */) {
206  for (u32 i=0; i < option::GetMapSize<Tag>(); i++) {
207  if (!var_bytes[i] && first_trace[i] != trace_bits[i]) {
208  var_bytes[i] = 1;
209  stage_max = option::GetCalCyclesLong(*this);
210  }
211  }
212  }
213  );
214 
215  var_detected = true;
216  } else {
217  testcase.exec_cksum = cksum;
218  inp_feed.ShowMemoryToFunc(
219  [&first_trace](const u8* trace_bits, u32 /* map_size */) {
220  std::memcpy(first_trace.data(), trace_bits, option::GetMapSize<Tag>());
221  }
222  );
223  }
224  }
225  }
226 
227  stop_us = fuzzuf::utils::GetCurTimeUs();
228 
229  total_cal_us += stop_us - start_us;
230  total_cal_cycles += stage_max;
231 
232  testcase.exec_us = (stop_us - start_us) / stage_max;
233  testcase.bitmap_size = inp_feed.CountNonZeroBytes();
234  testcase.handicap = handicap;
235  testcase.cal_failed = 0;
236 
237  total_bitmap_size += testcase.bitmap_size;
238  total_bitmap_entries++;
239 
240  UpdateBitmapScore(testcase, inp_feed);
241 
242  /* If this case didn't result in new output from the instrumentation, tell
243  parent. This is a non-critical problem, but something to warn the user
244  about. */
245 
246  if ( !setting->dumb_mode
247  && first_run
248  && exit_status.exit_reason == PUTExitReasonType::FAULT_NONE
249  && new_bits == 0) {
250  exit_status.exit_reason = PUTExitReasonType::FAULT_NOBITS;
251  }
252 
253 abort_calibration:
254 
255  if (new_bits == 2 && !testcase.has_new_cov) {
256  testcase.has_new_cov = true;
257  queued_with_cov++;
258  }
259 
260  /* Mark variable paths. */
261 
262  if (var_detected) {
263  var_byte_count = fuzzuf::utils::CountBytes(&var_bytes[0], var_bytes.size());
264  if (!testcase.var_behavior) {
265  MarkAsVariable(testcase);
266  queued_variable++;
267  }
268  }
269 
270  stage_name = old_sn;
271  stage_cur = old_sc;
272  stage_max = old_sm;
273 
274  if (!first_run) ShowStats();
275 
276  return exit_status.exit_reason;
277 }

311 template<class Testcase>
312 void AFLStateTemplate<Testcase>::UpdateBitmapScoreWithRawTrace(
313  Testcase &testcase,
314  const u8 *trace_bits,
315  u32 map_size
316 ) {
317  u64 fav_factor = testcase.exec_us * testcase.input->GetLen();
318 
319  for (u32 i=0; i<map_size; i++) {
320  if (trace_bits[i]) {
321  if (top_rated[i]) {
322  auto &top_testcase = top_rated[i].value().get();
323  u64 factor = top_testcase.exec_us * top_testcase.input->GetLen();
324  if (fav_factor > factor) continue;
325 
326  /* Looks like we're going to win. Decrease ref count for the
327  previous winner, discard its trace_bits[] if necessary. */
328  --top_testcase.tc_ref;
329  if (top_testcase.tc_ref == 0) {
330  top_testcase.trace_mini.reset();
331  }
332  }
333 
334  /* Insert ourselves as the new winner. */
335 
336  top_rated[i] = std::ref(testcase);
337  testcase.tc_ref++;
338 
339  if (!testcase.trace_mini) {
340  testcase.trace_mini.reset(
341  new std::bitset<option::GetMapSize<Tag>()>()
342  );
343 
344  auto& trace_mini = *testcase.trace_mini;
345  for (u32 j=0; j < option::GetMapSize<Tag>(); j++) {
346  trace_mini[j] = trace_bits[j] != 0;
347  }
348  }
349 
350  score_changed = true;
351  }
352  }

368 template<class Testcase>
369 bool AFLStateTemplate<Testcase>::SaveIfInteresting(
370  const u8 *buf,
371  u32 len,
372  InplaceMemoryFeedback &inp_feed,
373  ExitStatusFeedback &exit_status
374 ) {
375  bool keeping = false;
376 
377  std::string fn;
378  if (exit_status.exit_reason == crash_mode) {
379  /* Keep only if there are new bits in the map, add to queue for
380  future fuzzing, etc. */
381 
382  u8 hnb;
383 
384  inp_feed.ShowMemoryToFunc(
385  [this, &hnb](const u8* trace_bits, u32 /* map_size */) {
386  hnb = HasNewBits(trace_bits, &virgin_bits[0], option::GetMapSize<Tag>());
387  }
388  );
389 
390  if (!hnb) {
391  if (crash_mode == PUTExitReasonType::FAULT_CRASH) {
392  total_crashes++;
393  }
394  return false;
395  }
396 
397  if (!setting->simple_files) {
398  fn = fuzzuf::utils::StrPrintf("%s/queue/id:%06u,%s",
399  setting->out_dir.c_str(),
400  queued_paths,
401  routine::update::DescribeOp(*this, hnb).c_str()
402  );
403  } else {
404  fn = fuzzuf::utils::StrPrintf("%s/queue/id_%06u",
405  setting->out_dir.c_str(),
406  queued_paths
407  );
408  }
409 
410  auto testcase = AddToQueue(fn, buf, len, false);
411  if (hnb == 2) {
412  testcase->has_new_cov = 1;
413  queued_with_cov++;
414  }
415 
416  testcase->exec_cksum = inp_feed.CalcCksum32();
417 
418  // inp_feed will may be discard to start a new execution
419  // in that case inp_feed will receive the new feedback
420  PUTExitReasonType res = CalibrateCaseWithFeedDestroyed(
421  *testcase,
422  buf, len,
423  inp_feed,
424  exit_status,
425  queue_cycle - 1,
426  false);
427 
428  if (res == PUTExitReasonType::FAULT_ERROR) {
429  ERROR("Unable to execute target application");
430  }
431 
432  keeping = true;
433  }
434 
435  switch (exit_status.exit_reason) {
436  case PUTExitReasonType::FAULT_TMOUT:
437 
438  /* Timeouts are not very interesting, but we're still obliged to keep
439  a handful of samples. We use the presence of new bits in the
440  hang-specific bitmap as a signal of uniqueness. In "dumb" mode, we
441  just keep everything. */
442 
443  total_tmouts++;
444  if (unique_hangs >= option::GetKeepUniqueHang(*this)) {
445  // originally here "return keeping" is used, but this is clearer right?
446  return false;
447  }
448 
449  if (!setting->dumb_mode) {
450  if constexpr (sizeof(size_t) == 8) {
451  inp_feed.ModifyMemoryWithFunc(
452  [](u8* trace_bits, u32 /* map_size */) {
453  afl::util::SimplifyTrace<u64>((u64*)trace_bits, option::GetMapSize<Tag>());
454  }
455  );
456  } else {
457  inp_feed.ModifyMemoryWithFunc(
458  [](u8* trace_bits, u32 /* map_size */) {
459  afl::util::SimplifyTrace<u32>((u32*)trace_bits, option::GetMapSize<Tag>());
460  }
461  );
462  }
463 
464  u8 res;
465  inp_feed.ShowMemoryToFunc(
466  [this, &res](const u8* trace_bits, u32 /* map_size */) {
467  res = HasNewBits(trace_bits, &virgin_tmout[0], option::GetMapSize<Tag>());
468  }
469  );
470 
471  if (!res) {
472  // originally here "return keeping" is used, but this is clearer right?
473  return false;
474  }
475  }
476 
477  unique_tmouts++;
478 
479  /* Before saving, we make sure that it's a genuine hang by re-running
480  the target with a more generous timeout (unless the default timeout
481  is already generous). */
482 
483  if (setting->exec_timelimit_ms < hang_tmout) {
484  // discard inp_feed here because we will use executor
485  InplaceMemoryFeedback::DiscardActive(std::move(inp_feed));
486  inp_feed = RunExecutorWithClassifyCounts(
487  buf, len, exit_status, hang_tmout);
488 
489  /* A corner case that one user reported bumping into: increasing the
490  timeout actually uncovers a crash. Make sure we don't discard it if
491  so. */
492 
493  if (!stop_soon && exit_status.exit_reason == PUTExitReasonType::FAULT_CRASH) {
494  goto keep_as_crash; // FIXME: goto
495  }
496 
497  if ( stop_soon
498  || exit_status.exit_reason != PUTExitReasonType::FAULT_TMOUT) {
499  return false;
500  }
501  }
502 
503  if (!setting->simple_files) {
504  fn = fuzzuf::utils::StrPrintf("%s/hangs/id:%06llu,%s",
505  setting->out_dir.c_str(),
506  unique_hangs,
507  routine::update::DescribeOp(*this, 0).c_str()
508  );
509  } else {
510  fn = fuzzuf::utils::StrPrintf("%s/hangs/id_%06llu",
511  setting->out_dir.c_str(),
512  unique_hangs
513  );
514  }
515 
516  unique_hangs++;
517  last_hang_time = fuzzuf::utils::GetCurTimeMs();
518  break;
519 
520  case PUTExitReasonType::FAULT_CRASH:
521 keep_as_crash:
522 
523  /* This is handled in a manner roughly similar to timeouts,
524  except for slightly different limits and no need to re-run test
525  cases. */
526 
527  total_crashes++;
528 
529  if (unique_crashes >= option::GetKeepUniqueCrash(*this)) {
530  // unlike FAULT_TMOUT case, keeping can be true when "crash mode" is enabled
531  return keeping;
532  }
533 
534  if (!setting->dumb_mode) {
535  if constexpr (sizeof(size_t) == 8) {
536  inp_feed.ModifyMemoryWithFunc(
537  [](u8* trace_bits, u32 /* map_size */) {
538  afl::util::SimplifyTrace<u64>((u64*)trace_bits, option::GetMapSize<Tag>());
539  }
540  );
541  } else {
542  inp_feed.ModifyMemoryWithFunc(
543  [](u8* trace_bits, u32 /* map_size */) {
544  afl::util::SimplifyTrace<u32>((u32*)trace_bits, option::GetMapSize<Tag>());
545  }
546  );
547  }
548 
549  u8 res;
550  inp_feed.ShowMemoryToFunc(
551  [this, &res](const u8* trace_bits, u32 /* map_size */) {
552  res = HasNewBits(trace_bits, &virgin_crash[0], option::GetMapSize<Tag>());
553  }
554  );
555 
556  if (!res) {
557  // unlike FAULT_TMOUT case, keeping can be true when "crash mode" is enabled
558  return keeping;
559  }
560  }
561 
562 #if 0
563  if (!unique_crashes) WriteCrashReadme(); // FIXME?
564 #endif
565 
566  if (!setting->simple_files) {
567  fn = fuzzuf::utils::StrPrintf("%s/crashes/id:%06llu,sig:%02u,%s",
568  setting->out_dir.c_str(),
569  unique_crashes,
570  exit_status.signal,
571  routine::update::DescribeOp(*this, 0).c_str()
572  );
573  } else {
574  fn = fuzzuf::utils::StrPrintf("%s/hangs/id_%06llu_%02u",
575  setting->out_dir.c_str(),
576  unique_crashes,
577  exit_status.signal
578  );
579  }
580 
581  unique_crashes++;
582 
583  last_crash_time = fuzzuf::utils::GetCurTimeMs();
584  last_crash_execs = total_execs;
585 
586  break;
587 
588  case PUTExitReasonType::FAULT_ERROR:
589  ERROR("Unable to execute target application");
590 
591  default:
592  return keeping;
593  }
594 
595  /* If we're here, we apparently want to save the crash or hang
596  test case, too. */
597 
598  int fd = fuzzuf::utils::OpenFile(fn, O_WRONLY | O_CREAT | O_EXCL, 0600);
599  fuzzuf::utils::WriteFile(fd, buf, len);
600  fuzzuf::utils::CloseFile(fd);
601 
602  return keeping;
603 }

605 template<class Testcase>
606 u32 AFLStateTemplate<Testcase>::DoCalcScore(Testcase &testcase) {
607  u32 avg_exec_us = total_cal_us / total_cal_cycles;
608  u32 avg_bitmap_size = total_bitmap_size / total_bitmap_entries;
609  u32 perf_score = 100;
610 
611  /* Adjust score based on execution speed of this path, compared to the
612  global average. Multiplier ranges from 0.1x to 3x. Fast inputs are
613  less expensive to fuzz, so we're giving them more air time. */
614 
615  if (testcase.exec_us * 0.1 > avg_exec_us) perf_score = 10;
616  else if (testcase.exec_us * 0.25 > avg_exec_us) perf_score = 25;
617  else if (testcase.exec_us * 0.5 > avg_exec_us) perf_score = 50;
618  else if (testcase.exec_us * 0.75 > avg_exec_us) perf_score = 75;
619  else if (testcase.exec_us * 4 < avg_exec_us) perf_score = 300;
620  else if (testcase.exec_us * 3 < avg_exec_us) perf_score = 200;
621  else if (testcase.exec_us * 2 < avg_exec_us) perf_score = 150;
622 
623  /* Adjust score based on bitmap size. The working theory is that better
624  coverage translates to better targets. Multiplier from 0.25x to 3x. */
625 
626  if (testcase.bitmap_size * 0.3 > avg_bitmap_size) perf_score *= 3;
627  else if (testcase.bitmap_size * 0.5 > avg_bitmap_size) perf_score *= 2;
628  else if (testcase.bitmap_size * 0.75 > avg_bitmap_size) perf_score *= 1.5;
629  else if (testcase.bitmap_size * 3 < avg_bitmap_size) perf_score *= 0.25;
630  else if (testcase.bitmap_size * 2 < avg_bitmap_size) perf_score *= 0.5;
631  else if (testcase.bitmap_size * 1.5 < avg_bitmap_size) perf_score *= 0.75;
632 
633  /* Adjust score based on handicap. Handicap is proportional to how late
634  in the game we learned about this path. Latecomers are allowed to run
635  for a bit longer until they catch up with the rest. */
636 
637  if (testcase.handicap >= 4) {
638  perf_score *= 4;
639  testcase.handicap -= 4;
640  } else if (testcase.handicap) {
641  perf_score *= 2;

643  }

646  deeper test cases is more likely to reveal stuff that can't be
647  discovered with traditional fuzzers. */
648 
649  switch (testcase.depth) {
650  case 0 ... 3:
651  break;
652  case 4 ... 7:
653  perf_score *= 2;
654  break;
655  case 8 ... 13:
656  perf_score *= 3;
657  break;
658  case 14 ... 25:
659  perf_score *= 4;
660  break;
661  default:
662  perf_score *= 5;
663  break;
664  }
665 
666  /* Make sure that we don't go over limit. */
667 
668  if (perf_score > option::GetHavocMaxMult(*this) * 100) {
669  perf_score = option::GetHavocMaxMult(*this) * 100;
670  }
671 
672  return perf_score;
673 }

675 /* Check if the current execution path brings anything new to the table.
676  Update virgin bits to reflect the finds. Returns 1 if the only change is
677  the hit-count for a particular tuple; 2 if there are new tuples seen.
678  Updates the map, so subsequent calls will always return 0.
679  This function is called after every exec() on a fairly large buffer, so
680  it needs to be fast. We do this in 32-bit and 64-bit flavors. */
681 
682 template<class Testcase>
683 u8 AFLStateTemplate<Testcase>::HasNewBits(const u8 *trace_bits, u8 *virgin_map, u32 map_size) {
684  // we assume the word size is the same as sizeof(size_t)
685  static_assert( sizeof(size_t) == 4 || sizeof(size_t) == 8 );
686  using UInt = std::conditional_t<sizeof(size_t) == 4, u32, u64>;
687 
688  constexpr int width = sizeof(UInt);
689  constexpr int wlog = width == 4 ? 2 :
690  /* width == 8 */ 3 ;
691 
692  UInt* virgin = (UInt*)virgin_map;
693  const UInt* current = (const UInt*)trace_bits;
694 
695  u32 i = map_size >> wlog;
696 
697  u8 ret = 0;
698  while (i--) {
699  /* Optimize for (*current & *virgin) == 0 - i.e., no bits in current bitmap
700  that have not been already cleared from the virgin map - since this will
701  almost always be the case. */
702 
703  if (unlikely(*current) && unlikely(*current & *virgin)) {
704  if (likely(ret < 2)) {
705  u8* cur = (u8*)current;
706  u8* vir = (u8*)virgin;
707 
708  /* Looks like we have not found any new bytes yet; see if any non-zero
709  bytes in current[] are pristine in virgin[]. */
710 
711  for (int j=0; j < width; j++) {
712  if (cur[j] && vir[j] == 0xff) {
713  ret = 2;
714  break;
715  }
716  }
717  if (ret != 2) ret = 1;
718  }
719 
720  *virgin &= ~*current;
721  }
722 
723  current++;
724  virgin++;
725  }
726 
727  if (ret && virgin_map == &virgin_bits[0]) bitmap_changed = 1;
728 
729  return ret;
730 }
