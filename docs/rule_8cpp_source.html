89 std::string Rule::Unescape(const std::string& bytes) {
90  if (bytes.size() < 2) {
91  // Nothing to escape
92  return bytes;
93  }
94 
95  /* Convert "\{" into "{" and "\}" into "}" */
96  std::string res;
97  size_t i;
98  for (i = 0; i < bytes.size() - 1; i++) {
99  if (bytes[i] == '\\' && bytes[i+1] == '{') {
100  res += "{";
101  i++;
102  } else if (bytes[i] == '\\' && bytes[i+1] == '}') {
103  res += "}";
104  i++;
105  } else {
106  res += bytes[i];
107  }
108  }
109 
110  if (i < bytes.size()) {
111  res += bytes[bytes.size() - 1];
112  }
113 
114  return res;
115 }

126 std::vector<RuleChild> Rule::Tokenize(const std::string& format, Context& ctx) {
127  static std::regex TOKENIZER(R"((\{[^}\\]+\})|((?:[^{\\]|\\\{|\\\}|\\)+))");
128 
129  std::vector<RuleChild> r;
130  for (std::sregex_iterator it(format.begin(), format.end(), TOKENIZER), end;
131  it != end;
132  ++it) {
133  std::smatch m = *it;
134 
135  if (m[1].matched) {
136  // NT: "{A:a}"
137  r.emplace_back(m.str(), ctx);
138  } else if (m[2].matched) {
139  // "abc\{def\}ghi" --> "abc{def}ghi"
140  r.emplace_back(Rule::Unescape(m.str()));
141  } else {
142  throw exceptions::unreachable(
143  "Unexpected capturing group", __FILE__, __LINE__
144  );
145  }
146  }
147 
148  return r;
149 }

197 size_t Rule::Generate(Tree& tree, Context& ctx, size_t len) const {
198  /* Calculate required length */
199  size_t minimal_needed_len = 0;
200  for (NTermID nt: Nonterms())
201  minimal_needed_len += ctx.GetMinLenForNT(nt);
202 
203  DEBUG_ASSERT (minimal_needed_len <= len);
204 
205  size_t remaining_len = len - minimal_needed_len;
206 
207  size_t total_size = 1;
208  NodeID paren(tree.Size() - 1);
209 
210  /* Iterate over nonterminals */
211  std::vector<NTermID> nonterms = Nonterms();
212  for (size_t i = 0; i < nonterms.size(); i++) {
213  size_t cur_child_max_len;
214  std::vector<NTermID> new_nterms(nonterms.begin() + i, nonterms.end());
215 
216  if (new_nterms.size() != 0) {
217  cur_child_max_len = ctx.GetRandomLen(new_nterms.size(), remaining_len);
218  } else {
219  cur_child_max_len = remaining_len;
220  }
221  cur_child_max_len += ctx.GetMinLenForNT(nonterms[i]);
222 
223  RuleID rid = ctx.GetRandomRuleForNT(nonterms[i], cur_child_max_len);
224  // NOTE: RegExpRule should work differently here
225  RuleIDOrCustom rule_or_custom = RuleIDOrCustom(rid);
226 
227  DEBUG_ASSERT (tree.rules().size() == tree.sizes().size());
228  DEBUG_ASSERT (tree.paren().size() == tree.sizes().size());
229 
230  size_t offset = tree.Size();
231  tree.rules().emplace_back(rule_or_custom);
232  tree.sizes().emplace_back(0);
233  tree.paren().emplace_back(0);
234 
235  size_t consumed_len = ctx.GetRule(rid).Generate(
236  tree, ctx, cur_child_max_len - 1
237  );
238  tree.sizes()[offset] = consumed_len;
239  tree.paren()[offset] = paren;
240 
241  DEBUG_ASSERT (consumed_len <= cur_child_max_len);
242  DEBUG_ASSERT (consumed_len >= ctx.GetMinLenForNT(nonterms[i]));
243 
244  remaining_len += ctx.GetMinLenForNT(nonterms[i]);
245  remaining_len -= consumed_len;
246  total_size += consumed_len;
247  }
248 
249  return total_size;
250 }

307 std::string RuleChild::SplitNTDescription(const std::string& nonterm) const {
308  std::smatch m;
309  static std::regex SPLITTER(R"(^\{([A-Z][a-zA-Z_\-0-9]*)(?::([a-zA-Z_\-0-9]*))?\}$)");
310 
311  // Splits {A:a} or {A} into A and maybe a
312  if (!std::regex_match(nonterm, m, SPLITTER)) {
313  throw exceptions::fuzzuf_runtime_error(
314  Util::StrPrintf("Could not interpret Nonterminal %s. "
315  "Nonterminal Descriptions need to match "
316  "start with a capital letter and can only "
317  "contain [a-zA-Z_-0-9]", nonterm.c_str()),
318  __FILE__, __LINE__
319  );
320  }
321 
322  return m[1].str();
323 }

382 std::string ShowBytes(const std::string& bs) {
383  std::stringstream ss;
384  for (size_t i = 0; i < bs.size(); i++) {
385  char c = bs[i];
386  if (0x20 <= c && c <= 0x7e) {
387  ss << c;
388  } else {
389  switch (c) {
390  case '\t': ss << "\\t"; break;
391  case '\r': ss << "\\r"; break;
392  case '\n': ss << "\\n"; break;
393  case '\'': ss << "\\'"; break;
394  case '"': ss << "\\\""; break;
395  case '\\': ss << "\\\\"; break;
396  default:
397  ss << "\\x"
398  << std::setfill('0') << std::setw(2)
399  << std::hex << (uint8_t)c;
400  break;
401  }
402  }
403  }
404 
405  return ss.str();
406 }
