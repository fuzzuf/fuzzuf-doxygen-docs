80 RMutRules MutRules::operator()(QueueItem& inp) {

82  size_t end_index = start_index + 1;
83 
84  /* Deterministic tree mutation (Rules Mutation) */
85  if (state.DeterministicTreeMutation(inp, start_index, end_index)) {
86  if (cycle == state.setting->number_of_deterministic_mutations) {
87  inp.state = RandomState();

89  } else {
90  inp.state = DetState(cycle + 1, 0);

92 
93  } else {
94  inp.state = DetState(cycle, end_index);
95  }
96 
97  return GoToDefaultNext();
