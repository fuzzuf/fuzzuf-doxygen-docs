40 auto createMutator(const FuzzerCreateInfo &create_info) {
41  namespace hf = fuzzuf::hierarflow;
42 
43  dictionary::StaticDictionary manual_dictionary;
44  Load(create_info.dictionaries, manual_dictionary, false,
45  [](std::string &&m) { std::cerr << m << std::endl; });
46 
47  auto random = hf::CreateIrregularNode<standard_order::RandomCall<F, Ord>>();
48 
49  auto erase_bytes = hf::CreateNode<standard_order::EraseBytes<F, Ord>>();
50  auto insert_byte_ = hf::CreateNode<standard_order::InsertByte<F, Ord>>();
51  auto insert_repeated_bytes_ =
52  hf::CreateNode<standard_order::InsertRepeatedBytes<F, Ord>>();
53  auto change_byte_ = hf::CreateNode<standard_order::ChangeByte<F, Ord>>();
54  auto change_bit_ = hf::CreateNode<standard_order::ChangeBit<F, Ord>>();
55  auto shuffle_bytes_ = hf::CreateNode<standard_order::ShuffleBytes<F, Ord>>();
56  auto change_ascii_integer_ =
57  hf::CreateNode<standard_order::ChangeASCIIInteger<F, Ord>>();
58  auto change_binary_integer_ =
59  hf::CreateNode<standard_order::ChangeBinaryInteger<F, Ord>>();
60  auto copy_part_ = hf::CreateNode<standard_order::CopyPart<F, Ord>>();
61 
62  auto crossover_ = hf::CreateNode<standard_order::Crossover<F, Ord>>();
63  auto manual_dict = hf::CreateNode<
64  standard_order::StaticDict<F, dictionary::StaticDictionary, Ord>>(
65  std::move(manual_dictionary));
66  auto persistent_auto_dict =
67  hf::CreateNode<standard_order::DynamicDict<F, Ord>>();
68  auto to_ascii_ = create_info.only_ascii
69  ? hf::CreateNode<standard_order::ToASCII<F, Ord>>()
70  : hf::CreateNode<Proxy<F>>();
71  auto root = hf::CreateNode<Proxy<F>>();
72 
73  if (create_info.do_crossover) {
74  root << (random <= (erase_bytes || insert_byte_ ||
75  insert_repeated_bytes_ || change_byte_ ||
76  change_bit_ || shuffle_bytes_ ||
77  change_ascii_integer_ || change_binary_integer_ ||
78  copy_part_ || crossover_ || manual_dict ||
79  persistent_auto_dict) ||
80  to_ascii_);
81  } else {
82  root << (random <= (erase_bytes || insert_byte_ ||
83  insert_repeated_bytes_ || change_byte_ ||
84  change_bit_ || shuffle_bytes_ ||
85  change_ascii_integer_ || change_binary_integer_ ||
86  copy_part_ || manual_dict || persistent_auto_dict) ||
87  to_ascii_);
88  }
89 
90  return root;
91 }

115 template <typename F, typename Ord, typename Sink>
116 auto createExecuteAndFeedback(const fs::path &target_path,
117  const FuzzerCreateInfo &create_info,
118  bool force_add_to_corpus, bool may_delete_file,
119  bool persistent, bool strict_match,
120  const Sink &sink) {

122 
123  const auto output_file_path = create_info.output_dir / "result";
124  const auto path_to_write_seed = create_info.output_dir / "cur_input";
125 
126  auto create_coverage = hf::CreateNode<Clear<F, decltype(Ord::coverage)>>();
127 
128  std::unique_ptr<NativeLinuxExecutor> executor_(new NativeLinuxExecutor(
129  {target_path.string(), output_file_path.string()},
130  create_info.exec_timelimit_ms, create_info.exec_memlimit,
131  create_info.forksrv, path_to_write_seed, create_info.afl_shm_size,
132  create_info.bb_shm_size));

134  hf::CreateNode<standard_order::Execute<F, NativeLinuxExecutor, Ord>>(
135  std::move(executor_), create_info.use_afl_coverage);
136 
137  auto collect_features =
138  hf::CreateNode<standard_order::CollectFeatures<F, Ord>>();
139  auto add_to_corpus = hf::CreateNode<standard_order::AddToCorpus<F, Ord>>(
140  force_add_to_corpus, may_delete_file, persistent, strict_match,
141  create_info.output_dir, Sink(sink));
142 
143  create_coverage << execute_ << collect_features << add_to_corpus;
144 
145  return create_coverage;

165 template <typename F, typename Ord, typename Sink>
166 auto createInitialize(const fs::path &target_path,
167  const FuzzerCreateInfo &create_info,
168  ExecInputSet &initial_inputs, bool strict_match,
169  const Sink &sink) {
170  namespace hf = fuzzuf::hierarflow;
171 
172  auto for_each_initial_input = hf::CreateNode<ForEachStaticData<
173  F, ExecInputSetRange<true, ExecInputSetRangeInsertMode::NONE>,
174  decltype(Ord::input)>>(
175  initial_inputs |
176  adaptor::exec_input_set_range<true, ExecInputSetRangeInsertMode::NONE>);
177 
178  auto update_distribution = hf::CreateNode<
179  standard_order::UpdateDistribution<F, MakeVersion(12u, 0u, 0u), Ord>>(
180  create_info.sparse_energy_updates, create_info.max_mutation_factor, sink);
181 
182  auto new_cov = hf::CreateNode<standard_order::IfNewCoverage<F, Ord>>();
183 
184  auto add_to_solution_ = hf::CreateNode<standard_order::AddToSolution<F, Ord>>(
185  false, create_info.input_dir);
186 
187  auto nop3 = hf::CreateNode<Proxy<F>>();
188 
189  if (create_info.merge)
190  nop3 << (for_each_initial_input << createExecuteAndFeedback<F, Ord>(
191  target_path, create_info, true, true, false, strict_match,
192  sink) ||
193  (new_cov << add_to_solution_) || update_distribution);
194  else
195  nop3 << (for_each_initial_input << createExecuteAndFeedback<F, Ord>(
196  target_path, create_info, true, true, false, strict_match,
197  sink) ||
198  update_distribution);
199 
200  return nop3;
201 }

219 auto createRunone(const fs::path &target_path,

225  hf::CreateNode<StaticAppend<F, decltype(Ord::count)>>(1u);

234 
235  auto random = hf::CreateNode<standard_order::RandomCall<F, Ord>>();
236

238  hf::CreateNode<standard_order::ChooseRandomSeed<F, Ord>>(false);

240  ChooseRandomSeed<F, decltype(Ord::state && Ord::corpus && Ord::rng &&
241  Ord::crossover && Ord::exec_result)>>(
242  create_info.crossover_uniform_dist);
243 
244  auto update_dict = hf::CreateNode<standard_order::UpdateDictionary<F, Ord>>();
245 
246  auto increment_mutations_count_ =
247  hf::CreateNode<standard_order::IncrementMutationsCount<F, Ord>>();
248 
249  auto create_dict_entry =
250  hf::CreateNode<Clear<F, decltype(Ord::dict_history)>>();
251 
252  auto create_history =
253  hf::CreateNode<Clear<F, decltype(Ord::mutation_history)>>();
254 
255  auto print =
256  create_info.print_pcs

260  : hf::CreateNode<Proxy<F>>();
261 
262  auto nop1 = hf::CreateNode<Proxy<F>>();
263 
264  auto nop2 = hf::CreateNode<Proxy<F>>();
265 
266  auto nop3 = hf::CreateNode<Proxy<F>>();
267 
268  auto add_to_solution = hf::CreateNode<standard_order::AddToSolution<F, Ord>>(
269  create_info.crashed_only, create_info.output_dir);
270 
271  auto update_distribution = hf::CreateNode<
272  standard_order::UpdateDistribution<F, MakeVersion(12u, 0u, 0u), Ord>>(
273  create_info.sparse_energy_updates, create_info.max_mutation_factor, sink);
274 
275  auto nop4 = hf::CreateNode<Proxy<F>>();
276 
277  auto update_max_length =
278  create_info.len_control
279  ? hf::CreateNode<standard_order::UpdateMaxLength<F, Ord>>(
280  create_info.max_input_length, create_info.len_control)
281  : hf::CreateNode<Proxy<F>>();
282 
283  auto assign_last_corpus_update_run = hf::CreateNode<
284  DynamicAssign<F, decltype(Ord::count && Ord::last_corpus_update_run)>>();
285 
286  nop4 << (select_crossover || select_input ||
287  local_loop << (create_history || create_dict_entry ||
288  mutation_loop
289  << createMutator<F, Ord>(create_info) ||
290  nop2
291  << (increment_mutations_count_ ||
292  createExecuteAndFeedback<F, Ord>(
293  target_path, create_info, false, true,
294  false, false, sink) ||
295  new_cov << (update_dict || add_to_solution ||
296  print ||
297  assign_last_corpus_update_run)) ||
298  increment_counter) ||
299  update_distribution || update_max_length);
300  return nop4;
301 }

314 template <typename F, typename Ord, typename Sink>
315 auto create(const fs::path &target_path, const FuzzerCreateInfo &create_info,
316  ExecInputSet &initial_inputs, const Sink &sink) {
317  namespace hf = fuzzuf::hierarflow;
318 
319  auto global_loop =
320  hf::CreateNode<PartiallyDynamicRepeat<F, decltype(Ord::count)>>(
321  create_info.total_cycles);
322 
323  auto nop1 = hf::CreateNode<Proxy<F>>();
324 
325  auto dump_state = create_info.print_final_stats
326  ? hf::CreateNode<StaticDump<F, decltype(Ord::state)>>(
327  "state\n", 1, " ", sink)
328  : hf::CreateNode<Proxy<F>>();
329 
330  nop1 << (createInitialize<F, Ord>(target_path, create_info, initial_inputs,
331  false, sink) ||
332  global_loop << (createRunone<F, Ord>(target_path, create_info,
333  initial_inputs, sink)) ||
334  dump_state);
335 
336  return nop1;
337 }
