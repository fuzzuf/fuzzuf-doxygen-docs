40 static constexpr std::array<double, AFLPLUSPLUS_NUM_CASE>
41 AFLplusplusGetCaseWeights(bool has_extras, bool has_a_extras) {
42  std::array<double, AFLPLUSPLUS_NUM_CASE> weights{};
43 
44  weights[mutator::FLIP1] = 4.0; // case 0 ... 3
45  weights[mutator::INT8] = 4.0; // case 4 ... 7
46  weights[mutator::INT16_LE] = 2.0; // case 8 ... 9
47  weights[mutator::INT16_BE] = 2.0; // case 10 ... 11
48  weights[mutator::INT32_LE] = 2.0; // case 12 ... 13
49  weights[mutator::INT32_BE] = 2.0; // case 14 ... 15
50  weights[mutator::SUB8] = 4.0; // case 16 ... 19
51  weights[mutator::ADD8] = 4.0; // case 20 ... 23
52  weights[mutator::SUB16_LE] = 2.0; // case 24 ... 25
53  weights[mutator::SUB16_BE] = 2.0; // case 26 ... 27
54  weights[mutator::ADD16_LE] = 2.0; // case 28 ... 29
55  weights[mutator::ADD16_BE] = 2.0; // case 30 ... 31
56  weights[mutator::SUB32_LE] = 2.0; // case 32 ... 33
57  weights[mutator::SUB32_BE] = 2.0; // case 34 ... 35
58  weights[mutator::ADD32_LE] = 2.0; // case 36 ... 37
59  weights[mutator::ADD32_BE] = 2.0; // case 38 ... 39
60  weights[mutator::XOR] = 4.0; // case 40 ... 43
61  weights[mutator::CLONE_BYTES] = 3.0; // case 44 ... 46
62  weights[mutator::INSERT_SAME_BYTE] = 1.0; // case 47
63  weights[mutator::OVERWRITE_WITH_CHUNK] = 3.0; // case 48 ... 50
64  weights[mutator::OVERWRITE_WITH_SAME_BYTE] = 1.0; // case 51
65  weights[AFLPLUSPLUS_ADDBYTE] = 1.0; // case 52
66  weights[AFLPLUSPLUS_SUBBYTE] = 1.0; // case 53
67  weights[mutator::FLIP8] = 1.0; // case 54
68  weights[AFLPLUSPLUS_SWITCH_BYTES] = 2.0; // case 55 ... 56
69  weights[mutator::DELETE_BYTES] = 8.0; // case 57 ... 64
70 
71  if (has_extras && has_a_extras) {
72  weights[mutator::INSERT_EXTRA] = 1.0;
73  weights[mutator::OVERWRITE_WITH_EXTRA] = 1.0;
74  weights[mutator::INSERT_AEXTRA] = 1.0;
75  weights[mutator::OVERWRITE_WITH_AEXTRA] = 1.0;
76  } else if (has_extras) {
77  weights[mutator::INSERT_EXTRA] = 2.0;
78  weights[mutator::OVERWRITE_WITH_EXTRA] = 2.0;
79  } else if (has_a_extras) {
80  weights[mutator::INSERT_AEXTRA] = 2.0;
81  weights[mutator::OVERWRITE_WITH_AEXTRA] = 2.0;
82  }
83 
84  return weights;
85 }

126 void AFLplusplusCustomCases(
127  u32 case_idx, u8*& outbuf, u32& len,
128  [[maybe_unused]] const std::vector<afl::dictionary::AFLDictData>& extras,
129  [[maybe_unused]] const std::vector<afl::dictionary::AFLDictData>&
130  a_extras) {
131  auto UR = [](u32 limit) { return afl::util::UR(limit, -1); };
132  switch (case_idx) {
133  case AFLPLUSPLUS_ADDBYTE:
134  outbuf[UR(len)]++;
135  break;
136 
137  case AFLPLUSPLUS_SUBBYTE:
138  outbuf[UR(len)]--;
139  break;
140 
141  case AFLPLUSPLUS_SWITCH_BYTES: {
142  if (len < 4) {
143  break;
144  }
145 
146  u32 to_end, switch_to, switch_len, switch_from;
147  switch_from = UR(len);
148  do {
149  switch_to = UR(len);
150  } while (switch_from == switch_to);
151 
152  if (switch_from < switch_to) {
153  switch_len = switch_to - switch_from;
154  to_end = len - switch_to;
155  } else {
156  switch_len = switch_from - switch_to;
157  to_end = len - switch_from;
158  }
159 
160  switch_len = ChooseBlockLen(std::min(switch_len, to_end));
161 
162  std::unique_ptr<u8[]> new_buf(new u8[switch_len]);
163 
164  /* Backup */
165  memcpy(new_buf.get(), outbuf + switch_from, switch_len);
166 
167  /* Switch 1 */
168  memcpy(outbuf + switch_from, outbuf + switch_to, switch_len);
169 
170  /* Switch 2 */
171  memcpy(outbuf + switch_to, new_buf.get(), switch_len);
172 
173  break;
174  }
175 
176  default:
177  break;
178  }
179 }

181 // Temporarily copy-and-paste ChooseBlockLen() function, rather than modifying
182 // all function prototypes of CustomCases. Looking for a better way to achieve
183 // this...
184 u32 ChooseBlockLen(u32 limit) {
185  using Tag = aflplusplus::option::AFLplusplusTag;
186  u32 min_value, max_value;
187  u32 rlim = 3ULL;
188 
189  // just an alias of afl::util::UR
190  auto UR = [](u32 limit) { return afl::util::UR(limit, -1); };
191 
192  switch (UR(rlim)) {
193  case 0:
194  min_value = 1;
195  max_value = afl::option::GetHavocBlkSmall<Tag>();
196  break;
197 
198  case 1:
199  min_value = afl::option::GetHavocBlkSmall<Tag>();
200  max_value = afl::option::GetHavocBlkMedium<Tag>();
201  break;
202  default:
203  if (UR(10)) {
204  min_value = afl::option::GetHavocBlkMedium<Tag>();
205  max_value = afl::option::GetHavocBlkLarge<Tag>();
206  } else {
207  min_value = afl::option::GetHavocBlkLarge<Tag>();
208  max_value = afl::option::GetHavocBlkXl<Tag>();
209  }
210  }
211 
212  if (min_value >= limit) min_value = 1;
213 
214  return min_value + UR(std::min(max_value, limit) - min_value + 1);
215 }
