58  * Calculate offset for IJON Counter

60  u32 GetIjonCounterOffset() const {
61  return fuzzuf::utils::get_aligned_addr(afl_shm_size,

64  /*
65  * Calculate offset for IJON Max
66  */
67  u32 GetIjonMaxOffset() const {
68  return fuzzuf::utils::get_aligned_addr(
69  GetIjonCounterOffset() + ijon_counter_shm_size,
70  alignof(std::uint64_t));
71  }
72  /*
73  * Calculate total required shared memory size.
74  * Note that this is not AFL coverage size in IJON, but sum of all required buffer size.
75  * During such extra buffers are not in use, the value is equal to AFL coverage size.
76  */
77  u32 GetAflCoverageSize() const {
78  return fuzzuf::utils::get_aligned_addr(
79  GetIjonMaxOffset() + ijon_max_shm_size, alignof(std::uint64_t));
80  }
87  LinuxForkServerExecutorParameters &
88  AflCoverageSizeToEnvironmentVariables() {
89  std::string afl_coverage_size_str("__AFL_SIZE=");
90  afl_coverage_size_str += std::to_string(GetIjonCounterOffset());
91  environment_variables.emplace_back(std::move(afl_coverage_size_str));
92  if (ijon_counter_shm_size) {
93  std::string ijon_counter_size_str("__IJON_COUNTER_SIZE=");
94  ijon_counter_size_str +=
95  std::to_string(GetIjonMaxOffset() - GetIjonCounterOffset());
96  environment_variables.emplace_back(std::move(ijon_counter_size_str));
97  }
98  if (ijon_max_shm_size) {
99  std::string ijon_max_size_str("__IJON_MAX_SIZE=");
100  ijon_max_size_str +=
101  std::to_string(GetAflCoverageSize() - GetIjonMaxOffset());
102  environment_variables.emplace_back(std::move(ijon_max_size_str));
103  }
104  return *this;
105  }
