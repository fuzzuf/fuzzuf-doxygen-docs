44 // For C++20 concept capable compilers. This produces more precise error message

46 template <typename T>
47 concept AdaptableToReadOnce = requires(T &v, const fs::path &p) {
48  // Member function GetAllowedPath returns a range
49  { v.GetAllowedPath() }
50  ->std::ranges::range<>;
51  // Element of range retrived by GetAllowedPath is a path
52  { *v.GetAllowedPath().begin() }
53  ->std::convertible_to<fs::path>;
54  // Member function OpenDirectoryRecursive returns a range compatible iterator
55  { begin(v.OpenDirectoryRecursive(p)) }
56  ->std::input_iterator<>;
57  // value of iterator retrived by OpenDirectoryRecursive has member function
58  // path() and it returns fs::path value
59  { v.OpenDirectoryRecursive(p)->path() }
60  ->std::convertible_to<fs::path>;
61  // Member function IsRegularFile returns a bool
62  { v.IsRegularFile(p) }
63  ->std::convertible_to<bool>;
64  // Member function FileSize returns an integral value
65  { v.FileSize(p) }
66  ->std::integral<>;
67  // Member function Mmap returns a mapped_file_t
68  { v.Mmap(p, O_RDONLY, false) }
69  ->std::convertible_to<mapped_file_t>;
70  // Member function Remove is available
71  {v.Remove(p)};
72 };

132  std::vector<std::pair<fs::path, mapped_file_t>> MmapAll() {
133  std::vector<std::pair<fs::path, fuzzuf::utils::mapped_file_t>> files;
134  std::vector<fs::path> path_to_remove;
135  for (const auto &root_dir : GetAllowedPath()) {
136  for (const auto &de : base.OpenDirectoryRecursive(root_dir)) {
137  if (base.IsRegularFile(de.path()) && base.FileSize(de.path()) != 0u) {
138  auto filename = de.path().string();
139  auto mapped = base.Mmap(filename, O_RDONLY, false);
140  files.push_back(
141  std::make_pair(std::move(filename), std::move(mapped)));
142  path_to_remove.push_back(de.path());
143  }
144  }
145  }
146  for (const auto &p : path_to_remove) {
147  base.Remove(p);
148  }
149  files.shrink_to_fit();
150  return files;
151  }
155  void RemoveAll() {

162  }
