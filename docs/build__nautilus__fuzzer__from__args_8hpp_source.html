105 
106  /* Number of deterministic mutations */
107  ("detmut-num",
108  po::value<u64>(&number_of_deterministic_mutations)

115  po::value<u64>(&max_tree_size)
116  ->value_name("NUM")
117  ->default_value(GetDefaultMaxTreeSize()),
118  "Maximum size of tree (The larger this size is, the longer the input "
119  "will be.)")
120 
121  /* Fork server mode */
122  ("forksrv", po::value<bool>(&forksrv)->default_value(true),
123  "Enable/disable fork server mode. Default to true. (not recommended to "
124  "disable)")
125 
126  //
127  ("pargs", po::value<std::vector<std::string>>(&pargs),
128  "Specify PUT and args for PUT.");
129 
130  po::variables_map vm;
131  po::store(
132  po::command_line_parser(fuzzer_args.argc, fuzzer_args.argv)

139  fuzzuf::cli::fuzzer::nautilus::usage(

141  }

213  executor = std::make_shared<TExecutor>(std::move(qe));
214  break;
215  }
