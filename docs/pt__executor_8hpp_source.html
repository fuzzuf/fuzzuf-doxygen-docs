24 namespace fuzzuf::executor {
25 // A class for fuzz executions with Intel PT
26 // NOTE:
27 // This executor provides path coverage feedback introduced in PTrix
28 // (https://arxiv.org/pdf/1905.10499.pdf). The original PTrix implementation
29 // uses two bitmaps: trace_bits and pt_fav_bits. trace_bits records the last TIP
30 // IP hash value in the current context represented as one bit. This trace_bits
31 // semantics is quite different from the traditional AFL edge coverage, so we
32 // cannot use the trace_bits for calculating top_rated[] on AFL. The PTrix
33 // implementation introduces pt_fav_bits to calculate top_rated[]. From the
34 // above reasons, we should consider renaming trace_bits so that fuzzers can
35 // distinguish the shared memory semantics.
36 // https://github.com/junxzm1990/afl-pt/blob/master/afl-2.42b/pt-fuzz-fast.c#L1226-#L1229
37 class PTExecutor : public BaseProxyExecutor {
38  public:
39  // shm_size is fixed in PTrix pt-proxy-fast.
40  static constexpr u32 PATH_SHM_SIZE = (1U << 16);
41  static constexpr u32 FAV_SHM_SIZE = (1U << 16);
