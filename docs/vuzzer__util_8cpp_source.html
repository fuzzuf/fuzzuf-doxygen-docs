37 void ParseBBWeights(VUzzerState& state, const fs::path& path) {
38  /* Weight file format
39  * addr weight
40  * addr is address of BB
41  * weight is score of BB calculated by static analysis tool. (BB-weight.py)
42  */
43 
44  std::string off_raw, line;
45 
46  int off_fd = fuzzuf::utils::OpenFile(path.native(), O_RDONLY);
47 
48  struct stat sb {};
49 
50  fstat(off_fd, &sb);
51  off_raw.resize(sb.st_size);
52 
53  fuzzuf::utils::ReadFile(off_fd, (u8*)(off_raw.data()), sb.st_size);
54  fuzzuf::utils::CloseFile(off_fd);
55 
56  std::stringstream off_stream{off_raw};
57 
58  while (std::getline(off_stream, line)) {
59  std::stringstream line_stream(line);
60  std::string token;
61  std::vector<std::string> tokens;
62  while (std::getline(line_stream, token, ' ')) tokens.emplace_back(token);
63 
64  if (tokens.size() != 3)
65  throw -1; // XXX: Implment exception class for parse error;
66 
67  u32 addr = strtol(tokens[0].c_str(), NULL, 16);
68  u32 weight = strtol(tokens[1].c_str(), NULL, 16);
69  state.bb_weights[addr] = weight;
70  }
71 }

120 void ParseTaintInfo(VUzzerState& state,
121  const std::shared_ptr<VUzzerTestcase>& testcase,
122  feedback::FileFeedback& inp_feed) {
123  /* Taint file format
124  * CMP|LEA o1,o2,o3...o_n v1,v2,v3...v_m
125  * o_n is offsets touched by cmp/lea op.
126  * v_m is values referred by cmp/lea op.
127  */
128  u64 id = testcase->input->GetID();
129  std::string feed_raw, line;
130  fs::path feed_path = inp_feed.feed_path.native();
131 
132  int feed_fd = fuzzuf::utils::OpenFile(feed_path.native(), O_RDONLY);
133 
134  struct stat sb {};
135 
136  fstat(feed_fd, &sb);
137  feed_raw.resize(sb.st_size);
138 
139  fuzzuf::utils::ReadFile(feed_fd, (u8*)(feed_raw.data()), sb.st_size);
140  fuzzuf::utils::CloseFile(feed_fd);
141 
142  std::stringstream feed_stream{feed_raw};
143 
144  while (std::getline(feed_stream, line)) {
145  std::stringstream line_stream(line);
146  std::string token;
147  std::vector<std::string> tokens;
148  while (std::getline(line_stream, token, ' ')) tokens.emplace_back(token);
149 
150  if (tokens.size() != 3)
151  throw -1; // XXX: Implment exception class for parse error;
152 
153  std::stringstream off_stream(tokens[1]);
154  std::stringstream val_stream(tokens[2]);
155  std::string off_str, val_str;
156 
157  std::getline(off_stream, off_str, ','); // Take only offsets[0]
158  u32 offset = strtol(off_str.c_str(), NULL, 16);
159 
160  if (tokens[0] == "CMP") {
161  std::vector<u32> values;
162  while (std::getline(val_stream, val_str, ',')) {
163  u32 value = strtol(val_str.c_str(), NULL, 16);
164  values.emplace_back(value);
165  }
166  state.taint_cmp_all[id][offset] = values;
167  state.taint_cmp_offsets[id].insert(offset);
168 
169  for (auto v : values)
170  DEBUG("taint_cmp_all[%llu][0x%x] = 0x%x", id, offset, v);
171 
172  } else if (tokens[0] == "LEA") {
173  state.taint_lea_offsets[id].insert(offset);
174  } else {
175  throw -1;
176  }
177  }
178 }

216 /* Generate random n-bytes string from dictionaries */
217 std::unique_ptr<std::vector<u8>> GenerateRandomBytesFromDict(
218  u32 size, const std::vector<const dict_t*>& all_dicts) {
219  std::unique_ptr<std::vector<u8>> result(new std::vector<u8>);
220  while (result->size() < size) {
221  auto& dict = all_dicts[random() % all_dicts.size()];
222  const AFLDictData& word = dict->at(random() % dict->size());
223  result->insert(result->end(), word.data.begin(), word.data.end());
224  }
225  return result;
