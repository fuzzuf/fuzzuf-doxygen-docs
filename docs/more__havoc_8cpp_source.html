53 // Check if Mutator::Havoc crashes or causes any access violations.
54 // At the same time, this test makes sure that all the switch cases
55 // beyond mutator::NUM_CASE are implemented without omissions in Havoc.
56 BOOST_AUTO_TEST_CASE(MutatorMoreHavoc) {
57  // NOLINTBEGIN(cppcoreguidelines-pro-type-cstyle-cast,cppcoreguidelines-pro-type-member-init,cppcoreguidelines-special-member-functions,hicpp-explicit-conversions)

60  // We prepare (AFLPLUSPLUS_NUM_CASE - mutator::NUM_CASE) number of
61  // distributions. The i-th distribution always returns i. That is, chances of
62  // the integer i being returned are 100%. Using these distributions, we can
63  // check if each switch case is really implemented, and doesn't crash.
64 
65  // The seed can be anything, but it should be long to some extent
66  // and all its bytes should be different.
67  std::vector<u8> seed(100);
68  std::iota(seed.begin(), seed.end(), 1);
69 
70  // Also, extras and a_extras can be anything, but they should have at least
71  // one element.
72  using fuzzuf::algorithm::afl::dictionary::AFLDictData;
73  std::vector<AFLDictData> extras(1, AFLDictData({100, 101, 102, 103}));
74  std::vector<AFLDictData> a_extras(1, AFLDictData({'H', 'e', 'l', 'l', 'o'}));
75 
76  fuzzuf::exec_input::ExecInputSet
77  input_set; // to create OnMemoryInputSet, we need the set(factory)
78 
79  // The following objects are just dummy to create AFLplusplusState
80  using fuzzuf::algorithm::aflplusplus::AFLplusplusSetting;
81  using fuzzuf::executor::AFLExecutorInterface;
82  using fuzzuf::executor::NativeLinuxExecutor;
83  std::vector<std::string> argv{"non-existent-program", "someargument"};
84  auto setting = std::make_shared<AFLplusplusSetting>(
85  argv, input_dir.string(), output_dir.string(), 1000, 0, false, false,
86  fuzzuf::utils::CPUID_BIND_WHICHEVER,
87  fuzzuf::algorithm::aflfast::option::FAST, "test");
88  auto native_executor = std::make_shared<NativeLinuxExecutor>(
89  argv, 1000, 0, false, "non-existent-file", 0, 0);

91  std::make_shared<AFLExecutorInterface>(std::move(native_executor));
92 
93  std::cout
94  << "Going to check mutations provided by AFLplusplus custom cases..."
95  << std::endl;
96  namespace pph = fuzzuf::algorithm::aflplusplus::havoc;
97  for (u32 i = pph::AFLPLUSPLUS_ADDBYTE; i < pph::AFLPLUSPLUS_NUM_CASE; i++) {
98  std::cout << "check the " << i << "-th mutation." << std::endl;

100  // Create an instance of Mutator.
101  auto input = input_set.CreateOnMemory(&seed[0], seed.size());
102  auto mutator = fuzzuf::mutator::Mutator<TestTag>(*input);
103 
104  // Create the i-th distribution.
105  auto case_dist = ConstantMutopSelector(i);
106  std::unique_ptr<fuzzuf::optimizer::HavocOptimizer> havoc_optimizer(
107  new fuzzuf::optimizer::ConstantBatchHavocOptimizer(1024, case_dist));

129  auto mutator = fuzzuf::mutator::Mutator<TestTag>(*input);

135 
136  bool passed_custom_cases = false;
137  auto custom_cases = [&passed_custom_cases](u32 r, u8*&, u32&,
138  const std::vector<AFLDictData>&,
139  const std::vector<AFLDictData>&) {
140  // AFLplusplus's mutations always fall into custom cases.
141  // If r >= AFLPLUSPLUS_NUM_CASE, then it will jump to the default case as
142  // expected.
143  if (r >= pph::AFLPLUSPLUS_NUM_CASE) passed_custom_cases = true;
144  };
