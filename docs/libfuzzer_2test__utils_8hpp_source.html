129  fuzzuf::algorithm::libfuzzer::State state;
130  auto node = fuzzuf::hierarflow::CreateNode<L<Full, MutationPaths>>();
131  Variables v;
132  std::copy(data.begin(), data.end(), std::back_inserter(v.input[0]));

134  [](std::string &&m) { std::cout << m << std::flush; });
135  utils::ElapsedTimeTracer ett;
136  fuzzuf::hierarflow::WrapToMakeHeadNode(node)(v, tracer, ett);
137  data.clear();

139  ett.dump([](std::string &&m) { std::cout << m << std::flush; });
140 }
