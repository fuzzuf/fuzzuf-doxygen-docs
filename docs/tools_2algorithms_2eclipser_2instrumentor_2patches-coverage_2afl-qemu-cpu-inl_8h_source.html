59 /* Fork server logic, invoked once we hit _start. */
60 
61 static void afl_forkserver(CPUState *cpu) {
62 
63  static unsigned char tmp[4];
64 
65  if (atoi(getenv("ECL_FORK_SERVER")) != 1) return;
66 
67  /* Tell the parent that we're alive. If the parent doesn't want
68  to talk, assume that we're not running in forkserver mode. */
69 
70  if (write(FORKSRV_FD + 1, tmp, 4) != 4) return;
71 
72  afl_forksrv_pid = getpid();
73 
74  /* All right, let's await orders... */
75 
76  while (1) {
77 
78  pid_t child_pid;
79  int status, t_fd[2];
80 
81  /* Whoops, parent dead? */
82 
83  if (read(FORKSRV_FD, tmp, 4) != 4) exit(2);
84 
85  /* Establish a channel with child to grab translation commands. We'll
86  read from t_fd[0], child will write to TSL_FD. */
87 
88  if (pipe(t_fd) || dup2(t_fd[1], TSL_FD) < 0) exit(3);
89  close(t_fd[1]);
90 
91  child_pid = fork();
92  if (child_pid < 0) exit(4);
93 
94  if (!child_pid) {
95 
96  /* Child process. Close descriptors and run free. */
97 
98  afl_fork_child = 1;
99  close(FORKSRV_FD);
100  close(FORKSRV_FD + 1);
101  close(t_fd[0]);
102  return;
103 
104  }
105 
106  /* Parent. */
107 
108  close(TSL_FD);
109 
110  if (write(FORKSRV_FD + 1, &child_pid, 4) != 4) exit(5);
111 
112  /* Collect translation requests until child dies and closes the pipe. */
113 
114  afl_wait_tsl(cpu, t_fd[0]);
115 
116  /* Get and relay exit status to parent. */
117 
118  if (waitpid(child_pid, &status, 0) < 0) exit(6);
119  if (write(FORKSRV_FD + 1, &status, 4) != 4) exit(7);
120 
121  }
122 
123 }

125 /* This code is invoked whenever QEMU decides that it doesn't have a
126  translation of a particular block and needs to compute it. When this happens,
127  we tell the parent to mirror the operation, so that the next fork() has a
128  cached copy. */
129 
130 static void afl_request_tsl(target_ulong pc, target_ulong cb, uint64_t flags,
131  TranslationBlock* last_tb, int tb_exit) {
132  struct afl_tsl t;
133  struct afl_chain c;
134 
135  if (!afl_fork_child) return;
136 
137  t.pc = pc;
138  t.cs_base = cb;
139  t.flags = flags;
140  t.chained = (last_tb != NULL);
141 
142  if (write(TSL_FD, &t, sizeof(struct afl_tsl)) != sizeof(struct afl_tsl))
143  return;
144 
145  if (last_tb) {
146  c.last_pc = last_tb->pc;
147  c.last_cs_base = last_tb->cs_base;
148  c.last_flags = last_tb->flags;
149  c.tb_exit = tb_exit;
150 
151  if (write(TSL_FD, &c, sizeof(struct afl_chain)) != sizeof(struct afl_chain))
152  return;
153  }
154 }

156 /* This is the other side of the same channel. Since timeouts are handled by
157  afl-fuzz simply killing the child, we can just wait until the pipe breaks. */
158 
159 static void afl_wait_tsl(CPUState *cpu, int fd) {
160 
161  struct afl_tsl t;
162  struct afl_chain c;
163  TranslationBlock *tb, *last_tb;
164 
165  while (1) {
166 
167  /* Broken pipe means it's time to return to the fork server routine. */
168 
169  if (read(fd, &t, sizeof(struct afl_tsl)) != sizeof(struct afl_tsl))
170  break;
171 
172  tb = tb_htable_lookup(cpu, t.pc, t.cs_base, t.flags);
173 
174  if(!tb) {
175  mmap_lock();
176  tb_lock();
177  tb = tb_gen_code(cpu, t.pc, t.cs_base, t.flags, 0);
178  mmap_unlock();
179  tb_unlock();
180  }
181 
182  if (t.chained) {
183  if (read(fd, &c, sizeof(struct afl_chain)) != sizeof(struct afl_chain))
184  break;
185 
186  last_tb = tb_htable_lookup(cpu, c.last_pc, c.last_cs_base, c.last_flags);
187  if (last_tb) {
188  tb_lock();
189  if (!tb->invalid) {
190  tb_add_jump(last_tb, c.tb_exit, tb);
191  }
192  tb_unlock();
193  }
194  }
195  }
196 
197  close(fd);
198 
199 }
