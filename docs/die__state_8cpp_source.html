42 void DIEState::PivotInputs(void) {
43  ACTF("Creating hard links for all input files...");
44 
45  u32 id = 0;
46  for (const auto& testcase : case_queue) {
47  auto& input = *testcase->input;
48  auto& input_type = *testcase->input_type;
49 
50  std::string rsl = input.GetPath().filename().string();
51 
52  const std::string case_prefix = setting->simple_files ? "id_" : "id:";
53 
54  u32 orig_id;
55  std::string nfn;
56  if ( rsl.substr(0, 3) == case_prefix
57  && sscanf(rsl.c_str()+3, "%06u", &orig_id) == 1
58  && orig_id == id) {
59  resuming_fuzz = true;
60  /* We don't need to add ".js" here
61  because the original file name contains the extension */
62  nfn = Util::StrPrintf("%s/queue/%s",
63  setting->out_dir.c_str(), rsl.c_str());
64 
65  /* Since we're at it, let's also try to find parent and figure out the
66  appropriate depth for this entry. */
67 
68  u32 src_id;
69  auto pos = rsl.find(':');
70  if ( pos != std::string::npos
71  && sscanf(rsl.c_str()+pos+1, "%06u", &src_id) == 1) {
72 
73  if (src_id < case_queue.size()) {
74  testcase->depth = case_queue[src_id]->depth + 1;
75  }
76 
77  if (max_depth < testcase->depth) {
78  max_depth = testcase->depth;
79  }
80  }
81  } else {
82  /* No dice - invent a new name, capturing the original one as a
83  substring. */
84 
85  if (!setting->simple_files) {
86  auto pos = rsl.find(",orig:");
87 
88  if (pos != std::string::npos) pos += 6;
89  else pos = 0;
90 
91  nfn = Util::StrPrintf("%s/queue/id:%06u,orig:%s",
92  setting->out_dir.c_str(),
93  id,
94  rsl.c_str() + pos);
95  } else {
96  nfn = Util::StrPrintf("%s/queue/id_%06u",
97  setting->out_dir.c_str(), id);
98  }
99  }
100 
101  /* Pivot to the new queue entry. */
102 
103  if (!input.LinkAndRefer(nfn)) {
104  /* Save JS file */
105  input.CopyAndRefer(nfn);
106  }
107 
108  if (!input_type.LinkAndRefer(nfn + ".t")) {
109  /* Save type file */
110  input_type.CopyAndRefer(nfn + ".t");
111  }
112 
113  /* Make sure that the passed_det value carries over, too. */
114 
115  if (testcase->passed_det) MarkAsDetDone(*testcase);
116 
117  id++;
118  }
119 
120 #if 0
121  if (in_place_resume) NukeResumeDir();
122 #endif
123 }

136 std::shared_ptr<DIETestcase> DIEState::AddToQueue(
137  const std::string &fn_js, // path to js file
138  const u8 *buf_js, u32 len_js, // js data
139  const std::string &fn_ty, // path to type file
140  const u8 *buf_ty, u32 len_ty, // type data
141  bool passed_det
142 ) {
143  auto input_js = input_set.CreateOnDisk(fn_js);
144  auto input_ty = input_set.CreateOnDisk(fn_ty);
145  if (buf_js && buf_ty) {
146  input_js->OverwriteThenUnload(buf_js, len_js);
147  input_ty->OverwriteThenUnload(buf_ty, len_ty);
148  }
149 
150  std::shared_ptr<DIETestcase> testcase(
151  // A testcase contains JS and Type files
152  new DIETestcase(std::move(input_js), std::move(input_ty))
153  );
154 
155  testcase->depth = cur_depth + 1;
156  testcase->passed_det = passed_det;
157 
158  if (testcase->depth > max_depth) max_depth = testcase->depth;
159 
160  case_queue.emplace_back(testcase);
161 
162  queued_paths++;
163  pending_not_fuzzed++;
164  cycles_wo_finds = 0;
165  last_path_time = Util::GetCurTimeMs();
166 
167  return testcase;

174 void DIEState::ReadTestcases(void) {
175  int status;
176  std::vector<std::string> cmd;
177 
178  fs::path path_input = fs::absolute(setting->in_dir);
179  fs::path path_die = fs::absolute(setting->die_dir);
180  fs::path path_typer = path_die / "fuzz/TS/typer/typer.js";
181  fs::path path_esfuzz = path_die / "fuzz/TS/esfuzz.js";
182 
183  /* Create directory for JS testcase */
184  Util::CreateDir((setting->out_dir / "mutated").string());
185 
186  if (!fs::is_directory(path_input)) {
187  /* Exit if input path is not a directory */
188  ERROR("Invalid input path (not a directory): '%s'", path_input.c_str());
189  }
190 
191  if (!fs::exists(path_typer) || !fs::exists(path_esfuzz)) {
192  /* Exit if typer and fuzzer does not exist in DIE path */
193  ERROR("Invalid path to DIE: '%s'", path_die.c_str());
194  }
195 
196  if (!fs::exists(setting->typer_path)) {
197  /* Exit if typer script does not exist */
198  ERROR("Invalid path to 'typer.py': '%s'", setting->typer_path.c_str());
199  }
200 
201  MSG("Pre-processing input JavaScript files...\n"
202  "(This process may take a while if it's the first time.)\n");
203 
204  /* Scan input directory and call typer */
205  for (const auto& entry_js: fs::recursive_directory_iterator(path_input)) {
206  u32 type_size;
207 
208  /* Check extension */
209  fs::path path_js = entry_js.path();
210  if (path_js.extension() != ".js") {
211  if (path_js.extension() != ".jsi" && path_js.extension() != ".t")
212  ACTF("Testcase must have a \".js\" extension: %s\n", path_js.c_str());
213  continue;
214  }
215 
216  /* Check file size and type */
217  u32 js_size = fs::file_size(path_js);
218  fs::file_status st = fs::status(path_js);
219 
220  if (!fs::is_regular_file(st) || js_size == 0)
221  continue; // Skip empty files and non-regular files
222 
223  if (js_size > option::GetMaxFile())
224  EXIT("Test case '%s' is too big (%s, limit is %s)",
225  path_js.c_str(),
226  afl::util::DescribeMemorySize(js_size).c_str(),
227  afl::util::DescribeMemorySize(afl::option::GetMaxFile<Tag>()).c_str()
228  ); // File size is too big
229 
230  /* Check deterministic fuzzing */
231  fs::path dfn = path_input / ".state/deterministic_done" / path_js.filename();
232  bool passed_det = fs::exists(dfn);
233 
234  /* Create path to instrumented JS (jsi) */
235  fs::path path_jsi = path_js;
236  path_jsi.replace_extension(".jsi");
237 
238  /* We don't replace extension for type file
239  so that esfuzz can find this file */
240  fs::path path_type = path_js.string() + ".t";
241 
242  if (fs::exists(path_type)) {
243  /* Add to queue and skip if we already have type file */
244  type_size = fs::file_size(path_type);
245  AddToQueue(
246  path_js.string(), nullptr, js_size,
247  path_type.string(), nullptr, type_size,
248  passed_det
249  );
250 
251  ACTF("Type file exists. Skipping '%s'...", path_js.c_str());
252  continue;
253  }
254 
255  /* Instrument JS file */
256  ACTF("Instrumenting '%s'...", path_js.c_str());
257 
258  cmd = {setting->cmd_node, // node
259  path_typer.string(), // typer.js
260  path_js.string(), // js file
261  path_jsi.string()}; // instrumented js file
262  status = Util::ExecuteCommand(cmd);
263 
264  if (status != 0) {
265  /* Skip if instrumentation failed */
266  ACTF("Instrumentation failed :(");
267  continue;
268  }
269 
270  /* Collect type information */
271  ACTF("Profiling '%s'...", path_js.c_str());
272 
273  // Call typer.py
274  cmd.clear();
275  cmd = {setting->cmd_py, // python3
276  setting->typer_path, // typer.py
277  setting->d8_path, // path to d8 binary
278  setting->d8_flags, // flags for d8
279  path_jsi.string(), // instrumented js
280  path_type.string()}; // path of output type file
281  status = Util::ExecuteCommand(cmd);
282 
283  if (status != 0 || !fs::exists(path_type)) {
284  /* Skip if profiling failed */
285  ACTF("Profiling failed :(");
286  continue;
287  }
288 
289  /* Add testcase to queue */
290  type_size = fs::file_size(path_type);
291  AddToQueue(
292  path_js.string(), nullptr, js_size,
293  path_type.string(), nullptr, type_size,
294  passed_det
295  );
296  }
297 
298  if (!queued_paths) {
299  MSG("\n" cLRD "[-] " cRST
300  "Looks like there are no valid test cases in the input directory! The fuzzer\n"
301  " needs one or more test case to start with. The cases must be stored as\n"
302  " regular files directly in the input directory.\n");
303 
304  EXIT("No usable test cases in '%s'", path_input.c_str());
305  }
306 
307  /* Give some time to read the messages */
308  sleep(3);
309 
310  last_path_time = 0;
311  queued_at_start = queued_paths;
312 }

324 bool DIEState::SaveIfInteresting(
325  const u8 *buf_js, u32 len_js,
326  const u8 *buf_ty, u32 len_ty,
327  InplaceMemoryFeedback &inp_feed,
328  ExitStatusFeedback &exit_status
329 ) {
330  bool keeping = false;
331 
332  std::string fn;
333  if (exit_status.exit_reason == crash_mode) {
334  /* Keep only if there are new bits in the map, add to queue for
335  future fuzzing, etc. */
336 
337  u8 hnb;
338 
339  inp_feed.ShowMemoryToFunc(
340  [this, &hnb](const u8* trace_bits, u32 /* map_size */) {
341  hnb = HasNewBits(trace_bits, &virgin_bits[0], afl::option::GetMapSize<Tag>());
342  }
343  );
344 
345  if (!hnb) {
346  if (crash_mode == PUTExitReasonType::FAULT_CRASH) {
347  total_crashes++;
348  }
349  return false;
350  }
351 
352  if (!setting->simple_files) {
353  fn = Util::StrPrintf("%s/queue/id:%06u,%s.js",
354  setting->out_dir.c_str(),
355  queued_paths,
356  afl::routine::update::DescribeOp(*this, hnb).c_str()
357  );
358  } else {
359  fn = Util::StrPrintf("%s/queue/id_%06u.js",
360  setting->out_dir.c_str(),
361  queued_paths
362  );
363  }
364 
365  auto testcase = AddToQueue(
366  fn, buf_js, len_js,
367  fn + ".t", buf_ty, len_ty, // type file
368  false
369  );
370  if (hnb == 2) {
371  testcase->has_new_cov = 1;
372  queued_with_cov++;
373  }
374 
375  testcase->exec_cksum = inp_feed.CalcCksum32();
376 
377  /* FIXME: We should use refined coverage instead */
378  // inp_feed will maybe discard to start a new execution
379  // in that case inp_feed will receive the new feedback
380  PUTExitReasonType res = CalibrateCaseWithFeedDestroyed(
381  *testcase,
382  buf_js, len_js,
383  inp_feed,
384  exit_status,
385  queue_cycle - 1,
386  false);
387 
388  if (res == PUTExitReasonType::FAULT_ERROR) {
389  ERROR("Unable to execute target application");
390  }
391 
392  keeping = true;
393  }
394 
395  switch (exit_status.exit_reason) {
396  case PUTExitReasonType::FAULT_TMOUT:
397 
398  /* Timeouts are not very interesting, but we're still obliged to keep
399  a handful of samples. We use the presence of new bits in the
400  hang-specific bitmap as a signal of uniqueness. In "dumb" mode, we
401  just keep everything. */
402 
403  total_tmouts++;
404  if (unique_hangs >= afl::option::GetKeepUniqueHang(*this)) {
405  // originally here "return keeping" is used, but this is clearer right?
406  return false;
407  }
408 
409  if (!setting->dumb_mode) {
410  if constexpr (sizeof(size_t) == 8) {
411  inp_feed.ModifyMemoryWithFunc(
412  [](u8* trace_bits, u32 /* map_size */) {
413  afl::util::SimplifyTrace<u64>((u64*)trace_bits, afl::option::GetMapSize<Tag>());
414  }
415  );
416  } else {
417  inp_feed.ModifyMemoryWithFunc(
418  [](u8* trace_bits, u32 /* map_size */) {
419  afl::util::SimplifyTrace<u32>((u32*)trace_bits, afl::option::GetMapSize<Tag>());
420  }
421  );
422  }
423 
424  u8 res;
425  inp_feed.ShowMemoryToFunc(
426  [this, &res](const u8* trace_bits, u32 /* map_size */) {
427  res = HasNewBits(trace_bits, &virgin_tmout[0], afl::option::GetMapSize<Tag>());
428  }
429  );
430 
431  if (!res) {
432  // originally here "return keeping" is used, but this is clearer right?
433  return false;
434  }
435  }
436 
437  unique_tmouts++;
438 
439  /* Before saving, we make sure that it's a genuine hang by re-running
440  the target with a more generous timeout (unless the default timeout
441  is already generous). */
442 
443  if (setting->exec_timelimit_ms < hang_tmout) {
444  // discard inp_feed here because we will use executor
445  InplaceMemoryFeedback::DiscardActive(std::move(inp_feed));
446  inp_feed = RunExecutorWithClassifyCounts(
447  buf_js, len_js, exit_status, hang_tmout);
448 
449  /* A corner case that one user reported bumping into: increasing the
450  timeout actually uncovers a crash. Make sure we don't discard it if
451  so. */
452 
453  if (!stop_soon && exit_status.exit_reason == PUTExitReasonType::FAULT_CRASH) {
454  goto keep_as_crash; // FIXME: goto
455  }
456 
457  if ( stop_soon
458  || exit_status.exit_reason != PUTExitReasonType::FAULT_TMOUT) {
459  return false;
460  }
461  }
462 
463  if (!setting->simple_files) {
464  fn = Util::StrPrintf("%s/hangs/id:%06llu,%s.js",
465  setting->out_dir.c_str(),
466  unique_hangs,
467  afl::routine::update::DescribeOp(*this, 0).c_str()
468  );
469  } else {
470  fn = Util::StrPrintf("%s/hangs/id_%06llu.js",
471  setting->out_dir.c_str(),
472  unique_hangs
473  );
474  }
475 
476  unique_hangs++;
477  last_hang_time = Util::GetCurTimeMs();
478  break;
479 
480  case PUTExitReasonType::FAULT_CRASH:
481  keep_as_crash:
482 
483  /* This is handled in a manner roughly similar to timeouts,
484  except for slightly different limits and no need to re-run test
485  cases. */
486 
487  total_crashes++;
488 
489  if (unique_crashes >= afl::option::GetKeepUniqueCrash(*this)) {
490  // unlike FAULT_TMOUT case, keeping can be true when "crash mode" is enabled
491  return keeping;
492  }
493 
494  if (!setting->dumb_mode) {
495  if constexpr (sizeof(size_t) == 8) {
496  inp_feed.ModifyMemoryWithFunc(
497  [](u8* trace_bits, u32 /* map_size */) {
498  afl::util::SimplifyTrace<u64>((u64*)trace_bits, afl::option::GetMapSize<Tag>());
499  }
500  );
501  } else {
502  inp_feed.ModifyMemoryWithFunc(
503  [](u8* trace_bits, u32 /* map_size */) {
504  afl::util::SimplifyTrace<u32>((u32*)trace_bits, afl::option::GetMapSize<Tag>());
505  }
506  );
507  }
508 
509  u8 res;
510  inp_feed.ShowMemoryToFunc(
511  [this, &res](const u8* trace_bits, u32 /* map_size */) {
512  res = HasNewBits(trace_bits, &virgin_crash[0], afl::option::GetMapSize<Tag>());
513  }
514  );
515 
516  if (!res) {
517  // unlike FAULT_TMOUT case, keeping can be true when "crash mode" is enabled
518  return keeping;
519  }
520  }
521 
522 #if 0
523  if (!unique_crashes) WriteCrashReadme(); // FIXME?
524 #endif
525 
526  if (!setting->simple_files) {
527  fn = Util::StrPrintf("%s/crashes/id:%06llu,sig:%02u,%s.js",
528  setting->out_dir.c_str(),
529  unique_crashes,
530  exit_status.signal,
531  afl::routine::update::DescribeOp(*this, 0).c_str()
532  );
533  } else {
534  fn = Util::StrPrintf("%s/hangs/id_%06llu_%02u.js",
535  setting->out_dir.c_str(),
536  unique_crashes,
537  exit_status.signal
538  );
539  }
540 
541  unique_crashes++;
542 
543  last_crash_time = Util::GetCurTimeMs();
544  last_crash_execs = total_execs;
545 
546  break;
547 
548  case PUTExitReasonType::FAULT_ERROR:
549  ERROR("Unable to execute target application");
550 
551  default:
552  return keeping;
553  }
554 
555  /* If we're here, we apparently want to save the crash or hang
556  test case, too. */
557  int fd;
558 
559  fd = Util::OpenFile(fn, O_WRONLY | O_CREAT | O_EXCL, 0600);
560  Util::WriteFile(fd, buf_js, len_js);
561  Util::CloseFile(fd);
562 
563  // Save type file
564  fd = Util::OpenFile(fn + ".t", O_WRONLY | O_CREAT | O_EXCL, 0600);
565  Util::WriteFile(fd, buf_ty, len_ty);
566  Util::CloseFile(fd);
567 
568  return keeping;
569 }

575 void DIEState::ShowStats(void) {
576  // Lots of constants appear, so overlook this.
577  using namespace afl::option;
578 
579  const u32 MAP_SIZE = GetMapSize<Tag>();
580 
581  u64 cur_ms = Util::GetCurTimeMs();
582 
583  /* If not enough time has passed since last UI update, bail out. */
584  if (cur_ms - last_ms < 1000 / GetUiTargetHz(*this)) return;
585 
586  /* Check if we're past the 10 minute mark. */
587  if (cur_ms - start_time > 10 * 60 * 1000) run_over10m = true;
588 
589  /* Calculate smoothed exec speed stats. */
590  if (last_execs == 0) {
591  avg_exec = ((double)total_execs) * 1000 / (cur_ms - start_time);
592  } else {
593  double cur_avg = ((double)(total_execs - last_execs)) * 1000 /
594  (cur_ms - last_ms);
595 
596  /* If there is a dramatic (5x+) jump in speed, reset the indicator
597  more quickly. */
598  if (cur_avg * 5 < avg_exec || cur_avg / 5 > avg_exec)
599  avg_exec = cur_avg;
600 
601  avg_exec = avg_exec * (1.0 - 1.0 / GetAvgSmoothing(*this)) +
602  cur_avg * (1.0 / GetAvgSmoothing(*this));
603  }
604 
605  last_ms = cur_ms;
606  last_execs = total_execs;
607 
608  /* Tell the callers when to contact us (as measured in execs). */
609  stats_update_freq = avg_exec / (GetUiTargetHz(*this) * 10);
610  if (!stats_update_freq) stats_update_freq = 1;
611 
612  /* Do some bitmap stats. */
613  u32 t_bytes = Util::CountNon255Bytes(&virgin_bits[0], virgin_bits.size());
614  double t_byte_ratio = ((double)t_bytes * 100) / MAP_SIZE;
615 
616  double stab_ratio;
617  if (t_bytes)
618  stab_ratio = 100 - ((double)var_byte_count) * 100 / t_bytes;
619  else
620  stab_ratio = 100;
621 
622  /* Roughly every minute, update fuzzer stats and save auto tokens. */
623  if (cur_ms - last_stats_ms > GetStatsUpdateSec(*this) * 1000) {
624  last_stats_ms = cur_ms;
625 
626  WriteStatsFile(t_byte_ratio, stab_ratio, avg_exec);
627  SaveAuto();
628  WriteBitmap();
629  }
630 
631  /* Every now and then, write plot data. */
632  if (cur_ms - last_plot_ms > GetPlotUpdateSec(*this) * 1000) {
633  last_plot_ms = cur_ms;
634  MaybeUpdatePlotFile(t_byte_ratio, avg_exec);
635  }
636 
637  /* Honor AFL_EXIT_WHEN_DONE and AFL_BENCH_UNTIL_CRASH. */
638  if (!setting->dumb_mode && cycles_wo_finds > 100 && !pending_not_fuzzed &&
639  getenv("AFL_EXIT_WHEN_DONE")) stop_soon = 2;
640 
641  if (total_crashes && getenv("AFL_BENCH_UNTIL_CRASH")) stop_soon = 2;
642 
643  /* If we're not on TTY, bail out. */
644  if (not_on_tty) return;
645 
646  /* Compute some mildly useful bitmap stats. */
647  u32 t_bits = (MAP_SIZE << 3) - Util::CountBits(&virgin_bits[0], virgin_bits.size());
648 
649  /* Now, for the visuals... */
650  bool term_too_small = false;
651  if (clear_screen) {
652  MSG(TERM_CLEAR CURSOR_HIDE);
653  clear_screen = false;
654 
655  term_too_small = afl::CheckTermSize();
656  }
657 
658  MSG(TERM_HOME);
659 
660  if (term_too_small) {
661  MSG(cBRI "Your terminal is too small to display the UI.\n"
662  "Please resize terminal window to at least 80x25.\n" cRST);
663 
664  return;
665  }
666 
667  // FIXME: use stringstream...?
668  // anyways it's weird to mix up sprintf and std::string
669 
670  /* Let's start by drawing a centered banner. */
671  u32 banner_len = (crash_mode != PUTExitReasonType::FAULT_NONE ? 24 : 22)
672  + strlen(GetVersion(*this)) + use_banner.size();
673  u32 banner_pad = (80 - banner_len) / 2;
674 
675  auto fuzzer_name = crash_mode != PUTExitReasonType::FAULT_NONE
676  ? cPIN "fuzzuf peruvian were-rabbit"
677  : cYEL "fuzzuf american fuzzy lop";
678 
679  std::string tmp(banner_pad, ' ');
680  tmp += Util::StrPrintf("%s " cLCY "%s" cLGN " (%s)",
681  fuzzer_name, GetVersion(*this), use_banner.c_str());
682  MSG("\n%s\n\n", tmp.c_str());
683 
684  /* Lord, forgive me this. */
685  MSG(SET_G1 bSTG bLT bH bSTOP cCYA " process timing " bSTG bH30 bH5 bH2 bHB
686  bH bSTOP cCYA " overall results " bSTG bH5 bRT "\n");
687 
688  std::string col;
689  if (setting->dumb_mode) {
690  col = cRST;
