38 FuzzerArgs ParseGlobalOptionsForFuzzer(GlobalArgs &global_args, GlobalFuzzerOptions &global_options) {
39  // Parse a sub-command
40  po::positional_options_description subcommand;
41  subcommand.add("fuzzer", 1);
42  subcommand.add("fargs", -1);
43 
44  // Allocate variables to heap since `global_desc` outlives from this function
45 
46  // Define global options
47  po::options_description global_desc("Global options");
48  global_desc.add_options()
49  ("fuzzer",
50  po::value<std::string>(&global_options.fuzzer),
51  "Specify fuzzer to be used in your fuzzing campaign.")
52  ("help",
53  po::bool_switch(&global_options.help),
54  "Produce help message.")
55  ("in_dir,i",
56  po::value<std::string>(&global_options.in_dir),
57  "Set seed dir. Default is `./seeds`.")
58  ("out_dir,o",
59  po::value<std::string>(&global_options.out_dir),
60  "Set output dir. Default is `/tmp/fuzzuf-out_dir`.")
61  ("exec_timelimit_ms",
62  global_options.exec_timelimit_ms ?
63  po::value<u32>()->default_value(*global_options.exec_timelimit_ms):
64  po::value<u32>(),
65  "Limit execution time of PUT. Unit is milli-seconds.")
66  ("exec_memlimit",
67  global_options.exec_memlimit ?
68  po::value<u32>()->default_value(*global_options.exec_memlimit):
69  po::value<u32>(),
70  "Limit memory usage for PUT execution.")
71  ("log_file",
72  global_options.log_file ?
73  po::value<std::string>()->default_value(global_options.log_file->string()):
74  po::value<std::string>()->default_value(""),
75  "Enable LogFile logger and set the log file path for LogFile logger")
76  ;
77 
78  // Dummy options to parse global options but not PUT options
79  // NOTE: PUT options are parsed at fuzzer builder
80  po::options_description fargs("Fuzzer options");
81  fargs.add_options()
82  ("fargs", po::value<std::vector<std::string>>(), "Specify Fuzzer options and PUT args.") // pargs is not captured while parsing global options
83  ;
84 
85  // Obtain global fuzzing campaign settings from the command line
86  po::variables_map vm;
87  po::store(
88  po::command_line_parser(global_args.argc, global_args.argv)
89  .options(global_desc.add(fargs)).positional(subcommand).allow_unregistered().run(),
90  vm
91  );
92  po::notify(vm);
93 
94  // Show a usage and exit because none of fuzzers are specified
95  // TODO: Provide better help message
96  if (vm.count("fuzzer") == 0) {
97  if (global_options.help) {
98  std::cout << "fuzzuf" << std::endl;
99  std::cout << global_desc << std::endl;
100  exit(0);
101  } else {
102  throw exceptions::cli_error("`fuzzer` is not specified in command line. Run with `--help` to check usage", __FILE__, __LINE__);
103  }
104  }
105  DEBUG("[*] global_options.fuzzer = %s", global_options.fuzzer.c_str());
106 
107  // Store values to `global_options` manually
108  // since type T = { std::optional, fs::path, Logger (enum) }, is not cpmatible with po::value<T>()
109  if(vm.count("exec_timelimit_ms")) {
110  global_options.exec_timelimit_ms = vm["exec_timelimit_ms"].as<u32>();
111  }
112  if(vm.count("exec_memlimit")) {
113  global_options.exec_memlimit = vm["exec_memlimit"].as<u32>();
114  }
115  auto log_file = vm["log_file"].as<std::string>();
116  if (!log_file.empty()) {
117  global_options.log_file = fs::path(std::move(log_file));
118  global_options.logger = Logger::LogFile;
119  }
120 
121  return FuzzerArgs {
122  .argc = global_args.argc,
123  .argv = global_args.argv,
124  .global_options_description = global_desc
125  };
126 }
