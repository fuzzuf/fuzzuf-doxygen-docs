82  namespace phx = boost::phoenix;

84  escape = (qi::lit("\\\\")[qi::_val = '\\']) |

91  name = qi::as_string[+(qi::standard::graph - '@' - '=' - '"' - '#')];
92  comment = ('#' >> *(qi::standard::blank |

96  name = qi::as_string[+(qi::byte_ - '@' - '=' - '"' - '#' -
97  ' ' - qi::eol )];
98  comment = ('#' >> *(qi::byte_ - qi::eol))[qi::_pass = true];
99  }

102  qi::omit[qi::lit('"')] >> escaped_text >> qi::omit[qi::lit('"')];
103  root =
104  qi::skip(qi::standard::blank)
105  [(comment[qi::_pass = true]) |
106  ((quoted_text >> qi::omit[*qi::standard::blank >> -comment])
107  [qi::_pass =
108  phx::bind(&AFLDictRule::without_level, qi::_val,
109  std::string("(no name)"), qi::_1, eout)]) |
110  ((name >> '@' >> qi::uint_ >> '=' >> quoted_text >>
111  qi::omit[*qi::standard::blank >> -comment])

114  qi::_1, qi::_2, filter, qi::_3, eout)]) |
115  ((name >> '=' >> quoted_text >>
116  qi::omit[*qi::standard::blank >> -comment])
117  [qi::_pass = phx::bind(&AFLDictRule::without_level,
118  qi::_val, qi::_1, qi::_2, eout)]) |
119  ((*qi::standard::blank)[qi::_pass = true])] %

163 template <typename T>
164 auto LoadAFLDictionary(const std::string &filename_, T &dest, bool strict,
165  const std::function<void(std::string &&)> &eout)
166  -> utils::void_t<DictionaryWordT<T>> {
167 
168  namespace qi = boost::spirit::qi;
169 
170  unsigned int level = 0u;
171  std::string filename;
172 
173  {
174  fs::path path(filename_);
175  const std::string leaf = path.filename().string();
176  boost::fusion::vector<std::string, unsigned int> parsed;
177  auto iter = leaf.begin();
178  const auto end = leaf.end();
179  if (!qi::parse(iter, end,
180  qi::as_string[*(qi::char_ - '@')] >> '@' >> qi::uint_,
181  parsed) ||
182  iter != end)
183  filename = filename_;
184  else {
185  filename =
186  (path.remove_filename() / fs::path(boost::fusion::at_c<0>(parsed)))
187  .string();
188  level = boost::fusion::at_c<1>(parsed);
189  }
190  }
191 
192  auto mapped_file = utils::map_file(filename, O_RDONLY, true);
193  const AFLDictRule<uint8_t *, T> rule(level, strict, eout);
194  auto iter = mapped_file.begin().get();
195  const auto end = mapped_file.end().get();
196 #if BOOST_VERSION >= 107200
197 
206  if( strict ) {
207  if( std::find_if(
208  iter,
209  end,
210  []( char ch ) {
211  return !boost::spirit::char_encoding::standard::strict_ischar( ch );
212  }
213  ) != end )
214  throw exceptions::invalid_file("invalid dictionary file", __FILE__,
215  __LINE__);
216  }
217 #endif
218  T temp;
219  if (!qi::parse(iter, end, rule, temp) || iter != end)
220  throw exceptions::invalid_file("invalid dictionary file", __FILE__,
221  __LINE__);

224 }
