39 void InputInfo::updateEnergy(
40  std::size_t global_number_of_features, bool scale_per_exec_time,
41  std::chrono::microseconds average_unit_execution_time) {
42  enabled = true;
43  energy = 0.0;
44  sum_incidence = 0.0;
45 
46  // Apply add-one smoothing to locally discovered features.
47 #pragma GCC diagnostic push
48 #pragma GCC diagnostic ignored "-Wunused-variable"
49  for (const auto &[id, incidence] : feature_freqs) {
50 #pragma GCC diagnostic pop
51  double local_incidence = incidence + 1;
52  energy -= local_incidence * std::log(local_incidence);
53  sum_incidence += local_incidence;
54  }
55 
56  // Apply add-one smoothing to locally undiscovered features.
57  // PreciseEnergy -= 0; // since log(1.0) == 0)
58  sum_incidence += static_cast<double>(global_number_of_features) -
59  static_cast<double>(feature_freqs.size());
60 
61  // Add a single locally abundant feature apply add-one smoothing.
62  const auto abd_incidence = static_cast<double>(executed_mutations_count + 1);
63  energy -= abd_incidence * log(abd_incidence);
64  sum_incidence += abd_incidence;
65 
66  // Normalize.
67  if (sum_incidence != 0) {
68  energy = energy / sum_incidence + std::log(sum_incidence);
69  }
70 
71  if (scale_per_exec_time) {
72  // Scaling to favor inputs with lower execution time.
73  // NOLINTBEGIN(cppcoreguidelines-avoid-magic-numbers,readability-magic-numbers)
74  uint32_t perf_score = 100;
75  if (time_of_unit.count() > average_unit_execution_time.count() * 10) {
76  perf_score = 10;
77  } else if (time_of_unit.count() > average_unit_execution_time.count() * 4) {
78  perf_score = 25;
79  } else if (time_of_unit.count() > average_unit_execution_time.count() * 2) {
80  perf_score = 50;
81  } else if (time_of_unit.count() * 3 >
82  average_unit_execution_time.count() * 4) {
83  perf_score = 75;
84  } else if (time_of_unit.count() * 4 < average_unit_execution_time.count()) {
85  perf_score = 300;
86  } else if (time_of_unit.count() * 3 < average_unit_execution_time.count()) {
87  perf_score = 200;
88  } else if (time_of_unit.count() * 2 < average_unit_execution_time.count()) {
89  perf_score = 150;
90  }
91  // NOLINTEND(cppcoreguidelines-avoid-magic-numbers,readability-magic-numbers)
92 
93  energy *= perf_score;
94  }
95 }

103 auto InputInfo::delete_feature_freq(std::uint32_t index) -> bool {
104  if (feature_freqs.empty()) {
105  return false;
106  }
107 
108  // Binary search over local feature frequencies
109  // sorted by index.
110  auto lower =
111  std::lower_bound(feature_freqs.begin(), feature_freqs.end(),
112  std::pair<std::uint32_t, std::uint16_t>(index, 0U));
113 
114  if (lower != feature_freqs.end() && lower->first == index) {
115  feature_freqs.erase(lower);
116  return true;
117  }
118  return false;
119 }

126 void InputInfo::updateFeatureFrequency(std::uint32_t index) {
127  needs_energy_update = true;
128  // The local feature frequencies is an ordered vector of pairs.
129  // If there are no local feature frequencies, push_back preserves order.
130  // Set the feature frequency for feature Idx32 to 1.
131  if (feature_freqs.empty()) {
132  feature_freqs.emplace_back(index, 1U);
133  return;
134  }
135 
136  // Binary search over local feature frequencies sorted by index.
137  const auto lower =
138  std::lower_bound(feature_freqs.begin(), feature_freqs.end(),
139  std::pair<std::uint32_t, std::uint16_t>(index, 0U));
140 
141  // If feature Idx32 already exists, increment its frequency.
142  // Otherwise, insert a new pair right after the next lower index.
143  if (lower != feature_freqs.end() && lower->first == index) {
144  lower->second++;
145  } else {
146  feature_freqs.insert(lower,
147  std::pair<std::uint32_t, std::uint16_t>(index, 1U));
148  }
149 }
