54  void Setup(void) {
55  if (map_size == 0) return;
56  shmid = shmget(IPC_PRIVATE, map_size, IPC_CREAT | IPC_EXCL | 0600);
57  if (shmid < 0) ERROR("shmget() failed");
58 
59  trace_bits = (u8 *)shmat(shmid, nullptr, 0);
60  if (trace_bits == (u8 *)-1) ERROR("shmat() failed");
61  }
62 
63  void Reset(void) {
64  if (map_size == 0) return;
65  if (trace_bits != nullptr) {
66  std::memset(trace_bits, 0, map_size);
67  }
68 
69  MEM_BARRIER();
70  }
71 
72  void Erase(void) {
73  if (map_size == 0) return;
74  if (trace_bits != nullptr) {
75  if (shmdt(trace_bits) == -1) ERROR("shmdt() failed");
76  trace_bits = nullptr;
77  }
78  if (shmid != INVALID_SHMID) {
79  if (shmctl(shmid, IPC_RMID, 0) == -1) ERROR("shmctl() failed");
80  shmid = INVALID_SHMID;
81  }
82  }
83 
84  void SetupEnvironmentVariable(const char *shm_env_var) {
85  if (map_size > 0 && shmid != INVALID_SHMID) {
86  std::string shmstr = std::to_string(shmid);
87  setenv(shm_env_var, shmstr.c_str(), 1);
88  } else {
89  unsetenv(shm_env_var);
90  }
91  }
