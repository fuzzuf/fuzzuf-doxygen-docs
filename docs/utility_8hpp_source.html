182 template <class IReturn, class... IArgs, class OReturn, class... OArgs>
183 class FinalizeRoutine<IReturn(IArgs...), OReturn(OArgs...)>
184  : public HierarFlowRoutine<IReturn(IArgs...), IReturn(IArgs...)> {
185  using I = IReturn(IArgs...);
186  using O = OReturn(OArgs...);
187 
188  template <class A, class B>
189  friend HierarFlowNode<A, A> Finally(HierarFlowNode<A, B> node);
190 
191  public:
192  utils::NullableRef<HierarFlowCallee<IReturn(IArgs...)>> operator()(
193  IArgs... args) {
194  // FIXME: this should be updated as with `HierarFlowRoutine::CallSuccessors`
195  // when backward compatibility becomes unnecessary. For the details, see the
196  // comment in `HierarFlowRoutine::CallSuccessors`.
197 
198  utils::NullableRef<HierarFlowCallee<I>> succ_ref = std::nullopt;
199  auto& node = this->UnwrapCurrentLinkedNodeRef();
200  auto& succ_nodes = node.succ_nodes;
201 
202  if (!succ_nodes.empty()) {
203  succ_ref = *succ_nodes[0];
204  }
205 
206  // This is exactly the same as HierarFlowRoutine::CallSuccessors
207  while (succ_ref) {
208  auto& succ = succ_ref.value().get();
209  auto next_succ_ref = succ(args...);
210  succ_ref.swap(next_succ_ref);
211  }
212 
213  // Call the special node after all the ordinary successors have ended.
214  HierarFlowCallee<I>& final_ref = *final_node_impl;
215  final_ref(args...);
216 
217  if constexpr (!std::is_same_v<IReturn, void>) {
218  // To avoid unnecessary copy, use std::move.
219  node.SetParentResponseValue(std::move(node.resp_val));
220  }
221 
222  return this->GoToDefaultNext();
223  }
224 
225  private:
226  // Exposing this class to public is very dangerous because this routine has a
227  // HierarFlowNode instance.
228  FinalizeRoutine(std::shared_ptr<HierarFlowNodeImpl<I, O>> impl)
229  : final_node_impl(impl) {}
230 
231  // For the same reason, prohibit copy constructors(,
232  // HierarFlowNodeImpl::ShareRoutine and HierarFlowNode::HardLink).
233  FinalizeRoutine(const FinalizeRoutine&) = delete;
234  FinalizeRoutine& operator=(const FinalizeRoutine&) = delete;
235 
236  std::shared_ptr<HierarFlowNodeImpl<I, O>> final_node_impl;
237 };

249 template <class I, class O>
250 HierarFlowNode<I, I> Finally(HierarFlowNode<I, O> node) {
251  auto impl = node.ShareImpl();
252  auto ret = HierarFlowNode<I, I>(std::shared_ptr<HierarFlowRoutine<I, I>>(
253  new FinalizeRoutine<I, O>(impl)));
254 
255  // We don't connect ret and node by `ret << node`.
256  // This is because we don't want node to appear in the successors of ret.
257  // So we just tell node that ret is the parent of it instead.
258  impl->SetParentAndIndex(ret.ShareImpl().get(), 0);
